41  /**
42  * Collection future.
43  */
44  @GwtCompatible(emulated = true)
45  final class CollectionFuture<V, C> extends AbstractFuture.TrustedFuture<C> {
46    private static final Logger logger =
47        Logger.getLogger(CollectionFuture.class.getName());
48  
49    @SuppressWarnings({"rawtypes", "unchecked"})
50    private static final AtomicReferenceFieldUpdater<CollectionFuture<?, ?>, Set<Throwable>>
51        SEEN_EXCEPTIONS_UDPATER = newUpdater(
52            (Class) CollectionFuture.class, (Class) Set.class, "seenExceptions");
53  
54    private ImmutableCollection<? extends ListenableFuture<? extends V>> futures;
55    private final boolean allMustSucceed;
56    private final AtomicInteger remaining;
57    private FutureCollector<V, C> combiner;
58    private List<Optional<V>> values;
59    private volatile Set<Throwable> seenExceptions;
60  
61    CollectionFuture(
62        ImmutableCollection<? extends ListenableFuture<? extends V>> futures,
63        boolean allMustSucceed, Executor listenerExecutor,
64        FutureCollector<V, C> combiner) {
65      Preconditions.checkNotNull(futures);
66      Preconditions.checkNotNull(listenerExecutor);
67      Preconditions.checkNotNull(combiner);
68      this.futures = futures;
69      this.allMustSucceed = allMustSucceed;
70      this.remaining = new AtomicInteger(futures.size());
71      this.combiner = combiner;
72      this.values = Lists.newArrayListWithCapacity(futures.size());
73      this.seenExceptions = null; // Initialized once the first time we see an exception
74      init(listenerExecutor);
75    }
76  
77    @Override void done() {
78      // Let go of the memory held by other futures
79      this.futures = null;
80  
81      // By now the values array has either been set as the Future's value,
82      // or (in case of failure) is no longer useful.
83      this.values = null;
84  
85      // The combiner may also hold state, so free that as well
86      this.combiner = null;
87    }
88  
89    /**
90     * Must be called at the end of the constructor.
91     */
92    protected void init(final Executor listenerExecutor) {
93      // Now begin the "real" initialization.
94  
95      // Corner case: List is empty.
96      if (futures.isEmpty()) {
97        set(combiner.combine(ImmutableList.<Optional<V>>of()));
98        return;
99      }
100 
101     // Populate the results list with null initially.
102     for (int i = 0; i < futures.size(); ++i) {
103       values.add(null);
104     }
105 
106     // Register a listener on each Future in the list to update
107     // the state of this future.
108     // Note that if all the futures on the list are done prior to completing
109     // this loop, the last call to addListener() will callback to
110     // setOneValue(), transitively call our cleanup listener, and set
111     // this.futures to null.
112     // This is not actually a problem, since the foreach only needs
113     // this.futures to be non-null at the beginning of the loop.
114     int i = 0;
115     for (final ListenableFuture<? extends V> listenable : futures) {
116       final int index = i++;
117       listenable.addListener(new Runnable() {
118         @Override
119         public void run() {
120           setOneValue(index, listenable);
121         }
122       }, listenerExecutor);
123     }
124   }
125 
126   @Override
127   public boolean cancel(boolean mayInterruptIfRunning) {
128     // Cancel all the component futures.
129     ImmutableCollection<? extends ListenableFuture<?>> futuresToCancel = futures;
130     boolean cancelled = super.cancel(mayInterruptIfRunning);
131     if (cancelled && futuresToCancel != null) {
132       for (ListenableFuture<?> future : futuresToCancel) {
133         future.cancel(mayInterruptIfRunning);
134       }
135     }
136     return cancelled;
137   }
138 
139   /**
140    * Fails this future with the given Throwable if {@link #allMustSucceed} is
141    * true. Also, logs the throwable if it is an {@link Error} or if
142    * {@link #allMustSucceed} is {@code true}, the throwable did not cause
143    * this future to fail, and it is the first time we've seen that particular Throwable.
144    */
145   private void setExceptionAndMaybeLog(Throwable throwable) {
146     boolean visibleFromOutputFuture = false;
147     boolean firstTimeSeeingThisException = true;
148     if (allMustSucceed) {
149       // As soon as the first one fails, throw the exception up.
150       // The result of all other inputs is then ignored.
151       visibleFromOutputFuture = super.setException(throwable);
152 
153       // seenExceptions is only set once; but we don't allocate it until we get a failure
154       Set<Throwable> seenExceptionsLocal = seenExceptions;
155       if (seenExceptionsLocal == null) {
156         SEEN_EXCEPTIONS_UDPATER.compareAndSet(this, null, Sets.<Throwable>newConcurrentHashSet());
157         seenExceptionsLocal = seenExceptions;
158       }
159 
160       // Go up the causal chain to see if we've already seen this cause; if we have,
161       // even if it's wrapped by a different exception, don't log it.
162       Throwable currentThrowable = throwable;
163       while (currentThrowable != null) {
164         firstTimeSeeingThisException = seenExceptionsLocal.add(currentThrowable);
165         if (!firstTimeSeeingThisException) {
166           break;
167         }
168         currentThrowable = currentThrowable.getCause();
169       }
170     }
171 
172     if (throwable instanceof Error
173         || (allMustSucceed && !visibleFromOutputFuture && firstTimeSeeingThisException)) {
174       logger.log(Level.SEVERE, "input future failed.", throwable);
175     }
176   }
177 
178   /**
179    * Sets the value at the given index to that of the given future.
180    */
181   private void setOneValue(int index, Future<? extends V> future) {
182     List<Optional<V>> localValues = values;
183     if (isDone() || localValues == null) {
184       // Some other future failed or has been cancelled, causing this one to
185       // also be cancelled or have an exception set. This should only happen
186       // if allMustSucceed is true or if the output itself has been
187       // cancelled.
188       checkState(allMustSucceed || isCancelled(),
189           "Future was done before all dependencies completed");
190     }
191 
192     try {
193       checkState(future.isDone(),
194           "Tried to set value from future which is not done");
195       if (future.isCancelled()) {
196         if (allMustSucceed) {
197           // this.cancel propagates the cancellation to children; we use super.cancel
198           // to set our own state but let the input futures keep running
199           // as some of them may be used elsewhere.
200           super.cancel(false);
201         }
202       } else {
203         V returnValue = getUninterruptibly(future);
204         if (localValues != null) {
205           localValues.set(index, Optional.fromNullable(returnValue));
206         }
207       }
208     } catch (ExecutionException e) {
209       setExceptionAndMaybeLog(e.getCause());
210     } catch (Throwable t) {
211       setExceptionAndMaybeLog(t);
212     } finally {
213       int newRemaining = remaining.decrementAndGet();
214       checkState(newRemaining >= 0, "Less than 0 remaining futures");
215       if (newRemaining == 0) {
216         FutureCollector<V, C> localCombiner = combiner;
217         if (localCombiner != null && localValues != null) {
218           set(localCombiner.combine(localValues));
219         } else {
220           checkState(isDone());
221         }
222 
223         seenExceptions = null; // Done with tracking seen exceptions
224       }
225     }
226   }

228   interface FutureCollector<V, C> {
229     C combine(List<Optional<V>> values);
230   }
