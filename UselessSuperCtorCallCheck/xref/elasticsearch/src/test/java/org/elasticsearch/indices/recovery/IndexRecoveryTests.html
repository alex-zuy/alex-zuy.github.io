68      private void assertRecoveryStateWithoutStage(RecoveryState state, int shardId, Type type,
69                                                   String sourceNode, String targetNode, boolean hasRestoreSource) {
70          assertThat(state.getShardId().getId(), equalTo(shardId));
71          assertThat(state.getType(), equalTo(type));
72          if (sourceNode == null) {
73              assertNull(state.getSourceNode());
74          } else {
75              assertNotNull(state.getSourceNode());
76              assertThat(state.getSourceNode().getName(), equalTo(sourceNode));
77          }
78          if (targetNode == null) {
79              assertNull(state.getTargetNode());
80          } else {
81              assertNotNull(state.getTargetNode());
82              assertThat(state.getTargetNode().getName(), equalTo(targetNode));
83          }
84          if (hasRestoreSource) {
85              assertNotNull(state.getRestoreSource());
86          } else {
87              assertNull(state.getRestoreSource());
88          }
89  
90      }
91  
92      private void assertRecoveryState(RecoveryState state, int shardId, Type type, Stage stage,
93                                       String sourceNode, String targetNode, boolean hasRestoreSource) {
94          assertRecoveryStateWithoutStage(state, shardId, type, sourceNode, targetNode, hasRestoreSource);
95          assertThat(state.getStage(), equalTo(stage));
96      }
97  
98      private void assertOnGoingRecoveryState(RecoveryState state, int shardId, Type type,
99                                              String sourceNode, String targetNode, boolean hasRestoreSource) {
100         assertRecoveryStateWithoutStage(state, shardId, type, sourceNode, targetNode, hasRestoreSource);
101         assertThat(state.getStage(), not(equalTo(Stage.DONE)));
102     }

104     private void slowDownRecovery(ByteSizeValue shardSize) {
105         long chunkSize = shardSize.bytes() / 10;
106         assertTrue(client().admin().cluster().prepareUpdateSettings()
107                 .setTransientSettings(ImmutableSettings.builder()
108                                 // one chunk per sec..
109                                 .put(RecoverySettings.INDICES_RECOVERY_MAX_BYTES_PER_SEC, chunkSize)
110                                 .put(RecoverySettings.INDICES_RECOVERY_FILE_CHUNK_SIZE, chunkSize)
111                 )
112                 .get().isAcknowledged());
113     }
114 
115     private void restoreRecoverySpeed() {
116         assertTrue(client().admin().cluster().prepareUpdateSettings()
117                 .setTransientSettings(ImmutableSettings.builder()
118                                 .put(RecoverySettings.INDICES_RECOVERY_MAX_BYTES_PER_SEC, "20mb")
119                                 .put(RecoverySettings.INDICES_RECOVERY_FILE_CHUNK_SIZE, "512kb")
120                 )
121                 .get().isAcknowledged());
122     }

169     @Test
170     public void replicaRecoveryTest() throws Exception {
171         logger.info("--> start node A");
172         String nodeA = internalCluster().startNode(settingsBuilder().put("gateway.type", "local"));
173 
174         logger.info("--> create index on node: {}", nodeA);
175         createAndPopulateIndex(INDEX_NAME, 1, SHARD_COUNT, REPLICA_COUNT);
176 
177         logger.info("--> start node B");
178         String nodeB = internalCluster().startNode(settingsBuilder().put("gateway.type", "local"));
179         ensureGreen();
180 
181         // force a shard recovery from nodeA to nodeB
182         logger.info("--> bump replica count");
183         client().admin().indices().prepareUpdateSettings(INDEX_NAME)
184                 .setSettings(settingsBuilder().put("number_of_replicas", 1)).execute().actionGet();
185         ensureGreen();
186 
187         logger.info("--> request recoveries");
188         RecoveryResponse response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
189 
190         // we should now have two total shards, one primary and one replica
191         List<ShardRecoveryResponse> shardResponses = response.shardResponses().get(INDEX_NAME);
192         assertThat(shardResponses.size(), equalTo(2));
193 
194         List<ShardRecoveryResponse> nodeAResponses = findRecoveriesForTargetNode(nodeA, shardResponses);
195         assertThat(nodeAResponses.size(), equalTo(1));
196         List<ShardRecoveryResponse> nodeBResponses = findRecoveriesForTargetNode(nodeB, shardResponses);
197         assertThat(nodeBResponses.size(), equalTo(1));
198 
199         // validate node A recovery
200         ShardRecoveryResponse nodeAShardResponse = nodeAResponses.get(0);
201         assertRecoveryState(nodeAShardResponse.recoveryState(), 0, Type.GATEWAY, Stage.DONE, nodeA, nodeA, false);
202         validateIndexRecoveryState(nodeAShardResponse.recoveryState().getIndex());
203 
204         // validate node B recovery
205         ShardRecoveryResponse nodeBShardResponse = nodeBResponses.get(0);
206         assertRecoveryState(nodeBShardResponse.recoveryState(), 0, Type.REPLICA, Stage.DONE, nodeA, nodeB, false);
207         validateIndexRecoveryState(nodeBShardResponse.recoveryState().getIndex());
208     }

210     @Test
211     @TestLogging("indices.recovery:TRACE")
212     public void rerouteRecoveryTest() throws Exception {
213         logger.info("--> start node A");
214         String nodeA = internalCluster().startNode(settingsBuilder().put("gateway.type", "local"));
215 
216         logger.info("--> create index on node: {}", nodeA);
217         ByteSizeValue shardSize = createAndPopulateIndex(INDEX_NAME, 1, SHARD_COUNT, REPLICA_COUNT).getShards()[0].getStats().getStore().size();
218 
219         logger.info("--> start node B");
220         String nodeB = internalCluster().startNode(settingsBuilder().put("gateway.type", "local"));
221 
222         ensureGreen();
223 
224         logger.info("--> slowing down recoveries");
225         slowDownRecovery(shardSize);
226 
227         logger.info("--> move shard from: {} to: {}", nodeA, nodeB);
228         client().admin().cluster().prepareReroute()
229                 .add(new MoveAllocationCommand(new ShardId(INDEX_NAME, 0), nodeA, nodeB))
230                 .execute().actionGet().getState();
231 
232 
233         logger.info("--> request recoveries");
234         RecoveryResponse response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
235 
236         List<ShardRecoveryResponse> shardResponses = response.shardResponses().get(INDEX_NAME);
237         List<ShardRecoveryResponse> nodeAResponses = findRecoveriesForTargetNode(nodeA, shardResponses);
238         assertThat(nodeAResponses.size(), equalTo(1));
239         List<ShardRecoveryResponse> nodeBResponses = findRecoveriesForTargetNode(nodeB, shardResponses);
240         assertThat(nodeBResponses.size(), equalTo(1));
241 
242         assertRecoveryState(nodeAResponses.get(0).recoveryState(), 0, Type.GATEWAY, Stage.DONE, nodeA, nodeA, false);
243         validateIndexRecoveryState(nodeAResponses.get(0).recoveryState().getIndex());
244 
245         assertOnGoingRecoveryState(nodeBResponses.get(0).recoveryState(), 0, Type.RELOCATION, nodeA, nodeB, false);
246         validateIndexRecoveryState(nodeBResponses.get(0).recoveryState().getIndex());
247 
248         logger.info("--> speeding up recoveries");
249         restoreRecoverySpeed();
250 
251         // wait for it to be finished
252         ensureGreen();
253 
254         response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
255 
256         shardResponses = response.shardResponses().get(INDEX_NAME);
257         assertThat(shardResponses.size(), equalTo(1));
258 
259         assertRecoveryState(shardResponses.get(0).recoveryState(), 0, Type.RELOCATION, Stage.DONE, nodeA, nodeB, false);
260         validateIndexRecoveryState(shardResponses.get(0).recoveryState().getIndex());
261 
262         logger.info("--> bump replica count");
263         client().admin().indices().prepareUpdateSettings(INDEX_NAME)
264                 .setSettings(settingsBuilder().put("number_of_replicas", 1)).execute().actionGet();
265         ensureGreen();
266 
267         logger.info("--> start node C");
268         String nodeC = internalCluster().startNode(settingsBuilder().put("gateway.type", "local"));
269         assertFalse(client().admin().cluster().prepareHealth().setWaitForNodes("3").get().isTimedOut());
270 
271         logger.info("--> slowing down recoveries");
272         slowDownRecovery(shardSize);
273 
274         logger.info("--> move replica shard from: {} to: {}", nodeA, nodeC);
275         client().admin().cluster().prepareReroute()
276                 .add(new MoveAllocationCommand(new ShardId(INDEX_NAME, 0), nodeA, nodeC))
277                 .execute().actionGet().getState();
278 
279         response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
280         shardResponses = response.shardResponses().get(INDEX_NAME);
281 
282         nodeAResponses = findRecoveriesForTargetNode(nodeA, shardResponses);
283         assertThat(nodeAResponses.size(), equalTo(1));
284         nodeBResponses = findRecoveriesForTargetNode(nodeB, shardResponses);
285         assertThat(nodeBResponses.size(), equalTo(1));
286         List<ShardRecoveryResponse> nodeCResponses = findRecoveriesForTargetNode(nodeC, shardResponses);
287         assertThat(nodeCResponses.size(), equalTo(1));
288 
289         assertRecoveryState(nodeAResponses.get(0).recoveryState(), 0, Type.REPLICA, Stage.DONE, nodeB, nodeA, false);
290         validateIndexRecoveryState(nodeAResponses.get(0).recoveryState().getIndex());
291 
292         assertRecoveryState(nodeBResponses.get(0).recoveryState(), 0, Type.RELOCATION, Stage.DONE, nodeA, nodeB, false);
293         validateIndexRecoveryState(nodeBResponses.get(0).recoveryState().getIndex());
294 
295         // relocations of replicas are marked as REPLICA and the source node is the node holding the primary (B)
296         assertOnGoingRecoveryState(nodeCResponses.get(0).recoveryState(), 0, Type.REPLICA, nodeB, nodeC, false);
297         validateIndexRecoveryState(nodeCResponses.get(0).recoveryState().getIndex());
298 
299         logger.info("--> speeding up recoveries");
300         restoreRecoverySpeed();
301         ensureGreen();
302 
303         response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
304         shardResponses = response.shardResponses().get(INDEX_NAME);
305 
306         nodeAResponses = findRecoveriesForTargetNode(nodeA, shardResponses);
307         assertThat(nodeAResponses.size(), equalTo(0));
308         nodeBResponses = findRecoveriesForTargetNode(nodeB, shardResponses);
309         assertThat(nodeBResponses.size(), equalTo(1));
310         nodeCResponses = findRecoveriesForTargetNode(nodeC, shardResponses);
311         assertThat(nodeCResponses.size(), equalTo(1));
312 
313         assertRecoveryState(nodeBResponses.get(0).recoveryState(), 0, Type.RELOCATION, Stage.DONE, nodeA, nodeB, false);
314         validateIndexRecoveryState(nodeBResponses.get(0).recoveryState().getIndex());
315 
316         // relocations of replicas are marked as REPLICA and the source node is the node holding the primary (B)
317         assertRecoveryState(nodeCResponses.get(0).recoveryState(), 0, Type.REPLICA, Stage.DONE, nodeB, nodeC, false);
318         validateIndexRecoveryState(nodeCResponses.get(0).recoveryState().getIndex());
319     }

321     @Test
322     public void snapshotRecoveryTest() throws Exception {
323         logger.info("--> start node A");
324         String nodeA = internalCluster().startNode(settingsBuilder().put("gateway.type", "local"));
325 
326         logger.info("--> create repository");
327         assertAcked(client().admin().cluster().preparePutRepository(REPO_NAME)
328                 .setType("fs").setSettings(ImmutableSettings.settingsBuilder()
329                                 .put("location", newTempDir(LifecycleScope.SUITE))
330                                 .put("compress", false)
331                 ).get());
332 
333         ensureGreen();
334 
335         logger.info("--> create index on node: {}", nodeA);
336         createAndPopulateIndex(INDEX_NAME, 1, SHARD_COUNT, REPLICA_COUNT);
337 
338         logger.info("--> snapshot");
339         CreateSnapshotResponse createSnapshotResponse = client().admin().cluster().prepareCreateSnapshot(REPO_NAME, SNAP_NAME)
340                 .setWaitForCompletion(true).setIndices(INDEX_NAME).get();
341         assertThat(createSnapshotResponse.getSnapshotInfo().successfulShards(), greaterThan(0));
342         assertThat(createSnapshotResponse.getSnapshotInfo().successfulShards(), equalTo(createSnapshotResponse.getSnapshotInfo().totalShards()));
343 
344         assertThat(client().admin().cluster().prepareGetSnapshots(REPO_NAME).setSnapshots(SNAP_NAME).get()
345                 .getSnapshots().get(0).state(), equalTo(SnapshotState.SUCCESS));
346 
347         client().admin().indices().prepareClose(INDEX_NAME).execute().actionGet();
348 
349         logger.info("--> restore");
350         RestoreSnapshotResponse restoreSnapshotResponse = client().admin().cluster()
351                 .prepareRestoreSnapshot(REPO_NAME, SNAP_NAME).setWaitForCompletion(true).execute().actionGet();
352         int totalShards = restoreSnapshotResponse.getRestoreInfo().totalShards();
353         assertThat(totalShards, greaterThan(0));
354 
355         ensureGreen();
356 
357         logger.info("--> request recoveries");
358         RecoveryResponse response = client().admin().indices().prepareRecoveries(INDEX_NAME).execute().actionGet();
359 
360         for (Map.Entry<String, List<ShardRecoveryResponse>> shardRecoveryResponse : response.shardResponses().entrySet()) {
361 
362             assertThat(shardRecoveryResponse.getKey(), equalTo(INDEX_NAME));
363             List<ShardRecoveryResponse> shardRecoveryResponses = shardRecoveryResponse.getValue();
364             assertThat(shardRecoveryResponses.size(), equalTo(totalShards));
365 
366             for (ShardRecoveryResponse shardResponse : shardRecoveryResponses) {
367                 assertRecoveryState(shardResponse.recoveryState(), 0, Type.SNAPSHOT, Stage.DONE, null, nodeA, true);
368                 validateIndexRecoveryState(shardResponse.recoveryState().getIndex());
369             }
370         }
371     }

373     private List<ShardRecoveryResponse> findRecoveriesForTargetNode(String nodeName, List<ShardRecoveryResponse> responses) {
374         List<ShardRecoveryResponse> nodeResponses = new ArrayList<>();
375         for (ShardRecoveryResponse response : responses) {
376             if (response.recoveryState().getTargetNode().getName().equals(nodeName)) {
377                 nodeResponses.add(response);
378             }
379         }
380         return nodeResponses;
381     }

383     private IndicesStatsResponse createAndPopulateIndex(String name, int nodeCount, int shardCount, int replicaCount)
384             throws ExecutionException, InterruptedException {
385 
386         logger.info("--> creating test index: {}", name);
387         assertAcked(prepareCreate(name, nodeCount, settingsBuilder().put("number_of_shards", shardCount)
388                 .put("number_of_replicas", replicaCount)));
389         ensureGreen();
390 
391         logger.info("--> indexing sample data");
392         final int numDocs = between(MIN_DOC_COUNT, MAX_DOC_COUNT);
393         final IndexRequestBuilder[] docs = new IndexRequestBuilder[numDocs];
394 
395         for (int i = 0; i < numDocs; i++) {
396             docs[i] = client().prepareIndex(INDEX_NAME, INDEX_TYPE).
397                     setSource("foo-int", randomInt(),
398                             "foo-string", randomAsciiOfLength(32),
399                             "foo-float", randomFloat());
400         }
401 
402         indexRandom(true, docs);
403         flush();
404         assertThat(client().prepareCount(INDEX_NAME).get().getCount(), equalTo((long) numDocs));
405         return client().admin().indices().prepareStats(INDEX_NAME).execute().actionGet();
406     }

408     private void validateIndexRecoveryState(RecoveryState.Index indexState) {
409         assertThat(indexState.time(), greaterThanOrEqualTo(0L));
410         assertThat(indexState.percentFilesRecovered(), greaterThanOrEqualTo(0.0f));
411         assertThat(indexState.percentFilesRecovered(), lessThanOrEqualTo(100.0f));
412         assertThat(indexState.percentBytesRecovered(), greaterThanOrEqualTo(0.0f));
413         assertThat(indexState.percentBytesRecovered(), lessThanOrEqualTo(100.0f));
414     }
