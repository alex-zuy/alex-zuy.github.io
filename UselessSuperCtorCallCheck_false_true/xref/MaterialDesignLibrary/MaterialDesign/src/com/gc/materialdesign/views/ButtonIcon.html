1   package com.gc.materialdesign.views;

3   import android.content.Context;
4   import android.graphics.Canvas;
5   import android.graphics.Paint;
6   import android.graphics.drawable.ColorDrawable;
7   import android.util.AttributeSet;
8   import android.view.MotionEvent;
9   
10  public class ButtonIcon extends ButtonFloat {
11  
12  	public ButtonIcon(Context context, AttributeSet attrs) {
13  		super(context, attrs);
14  	}
15  	
16  	@Override
17  	protected void onInitDefaultValues() {
18  		super.onInitDefaultValues();
19  		rippleSpeed = 2;
20  		rippleSize = 5;
21  		backgroundResId = -1;
22  		// Background shape
23  		setBackgroundDrawable(new ColorDrawable(getResources().getColor(android.R.color.transparent)));
24  	}
25  	
26  	@Override
27  	public boolean onTouchEvent(MotionEvent event) {
28  		boolean returnBool = super.onTouchEvent(event);
29  		if(x != -1){
30  			x = getWidth() / 2;
31  			y = getHeight() / 2;
32  		}
33  		return returnBool;
34  	}
35  	
36  	@Override
37  	protected void onDraw(Canvas canvas) {
38  		if (x != -1) {
39  			Paint paint = new Paint();
40  			paint.setAntiAlias(true);
41  			paint.setColor(backgroundColor);
42  			canvas.drawCircle(x, y, radius, paint);
43  			if(radius > getHeight()/rippleSize)
44  				radius += rippleSpeed;
45  			if(radius >= getWidth() / 2 - rippleSpeed){
46  				x = -1;
47  				y = -1;
48  				radius = getHeight()/rippleSize;
49  				if(isEnabled() && clickAfterRipple == true && onClickListener != null)
50  					onClickListener.onClick(this);
51  			}
52  		}
53  		invalidate();
54  	}
55  
56  }
