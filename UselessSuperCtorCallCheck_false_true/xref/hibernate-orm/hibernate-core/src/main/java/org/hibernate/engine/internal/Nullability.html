24  package org.hibernate.engine.internal;
25  
26  import java.util.Iterator;
27  
28  import org.hibernate.HibernateException;
29  import org.hibernate.PropertyValueException;
30  import org.hibernate.bytecode.instrumentation.spi.LazyPropertyInitializer;
31  import org.hibernate.engine.spi.CascadingActions;
32  import org.hibernate.engine.spi.SessionImplementor;
33  import org.hibernate.persister.entity.EntityPersister;
34  import org.hibernate.type.CollectionType;
35  import org.hibernate.type.CompositeType;
36  import org.hibernate.type.Type;
37  
38  /**
39   * Implements the algorithm for validating property values for illegal null values
40   * 
41   * @author Gavin King
42   */
43  public final class Nullability {
44  	private final SessionImplementor session;
45  	private final boolean checkNullability;
46  
47  	/**
48  	 * Constructs a Nullability
49  	 *
50  	 * @param session The session
51  	 */
52  	public Nullability(SessionImplementor session) {
53  		this.session = session;
54  		this.checkNullability = session.getFactory().getSettings().isCheckNullability();
55  	}
56  	/**
57  	 * Check nullability of the class persister properties
58  	 *
59  	 * @param values entity properties
60  	 * @param persister class persister
61  	 * @param isUpdate whether it is intended to be updated or saved
62  	 *
63  	 * @throws PropertyValueException Break the nullability of one property
64  	 * @throws HibernateException error while getting Component values
65  	 */
66  	public void checkNullability(
67  			final Object[] values,
68  			final EntityPersister persister,
69  			final boolean isUpdate) throws HibernateException {
70  		/*
71  		 * Typically when Bean Validation is on, we don't want to validate null values
72  		 * at the Hibernate Core level. Hence the checkNullability setting.
73  		 */
74  		if ( checkNullability ) {
75  			/*
76  			  * Algorithm
77  			  * Check for any level one nullability breaks
78  			  * Look at non null components to
79  			  *   recursively check next level of nullability breaks
80  			  * Look at Collections contraining component to
81  			  *   recursively check next level of nullability breaks
82  			  *
83  			  *
84  			  * In the previous implementation, not-null stuffs where checked
85  			  * filtering by level one only updateable
86  			  * or insertable columns. So setting a sub component as update="false"
87  			  * has no effect on not-null check if the main component had good checkeability
88  			  * In this implementation, we keep this feature.
89  			  * However, I never see any documentation mentioning that, but it's for
90  			  * sure a limitation.
91  			  */
92  
93  			final boolean[] nullability = persister.getPropertyNullability();
94  			final boolean[] checkability = isUpdate ?
95  				persister.getPropertyUpdateability() :
96  				persister.getPropertyInsertability();
97  			final Type[] propertyTypes = persister.getPropertyTypes();
98  
99  			for ( int i = 0; i < values.length; i++ ) {
100 
101 				if ( checkability[i] && values[i]!= LazyPropertyInitializer.UNFETCHED_PROPERTY ) {
102 					final Object value = values[i];
103 					if ( !nullability[i] && value == null ) {
104 
105 						//check basic level one nullablilty
106 						throw new PropertyValueException(
107 								"not-null property references a null or transient value",
108 								persister.getEntityName(),
109 								persister.getPropertyNames()[i]
110 							);
111 
112 					}
113 					else if ( value != null ) {
114 						//values is not null and is checkable, we'll look deeper
115 						final String breakProperties = checkSubElementsNullability( propertyTypes[i], value );
116 						if ( breakProperties != null ) {
117 							throw new PropertyValueException(
118 								"not-null property references a null or transient value",
119 								persister.getEntityName(),
120 								buildPropertyPath( persister.getPropertyNames()[i], breakProperties )
121 							);
122 						}
123 
124 					}
125 				}
126 
127 			}
128 		}
129 	}
130 
131 	/**
132 	 * check sub elements-nullability. Returns property path that break
133 	 * nullability or null if none
134 	 *
135 	 * @param propertyType type to check
136 	 * @param value value to check
137 	 *
138 	 * @return property path
139 	 * @throws HibernateException error while getting subcomponent values
140 	 */
141 	private String checkSubElementsNullability(Type propertyType, Object value) throws HibernateException {
142 		if ( propertyType.isComponentType() ) {
143 			return checkComponentNullability( value, (CompositeType) propertyType );
144 		}
145 
146 		if ( propertyType.isCollectionType() ) {
147 			// persistent collections may have components
148 			final CollectionType collectionType = (CollectionType) propertyType;
149 			final Type collectionElementType = collectionType.getElementType( session.getFactory() );
150 
151 			if ( collectionElementType.isComponentType() ) {
152 				// check for all components values in the collection
153 				final CompositeType componentType = (CompositeType) collectionElementType;
154 				final Iterator itr = CascadingActions.getLoadedElementsIterator( session, collectionType, value );
155 				while ( itr.hasNext() ) {
156 					final Object compositeElement = itr.next();
157 					if ( compositeElement != null ) {
158 						return checkComponentNullability( compositeElement, componentType );
159 					}
160 				}
161 			}
162 		}
163 
164 		return null;
165 	}
166 
167 	/**
168 	 * check component nullability. Returns property path that break
169 	 * nullability or null if none
170 	 *
171 	 * @param value component properties
172 	 * @param compositeType component not-nullable type
173 	 *
174 	 * @return property path
175 	 * @throws HibernateException error while getting subcomponent values
176 	 */
177 	private String checkComponentNullability(Object value, CompositeType compositeType) throws HibernateException {
178 		// IMPL NOTE : we currently skip checking "any" and "many to any" mappings.
179 		//
180 		// This is not the best solution.  But atm there is a mismatch between AnyType#getPropertyNullability
181 		// and the fact that cascaded-saves for "many to any" mappings are not performed until after this nullability
182 		// check.  So the nullability check fails for transient entity elements with generated identifiers because
183 		// the identifier is not yet generated/assigned (is null)
184 		//
185 		// The more correct fix would be to cascade saves of the many-to-any elements before the Nullability checking
186 
187 		if ( compositeType.isAnyType() ) {
188 			return null;
189 		}
190 
191 		final boolean[] nullability = compositeType.getPropertyNullability();
192 		if ( nullability != null ) {
193 			//do the test
194 			final Object[] subValues = compositeType.getPropertyValues( value, session );
195 			final Type[] propertyTypes = compositeType.getSubtypes();
196 			for ( int i = 0; i < subValues.length; i++ ) {
197 				final Object subValue = subValues[i];
198 				if ( !nullability[i] && subValue==null ) {
199 					return compositeType.getPropertyNames()[i];
200 				}
201 				else if ( subValue != null ) {
202 					final String breakProperties = checkSubElementsNullability( propertyTypes[i], subValue );
203 					if ( breakProperties != null ) {
204 						return buildPropertyPath( compositeType.getPropertyNames()[i], breakProperties );
205 					}
206 				}
207 			}
208 		}
209 		return null;
210 	}
211 
212 	/**
213 	 * Return a well formed property path. Basically, it will return parent.child
214 	 *
215 	 * @param parent parent in path
216 	 * @param child child in path
217 	 *
218 	 * @return parent-child path
219 	 */
220 	private static String buildPropertyPath(String parent, String child) {
221 		return parent + '.' + child;
222 	}

224 }
