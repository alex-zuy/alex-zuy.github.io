24  package org.hibernate.procedure.internal;
25  
26  import java.util.List;
27  import java.util.Map;
28  import java.util.Set;
29  import javax.persistence.ParameterMode;
30  
31  import org.hibernate.Session;
32  import org.hibernate.engine.query.spi.sql.NativeSQLQueryReturn;
33  import org.hibernate.engine.spi.SessionImplementor;
34  import org.hibernate.procedure.ProcedureCall;
35  import org.hibernate.procedure.ProcedureCallMemento;
36  import org.hibernate.procedure.spi.ParameterRegistrationImplementor;
37  import org.hibernate.procedure.spi.ParameterStrategy;
38  import org.hibernate.type.Type;
39  
40  /**
41   * Implementation of ProcedureCallMemento
42   *
43   * @author Steve Ebersole
44   */
45  public class ProcedureCallMementoImpl implements ProcedureCallMemento {
46  	private final String procedureName;
47  	private final NativeSQLQueryReturn[] queryReturns;
48  
49  	private final ParameterStrategy parameterStrategy;
50  	private final List<ParameterMemento> parameterDeclarations;
51  
52  	private final Set<String> synchronizedQuerySpaces;
53  
54  	private final Map<String, Object> hintsMap;
55  
56  	/**
57  	 * Constructs a ProcedureCallImpl
58  	 *
59  	 * @param procedureName The name of the procedure to be called
60  	 * @param queryReturns The result mappings
61  	 * @param parameterStrategy Are parameters named or positional?
62  	 * @param parameterDeclarations The parameters registrations
63  	 * @param synchronizedQuerySpaces Any query spaces to synchronize on execution
64  	 * @param hintsMap Map of JPA query hints
65  	 */
66  	public ProcedureCallMementoImpl(
67  			String procedureName,
68  			NativeSQLQueryReturn[] queryReturns,
69  			ParameterStrategy parameterStrategy,
70  			List<ParameterMemento> parameterDeclarations,
71  			Set<String> synchronizedQuerySpaces,
72  			Map<String, Object> hintsMap) {
73  		this.procedureName = procedureName;
74  		this.queryReturns = queryReturns;
75  		this.parameterStrategy = parameterStrategy;
76  		this.parameterDeclarations = parameterDeclarations;
77  		this.synchronizedQuerySpaces = synchronizedQuerySpaces;
78  		this.hintsMap = hintsMap;
79  	}
80  
81  	@Override
82  	public ProcedureCall makeProcedureCall(Session session) {
83  		return new ProcedureCallImpl( (SessionImplementor) session, this );
84  	}
85  
86  	@Override
87  	public ProcedureCall makeProcedureCall(SessionImplementor session) {
88  		return new ProcedureCallImpl( session, this );
89  	}
90  
91  	public String getProcedureName() {
92  		return procedureName;
93  	}
94  
95  	public NativeSQLQueryReturn[] getQueryReturns() {
96  		return queryReturns;
97  	}
98  
99  	public ParameterStrategy getParameterStrategy() {
100 		return parameterStrategy;
101 	}
102 
103 	public List<ParameterMemento> getParameterDeclarations() {
104 		return parameterDeclarations;
105 	}
106 
107 	public Set<String> getSynchronizedQuerySpaces() {
108 		return synchronizedQuerySpaces;
109 	}
110 
111 	@Override
112 	public Map<String, Object> getHintsMap() {
113 		return hintsMap;
114 	}
115 
116 	/**
117 	 * A "disconnected" copy of the metadata for a parameter, that can be used in ProcedureCallMementoImpl.
118 	 */
119 	public static class ParameterMemento {
120 		private final Integer position;
121 		private final String name;
122 		private final ParameterMode mode;
123 		private final Class type;
124 		private final Type hibernateType;
125 
126 		/**
127 		 * Create the memento
128 		 *
129 		 * @param position The parameter position
130 		 * @param name The parameter name
131 		 * @param mode The parameter mode
132 		 * @param type The Java type of the parameter
133 		 * @param hibernateType The Hibernate Type.
134 		 */
135 		public ParameterMemento(int position, String name, ParameterMode mode, Class type, Type hibernateType) {
136 			this.position = position;
137 			this.name = name;
138 			this.mode = mode;
139 			this.type = type;
140 			this.hibernateType = hibernateType;
141 		}
142 
143 		public Integer getPosition() {
144 			return position;
145 		}
146 
147 		public String getName() {
148 			return name;
149 		}
150 
151 		public ParameterMode getMode() {
152 			return mode;
153 		}
154 
155 		public Class getType() {
156 			return type;
157 		}
158 
159 		public Type getHibernateType() {
160 			return hibernateType;
161 		}
162 
163 		/**
164 		 * Build a ParameterMemento from the given parameter registration
165 		 *
166 		 * @param registration The parameter registration from a ProcedureCall
167 		 *
168 		 * @return The memento
169 		 */
170 		public static ParameterMemento fromRegistration(ParameterRegistrationImplementor registration) {
171 			return new ParameterMemento(
172 					registration.getPosition(),
173 					registration.getName(),
174 					registration.getMode(),
175 					registration.getType(),
176 					registration.getHibernateType()
177 			);
178 		}
179 
180 	}
181 }
