141    * Tests that the function is invoked only once, even if it throws an
142    * exception.

144   public void testTransformValueRemainsMemoized() throws Exception {
145     class Holder {
146       int value = 2;
147     }
148     final Holder holder = new Holder();
149 
150     // This function adds the holder's value to the input value.
151     Function<Integer, Integer> adder = new Function<Integer, Integer>() {
152       @Override public Integer apply(Integer from) {
153         return from + holder.value;
154       }
155     };
156 
157     // Since holder.value is 2, applying 4 should yield 6.
158     assertEquals(6, adder.apply(4).intValue());
159 
160     ListenableFuture<Integer> immediateFuture = Futures.immediateFuture(4);
161     Future<Integer> transformedFuture = Futures.transform(immediateFuture, adder);
162 
163     // The composed future also yields 6.
164     assertEquals(6, transformedFuture.get().intValue());
165 
166     // Repeated calls yield the same value even though the function's behavior
167     // changes
168     holder.value = 3;
169     assertEquals(6, transformedFuture.get().intValue());
170     assertEquals(7, adder.apply(4).intValue());
171 
172     // Once more, with feeling.
173     holder.value = 4;
174     assertEquals(6, transformedFuture.get().intValue());
175     assertEquals(8, adder.apply(4).intValue());
176 
177     // Memoized get also retains the value.
178     assertEquals(6, transformedFuture.get(1000, TimeUnit.SECONDS).intValue());
179 
180     // Unsurprisingly, recomposing the future will return an updated value.
181     assertEquals(8, Futures.transform(immediateFuture, adder).get().intValue());
182 
183     // Repeating, with the timeout version
184     assertEquals(8, Futures.transform(immediateFuture, adder).get(
185         1000, TimeUnit.SECONDS).intValue());
186   }

192   static class ExecutorSpy implements Executor {
193     Executor delegate;
194     boolean wasExecuted;
195 
196     public ExecutorSpy(Executor delegate) {
197       this.delegate = delegate;
198     }
199 
200     @Override public void execute(Runnable command) {
201       delegate.execute(command);
202       wasExecuted = true;
203     }
