46  @ClusterScope(scope = Scope.TEST, numDataNodes = 0)
47  public class IndicesLifecycleListenerTests extends ElasticsearchIntegrationTest {
48  
49      @Test
50      public void testIndexStateShardChanged() throws Throwable {
51  
52          //start with a single node
53          String node1 = internalCluster().startNode();
54          IndexShardStateChangeListener stateChangeListenerNode1 = new IndexShardStateChangeListener();
55          //add a listener that keeps track of the shard state changes
56          internalCluster().getInstance(IndicesLifecycle.class, node1).addListener(stateChangeListenerNode1);
57  
58          //create an index
59          assertAcked(client().admin().indices().prepareCreate("test")
60                  .setSettings(SETTING_NUMBER_OF_SHARDS, 6, SETTING_NUMBER_OF_REPLICAS, 0));
61          ensureGreen();
62  
63          //new shards got started
64          assertShardStatesMatch(stateChangeListenerNode1, 6, CREATED, RECOVERING, POST_RECOVERY, STARTED);
65  
66  
67          //add a node: 3 out of the 6 shards will be relocated to it
68          //disable allocation before starting a new node, as we need to register the listener first
69          assertAcked(client().admin().cluster().prepareUpdateSettings()
70                  .setPersistentSettings(builder().put(CLUSTER_ROUTING_ALLOCATION_DISABLE_ALLOCATION, true)));
71          String node2 = internalCluster().startNode();
72          IndexShardStateChangeListener stateChangeListenerNode2 = new IndexShardStateChangeListener();
73          //add a listener that keeps track of the shard state changes
74          internalCluster().getInstance(IndicesLifecycle.class, node2).addListener(stateChangeListenerNode2);
75          //re-enable allocation
76          assertAcked(client().admin().cluster().prepareUpdateSettings()
77                  .setPersistentSettings(builder().put(CLUSTER_ROUTING_ALLOCATION_DISABLE_ALLOCATION, false)));
78          ensureGreen();
79  
80          //the 3 relocated shards get closed on the first node
81          assertShardStatesMatch(stateChangeListenerNode1, 3, CLOSED);
82          //the 3 relocated shards get created on the second node
83          assertShardStatesMatch(stateChangeListenerNode2, 3, CREATED, RECOVERING, POST_RECOVERY, STARTED);
84  
85  
86          //increase replicas from 0 to 1
87          assertAcked(client().admin().indices().prepareUpdateSettings("test").setSettings(builder().put(SETTING_NUMBER_OF_REPLICAS, 1)));
88          ensureGreen();
89  
90          //3 replicas are allocated to the first node
91          assertShardStatesMatch(stateChangeListenerNode1, 3, CREATED, RECOVERING, POST_RECOVERY, STARTED);
92  
93          //3 replicas are allocated to the second node
94          assertShardStatesMatch(stateChangeListenerNode2, 3, CREATED, RECOVERING, POST_RECOVERY, STARTED);
95  
96  
97          //close the index
98          assertAcked(client().admin().indices().prepareClose("test"));
99  
100         assertShardStatesMatch(stateChangeListenerNode1, 6, CLOSED);
101         assertShardStatesMatch(stateChangeListenerNode2, 6, CLOSED);
102     }

104     private static void assertShardStatesMatch(final IndexShardStateChangeListener stateChangeListener, final int numShards, final IndexShardState... shardStates)
105             throws InterruptedException {
106 
107         Predicate<Object> waitPredicate = new Predicate<Object>() {
108             @Override
109             public boolean apply(Object input) {
110                 if (stateChangeListener.shardStates.size() != numShards) {
111                     return false;
112                 }
113                 for (List<IndexShardState> indexShardStates : stateChangeListener.shardStates.values()) {
114                     if (indexShardStates == null || indexShardStates.size() != shardStates.length) {
115                         return false;
116                     }
117                     for (int i = 0; i < shardStates.length; i++) {
118                         if (indexShardStates.get(i) != shardStates[i]) {
119                             return false;
120                         }
121                     }
122                 }
123                 return true;
124             }
125         };
126         if (!awaitBusy(waitPredicate, 1, TimeUnit.MINUTES)) {
127             fail("failed to observe expect shard states\n" +
128                     "expected: [" + numShards + "] shards with states: " + Strings.arrayToCommaDelimitedString(shardStates) + "\n" +
129                     "observed:\n" + stateChangeListener);
130         }
131 
132         stateChangeListener.shardStates.clear();
133     }
134 
135     private static class IndexShardStateChangeListener extends IndicesLifecycle.Listener {
136         //we keep track of all the states (ordered) a shard goes through
137         final ConcurrentMap<ShardId, List<IndexShardState>> shardStates = Maps.newConcurrentMap();
138 
139         @Override
140         public void indexShardStateChanged(IndexShard indexShard, @Nullable IndexShardState previousState, IndexShardState newState, @Nullable String reason) {
141             List<IndexShardState> shardStates = this.shardStates.putIfAbsent(indexShard.shardId(),
142                     new CopyOnWriteArrayList<>(new IndexShardState[]{newState}));
143             if (shardStates != null) {
144                 shardStates.add(newState);
145             }
146         }
147 
148         @Override
149         public String toString() {
150             StringBuilder sb = new StringBuilder();
151             for (Map.Entry<ShardId, List<IndexShardState>> entry : shardStates.entrySet()) {
152                 sb.append(entry.getKey()).append(" --> ").append(Strings.collectionToCommaDelimitedString(entry.getValue())).append("\n");
153             }
154             return sb.toString();
155         }
156     }
