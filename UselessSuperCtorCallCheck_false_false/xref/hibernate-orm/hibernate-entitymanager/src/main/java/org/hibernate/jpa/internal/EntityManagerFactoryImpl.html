132 	public EntityManagerFactoryImpl(
133 			String persistenceUnitName,
134 			SessionFactoryImplementor sessionFactory,
135 			SettingsImpl settings,
136 			Map<?, ?> configurationValues,
137 			Configuration cfg) {
138 		this.sessionFactory = (SessionFactoryImpl) sessionFactory;
139 		this.transactionType = settings.getTransactionType();
140 		this.discardOnClose = settings.isReleaseResourcesOnCloseEnabled();
141 		this.sessionInterceptorClass = settings.getSessionInterceptorClass();
142 
143 		final JpaMetaModelPopulationSetting jpaMetaModelPopulationSetting = determineJpaMetaModelPopulationSetting( cfg );
144 		if ( JpaMetaModelPopulationSetting.DISABLED == jpaMetaModelPopulationSetting ) {
145 			this.metamodel = null;
146 		}
147 		else {
148 			this.metamodel = MetamodelImpl.buildMetamodel(
149 					cfg.getClassMappings(),
150 					cfg.getMappedSuperclassMappingsCopy(),
151 					sessionFactory,
152 					JpaMetaModelPopulationSetting.IGNORE_UNSUPPORTED == jpaMetaModelPopulationSetting
153 			);
154 		}
155 		this.criteriaBuilder = new CriteriaBuilderImpl( this );
156 		this.util = new HibernatePersistenceUnitUtil( this );
157 
158 		HashMap<String,Object> props = new HashMap<String, Object>();
159 		addAll( props, sessionFactory.getProperties() );
160 		addAll( props, cfg.getProperties() );
161 		addAll( props, configurationValues );
162 		maskOutSensitiveInformation( props );
163 		this.properties = Collections.unmodifiableMap( props );
164 		String entityManagerFactoryName = (String)this.properties.get( AvailableSettings.ENTITY_MANAGER_FACTORY_NAME);
165 		if (entityManagerFactoryName == null) {
166 			entityManagerFactoryName = persistenceUnitName;
167 		}
168 		if (entityManagerFactoryName == null) {
169 			entityManagerFactoryName = (String) UUID_GENERATOR.generate(null, null);
170 		}
171 		this.entityManagerFactoryName = entityManagerFactoryName;
172 
173 		applyNamedEntityGraphs( cfg.getNamedEntityGraphs() );
174 
175 		EntityManagerFactoryRegistry.INSTANCE.addEntityManagerFactory( entityManagerFactoryName, this );
176 	}

178 	private enum JpaMetaModelPopulationSetting {
179 		ENABLED,
180 		DISABLED,
181 		IGNORE_UNSUPPORTED;
182 		
183 		private static JpaMetaModelPopulationSetting parse(String setting) {
184 			if ( "enabled".equalsIgnoreCase( setting ) ) {
185 				return ENABLED;
186 			}
187 			else if ( "disabled".equalsIgnoreCase( setting ) ) {
188 				return DISABLED;
189 			}
190 			else {
191 				return IGNORE_UNSUPPORTED;
192 			}
193 		}
194 	}
195 	
196 	protected JpaMetaModelPopulationSetting determineJpaMetaModelPopulationSetting(Configuration cfg) {
197 		String setting = ConfigurationHelper.getString(
198 				AvailableSettings.JPA_METAMODEL_POPULATION,
199 				cfg.getProperties(),
200 				null
201 		);
202 		if ( setting == null ) {
203 			setting = ConfigurationHelper.getString( AvailableSettings.JPA_METAMODEL_GENERATION, cfg.getProperties(), null );
204 			if ( setting != null ) {
205 				log.infof( 
206 						"Encountered deprecated setting [%s], use [%s] instead",
207 						AvailableSettings.JPA_METAMODEL_GENERATION,
208 						AvailableSettings.JPA_METAMODEL_POPULATION
209 				);
210 			}
211 		}
212 		return JpaMetaModelPopulationSetting.parse( setting );
213 	}

234 	@SuppressWarnings("unchecked")
235 	private void applyNamedEntityGraphs(Collection<NamedEntityGraphDefinition> namedEntityGraphs) {
236 		for ( NamedEntityGraphDefinition definition : namedEntityGraphs ) {
237 			log.debugf(
238 					"Applying named entity graph [name=%s, entity-name=%s, jpa-entity-name=%s",
239 					definition.getRegisteredName(),
240 					definition.getEntityName(),
241 					definition.getJpaEntityName()
242 			);
243 			final EntityType entityType = metamodel.getEntityTypeByName( definition.getEntityName() );
244 			if ( entityType == null ) {
245 				throw new IllegalArgumentException(
246 						"Attempted to register named entity graph [" + definition.getRegisteredName()
247 								+ "] for unknown entity ["+ definition.getEntityName() + "]"
248 
249 				);
250 			}
251 			final EntityGraphImpl entityGraph = new EntityGraphImpl(
252 					definition.getRegisteredName(),
253 					entityType,
254 					this
255 			);
256 
257 			final NamedEntityGraph namedEntityGraph = definition.getAnnotation();
258 
259 			if ( namedEntityGraph.includeAllAttributes() ) {
260 				for ( Object attributeObject : entityType.getAttributes() ) {
261 					entityGraph.addAttributeNodes( (Attribute) attributeObject );
262 				}
263 			}
264 
265 			if ( namedEntityGraph.attributeNodes() != null ) {
266 				applyNamedAttributeNodes( namedEntityGraph.attributeNodes(), namedEntityGraph, entityGraph );
267 			}
268 
269 			entityGraphs.put( definition.getRegisteredName(), entityGraph );
270 		}
271 	}
272 
273 	private void applyNamedAttributeNodes(
274 			NamedAttributeNode[] namedAttributeNodes,
275 			NamedEntityGraph namedEntityGraph,
276 			AbstractGraphNode graphNode) {
277 		for ( NamedAttributeNode namedAttributeNode : namedAttributeNodes ) {
278 			final String value = namedAttributeNode.value();
279 			AttributeNodeImpl attributeNode = graphNode.addAttribute( value );
280 			if ( StringHelper.isNotEmpty( namedAttributeNode.subgraph() ) ) {
281 				final SubgraphImpl subgraph = attributeNode.makeSubgraph();
282 				applyNamedSubgraphs(
283 						namedEntityGraph,
284 						namedAttributeNode.subgraph(),
285 						subgraph
286 				);
287 			}
288 			if ( StringHelper.isNotEmpty( namedAttributeNode.keySubgraph() ) ) {
289 				final SubgraphImpl subgraph = attributeNode.makeKeySubgraph();
290 
291 				applyNamedSubgraphs(
292 						namedEntityGraph,
293 						namedAttributeNode.keySubgraph(),
294 						subgraph
295 				);
296 			}
297 		}
298 	}
299 
300 	private void applyNamedSubgraphs(NamedEntityGraph namedEntityGraph, String subgraphName, SubgraphImpl subgraph) {
301 		for ( NamedSubgraph namedSubgraph : namedEntityGraph.subgraphs() ) {
302 			if ( subgraphName.equals( namedSubgraph.name() ) ) {
303 				applyNamedAttributeNodes(
304 						namedSubgraph.attributeNodes(),
305 						namedEntityGraph,
306 						subgraph
307 				);
308 			}
309 		}
310 	}

397 	@Override
398 	public void addNamedQuery(String name, Query query) {
399 		validateNotClosed();
400 
401 		// NOTE : we use Query#unwrap here (rather than direct type checking) to account for possibly wrapped
402 		// query implementations
403 
404 		// first, handle StoredProcedureQuery
405 		try {
406 			final StoredProcedureQueryImpl unwrapped = query.unwrap( StoredProcedureQueryImpl.class );
407 			if ( unwrapped != null ) {
408 				addNamedStoredProcedureQuery( name, unwrapped );
409 				return;
410 			}
411 		}
412 		catch ( PersistenceException ignore ) {
413 			// this means 'query' is not a StoredProcedureQueryImpl
414 		}
415 
416 		// then try as a native-SQL or JPQL query
417 		try {
418 			final HibernateQuery unwrapped = query.unwrap( HibernateQuery.class );
419 			if ( unwrapped != null ) {
420 				// create and register the proper NamedQueryDefinition...
421 				final org.hibernate.Query hibernateQuery = unwrapped.getHibernateQuery();
422 				if ( org.hibernate.SQLQuery.class.isInstance( hibernateQuery ) ) {
423 					sessionFactory.registerNamedSQLQueryDefinition(
424 							name,
425 							extractSqlQueryDefinition( (org.hibernate.SQLQuery) hibernateQuery, name )
426 					);
427 				}
428 				else {
429 					sessionFactory.registerNamedQueryDefinition( name, extractHqlQueryDefinition( hibernateQuery, name ) );
430 				}
431 				return;
432 			}
433 		}
434 		catch ( PersistenceException ignore ) {
435 			// this means 'query' is not a native-SQL or JPQL query
436 		}
437 
438 
439 		// if we get here, we are unsure how to properly unwrap the incoming query to extract the needed information
440 		throw new PersistenceException(
441 				String.format(
442 						"Unsure how to how to properly unwrap given Query [%s] as basis for named query",
443 						query
444 				)
445 		);
446 	}

448 	private void addNamedStoredProcedureQuery(String name, StoredProcedureQueryImpl query) {
449 		final ProcedureCall procedureCall = query.getHibernateProcedureCall();
450 		sessionFactory.getNamedQueryRepository().registerNamedProcedureCallMemento(
451 				name,
452 				procedureCall.extractMemento( query.getHints() )
453 		);
454 	}

456 	private NamedSQLQueryDefinition extractSqlQueryDefinition(org.hibernate.SQLQuery nativeSqlQuery, String name) {
457 		final NamedSQLQueryDefinitionBuilder builder = new NamedSQLQueryDefinitionBuilder( name );
458 		fillInNamedQueryBuilder( builder, nativeSqlQuery );
459 		builder.setCallable( nativeSqlQuery.isCallable() )
460 				.setQuerySpaces( nativeSqlQuery.getSynchronizedQuerySpaces() )
461 				.setQueryReturns( nativeSqlQuery.getQueryReturns() );
462 		return builder.createNamedQueryDefinition();
463 	}
464 
465 	private NamedQueryDefinition extractHqlQueryDefinition(org.hibernate.Query hqlQuery, String name) {
466 		final NamedQueryDefinitionBuilder builder = new NamedQueryDefinitionBuilder( name );
467 		fillInNamedQueryBuilder( builder, hqlQuery );
468 		// LockOptions only valid for HQL/JPQL queries...
469 		builder.setLockOptions( hqlQuery.getLockOptions().makeCopy() );
470 		return builder.createNamedQueryDefinition();
471 	}
472 
473 	private void fillInNamedQueryBuilder(NamedQueryDefinitionBuilder builder, org.hibernate.Query query) {
474 		builder.setQuery( query.getQueryString() )
475 				.setComment( query.getComment() )
476 				.setCacheable( query.isCacheable() )
477 				.setCacheRegion( query.getCacheRegion() )
478 				.setCacheMode( query.getCacheMode() )
479 				.setTimeout( query.getTimeout() )
480 				.setFetchSize( query.getFetchSize() )
481 				.setFirstResult( query.getFirstResult() )
482 				.setMaxResults( query.getMaxResults() )
483 				.setReadOnly( query.isReadOnly() )
484 				.setFlushMode( query.getFlushMode() );
485 	}

488 	@SuppressWarnings("unchecked")
489 	public <T> T unwrap(Class<T> cls) {
490 		if ( SessionFactory.class.isAssignableFrom( cls ) ) {
491 			return ( T ) sessionFactory;
492 		}
493 		if ( SessionFactoryImplementor.class.isAssignableFrom( cls ) ) {
494 			return ( T ) sessionFactory;
495 		}
496 		if ( EntityManager.class.isAssignableFrom( cls ) ) {
497 			return ( T ) this;
498 		}
499 		throw new PersistenceException( "Hibernate cannot unwrap EntityManagerFactory as " + cls.getName() );
500 	}

502 	@Override
503 	public <T> void addNamedEntityGraph(String graphName, EntityGraph<T> entityGraph) {
504 		if ( ! EntityGraphImpl.class.isInstance( entityGraph ) ) {
505 			throw new IllegalArgumentException(
506 					"Unknown type of EntityGraph for making named : " + entityGraph.getClass().getName()
507 			);
508 		}
509 		final EntityGraphImpl<T> copy = ( (EntityGraphImpl<T>) entityGraph ).makeImmutableCopy( graphName );
510 		final EntityGraphImpl old = entityGraphs.put( graphName, copy );
511 		if ( old != null ) {
512 			log.debugf( "EntityGraph being replaced on EntityManagerFactory for name %s", graphName );
513 		}
514 	}

520 	@SuppressWarnings("unchecked")
521 	public <T> List<EntityGraph<? super T>> findEntityGraphsByType(Class<T> entityClass) {
522 		final EntityType<T> entityType = getMetamodel().entity( entityClass );
523 		if ( entityType == null ) {
524 			throw new IllegalArgumentException( "Given class is not an entity : " + entityClass.getName() );
525 		}
526 
527 		final List<EntityGraph<? super T>> results = new ArrayList<EntityGraph<? super T>>();
528 		for ( EntityGraphImpl entityGraph : this.entityGraphs.values() ) {
529 			if ( entityGraph.appliesTo( entityType ) ) {
530 				results.add( entityGraph );
531 			}
532 		}
533 		return results;
534 	}

557 	private static class JPACache implements Cache {
558 		private SessionFactoryImplementor sessionFactory;
559 
560 		private JPACache(SessionFactoryImplementor sessionFactory) {
561 			this.sessionFactory = sessionFactory;
562 		}
563 
564 		public boolean contains(Class entityClass, Object identifier) {
565 			return sessionFactory.getCache().containsEntity( entityClass, ( Serializable ) identifier );
566 		}
567 
568 		public void evict(Class entityClass, Object identifier) {
569 			sessionFactory.getCache().evictEntity( entityClass, ( Serializable ) identifier );
570 		}
571 
572 		public void evict(Class entityClass) {
573 			sessionFactory.getCache().evictEntityRegion( entityClass );
574 		}
575 
576 		public void evictAll() {
577 			// Evict only the "JPA cache", which is purely defined as the entity regions.
578 			sessionFactory.getCache().evictEntityRegions();
579 // TODO : if we want to allow an optional clearing of all cache data, the additional calls would be:
580 //			sessionFactory.getCache().evictCollectionRegions();
581 //			sessionFactory.getCache().evictQueryRegions();
582 		}
583 
584 		@Override
585 		@SuppressWarnings("unchecked")
586 		public <T> T unwrap(Class<T> cls) {
587 			if ( RegionFactory.class.isAssignableFrom( cls ) ) {
588 				return (T) sessionFactory.getSettings().getRegionFactory();
589 			}
590 			if ( org.hibernate.Cache.class.isAssignableFrom( cls ) ) {
591 				return (T) sessionFactory.getCache();
592 			}
593 			throw new PersistenceException( "Hibernate cannot unwrap Cache as " + cls.getName() );
594 		}
595 	}

605 	}

628 		private final HibernateEntityManagerFactory emf;
629 		private transient PersistenceUtilHelper.MetadataCache cache;
630 
631 		private HibernatePersistenceUnitUtil(EntityManagerFactoryImpl emf) {
632 			this.emf = emf;
633 			this.cache = emf.cache;
634 		}
635 
636 		public boolean isLoaded(Object entity, String attributeName) {
637 			// added log message to help with HHH-7454, if state == LoadState,NOT_LOADED, returning true or false is not accurate.
638 			log.debug("PersistenceUnitUtil#isLoaded is not always accurate; consider using EntityManager#contains instead");
639 			LoadState state = PersistenceUtilHelper.isLoadedWithoutReference( entity, attributeName, cache );
640 			if (state == LoadState.LOADED) {
641 				return true;
642 			}
643 			else if (state == LoadState.NOT_LOADED ) {
644 				return false;
645 			}
646 			else {
647 				return PersistenceUtilHelper.isLoadedWithReference( entity, attributeName, cache ) != LoadState.NOT_LOADED;
648 			}
649 		}
650 
651 		public boolean isLoaded(Object entity) {
652 			// added log message to help with HHH-7454, if state == LoadState,NOT_LOADED, returning true or false is not accurate.
653 			log.debug("PersistenceUnitUtil#isLoaded is not always accurate; consider using EntityManager#contains instead");
654 			return PersistenceUtilHelper.isLoaded( entity ) != LoadState.NOT_LOADED;
655 		}
656 
657 		public Object getIdentifier(Object entity) {
658 			final Class entityClass = Hibernate.getClass( entity );
659 			final ClassMetadata classMetadata = emf.getSessionFactory().getClassMetadata( entityClass );
660 			if (classMetadata == null) {
661 				throw new IllegalArgumentException( entityClass + " is not an entity" );
662 			}
663 			//TODO does that work for @IdClass?
664 			return classMetadata.getIdentifier( entity );
665 		}
666 	}
