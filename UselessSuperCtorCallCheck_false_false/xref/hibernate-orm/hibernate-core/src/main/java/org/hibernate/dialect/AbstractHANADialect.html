85  	// the ClobTypeDescriptor and NClobTypeDescriptor for HANA are slightly
86  	// changed from the standard ones. The HANA JDBC driver currently closes any
87  	// stream passed in via
88  	// PreparedStatement.setCharacterStream(int,Reader,long)
89  	// after the stream has been processed. this causes problems later if we are
90  	// using non-contexual lob creation and HANA then closes our StringReader.
91  	// see test case LobLocatorTest
92  
93  	private static final ClobTypeDescriptor HANA_CLOB_STREAM_BINDING = new ClobTypeDescriptor() {
94  		/** serial version uid. */
95  		private static final long serialVersionUID = -379042275442752102L;
96  
97  		@Override
98  		public <X> BasicBinder<X> getClobBinder(final JavaTypeDescriptor<X> javaTypeDescriptor) {
99  			return new BasicBinder<X>( javaTypeDescriptor, this ) {
100 				@Override
101 				protected void doBind(final PreparedStatement st, final X value, final int index,
102 						final WrapperOptions options) throws SQLException {
103 					final CharacterStream characterStream = javaTypeDescriptor.unwrap( value, CharacterStream.class,
104 							options );
105 
106 					if ( value instanceof ClobImplementer ) {
107 						st.setCharacterStream( index, new CloseSuppressingReader( characterStream.asReader() ),
108 								characterStream.getLength() );
109 					}
110 					else {
111 						st.setCharacterStream( index, characterStream.asReader(), characterStream.getLength() );
112 					}
113 
114 				}
115 			};
116 		}
117 	};

314 	@Override
315 	public SQLExceptionConversionDelegate buildSQLExceptionConversionDelegate() {
316 		return new SQLExceptionConversionDelegate() {
317 			@Override
318 			public JDBCException convert(final SQLException sqlException, final String message, final String sql) {
319 
320 				final int errorCode = JdbcExceptionHelper.extractErrorCode( sqlException );
321 
322 				if ( errorCode == 131 ) {
323 					// 131 - Transaction rolled back by lock wait timeout
324 					return new LockTimeoutException( message, sqlException, sql );
325 				}
326 
327 				if ( errorCode == 146 ) {
328 					// 146 - Resource busy and acquire with NOWAIT specified
329 					return new LockTimeoutException( message, sqlException, sql );
330 				}
331 
332 				if ( errorCode == 132 ) {
333 					// 132 - Transaction rolled back due to unavailable resource
334 					return new LockAcquisitionException( message, sqlException, sql );
335 				}
336 
337 				if ( errorCode == 133 ) {
338 					// 133 - Transaction rolled back by detected deadlock
339 					return new LockAcquisitionException( message, sqlException, sql );
340 				}
341 
342 				// 259 - Invalid table name
343 				// 260 - Invalid column name
344 				// 261 - Invalid index name
345 				// 262 - Invalid query name
346 				// 263 - Invalid alias name
347 				if ( errorCode == 257 || ( errorCode >= 259 && errorCode <= 263 ) ) {
348 					throw new SQLGrammarException( message, sqlException, sql );
349 				}
350 
351 				// 257 - Cannot insert NULL or update to NULL
352 				// 301 - Unique constraint violated
353 				// 461 - foreign key constraint violation
354 				// 462 - failed on update or delete by foreign key constraint violation
355 				if ( errorCode == 287 || errorCode == 301 || errorCode == 461 || errorCode == 462 ) {
356 					final String constraintName = getViolatedConstraintNameExtracter().extractConstraintName(
357 							sqlException );
358 
359 					return new ConstraintViolationException( message, sqlException, sql, constraintName );
360 				}
361 
362 				return null;
363 			}
364 		};
365 	}

412 	@Override
413 	public String getForUpdateString(final String aliases, final LockOptions lockOptions) {
414 		LockMode lockMode = lockOptions.getLockMode();
415 		final Iterator<Map.Entry<String, LockMode>> itr = lockOptions.getAliasLockIterator();
416 		while ( itr.hasNext() ) {
417 			// seek the highest lock mode
418 			final Map.Entry<String, LockMode> entry = itr.next();
419 			final LockMode lm = entry.getValue();
420 			if ( lm.greaterThan( lockMode ) ) {
421 				lockMode = lm;
422 			}
423 		}
424 
425 		// not sure why this is sometimes empty
426 		if ( aliases == null || "".equals( aliases ) ) {
427 			return getForUpdateString( lockMode );
428 		}
429 
430 		return getForUpdateString( lockMode ) + " of " + aliases;
431 	}
