198     @Test
199     public void testSynchronizationOfMultipleSequences() throws Throwable {
200         final TestASynchronousObservable o1 = new TestASynchronousObservable();
201         final TestASynchronousObservable o2 = new TestASynchronousObservable();
202 
203         // use this latch to cause onNext to wait until we're ready to let it go
204         final CountDownLatch endLatch = new CountDownLatch(1);
205 
206         final AtomicInteger concurrentCounter = new AtomicInteger();
207         final AtomicInteger totalCounter = new AtomicInteger();
208 
209         Observable<String> m = Observable.merge(Observable.create(o1), Observable.create(o2));
210         m.subscribe(new Subscriber<String>() {
211 
212             @Override
213             public void onCompleted() {
214 
215             }
216 
217             @Override
218             public void onError(Throwable e) {
219                 throw new RuntimeException("failed", e);
220             }
221 
222             @Override
223             public void onNext(String v) {
224                 totalCounter.incrementAndGet();
225                 concurrentCounter.incrementAndGet();
226                 try {
227                     // wait here until we're done asserting
228                     endLatch.await();
229                 } catch (InterruptedException e) {
230                     e.printStackTrace();
231                     throw new RuntimeException("failed", e);
232                 } finally {
233                     concurrentCounter.decrementAndGet();
234                 }
235             }
236 
237         });
238 
239         // wait for both observables to send (one should be blocked)
240         o1.onNextBeingSent.await();
241         o2.onNextBeingSent.await();
242 
243         // I can't think of a way to know for sure that both threads have or are trying to send onNext
244         // since I can't use a CountDownLatch for "after" onNext since I want to catch during it
245         // but I can't know for sure onNext is invoked
246         // so I'm unfortunately reverting to using a Thread.sleep to allow the process scheduler time
247         // to make sure after o1.onNextBeingSent and o2.onNextBeingSent are hit that the following
248         // onNext is invoked.
249 
250         Thread.sleep(300);
251 
252         try { // in try/finally so threads are released via latch countDown even if assertion fails
253             assertEquals(1, concurrentCounter.get());
254         } finally {
255             // release so it can finish
256             endLatch.countDown();
257         }
258 
259         try {
260             o1.t.join();
261             o2.t.join();
262         } catch (InterruptedException e) {
263             throw new RuntimeException(e);
264         }
265 
266         assertEquals(2, totalCounter.get());
267         assertEquals(0, concurrentCounter.get());
268     }

432         assertTrue(os2.get());

736      * For example, 10,000 Observables are being merged (bad use case to begin with, but ...) and it's only one of them
737      * that will ever emit. If backpressure only allowed the first 1,000 to be sent, we would hang and never receive an event.
738      * 
739      * Thus, we must allow all Observables to be sent. The ScalarSynchronousObservable use case is an exception to this since
740      * we can grab the value synchronously.
