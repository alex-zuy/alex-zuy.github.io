220    * Tests that the function is invoked only once, even if it throws an exception.

222   public void testTransformValueRemainsMemoized() throws Exception {
223     class Holder {
224 
225       int value = 2;
226     }
227     final Holder holder = new Holder();
228 
229     // This function adds the holder's value to the input value.
230     Function<Integer, Integer> adder = new Function<Integer, Integer>() {
231       @Override
232       public Integer apply(Integer from) {
233         return from + holder.value;
234       }
235     };
236 
237     // Since holder.value is 2, applying 4 should yield 6.
238     assertEquals(6, adder.apply(4).intValue());
239 
240     ListenableFuture<Integer> immediateFuture = Futures.immediateFuture(4);
241     Future<Integer> transformedFuture = Futures.transform(immediateFuture, adder);
242 
243     // The composed future also yields 6.
244     assertEquals(6, transformedFuture.get().intValue());
245 
246     // Repeated calls yield the same value even though the function's behavior
247     // changes
248     holder.value = 3;
249     assertEquals(6, transformedFuture.get().intValue());
250     assertEquals(7, adder.apply(4).intValue());
251 
252     // Once more, with feeling.
253     holder.value = 4;
254     assertEquals(6, transformedFuture.get().intValue());
255     assertEquals(8, adder.apply(4).intValue());
256 
257     // Memoized get also retains the value.
258     assertEquals(6, transformedFuture.get(1000, TimeUnit.SECONDS).intValue());
259 
260     // Unsurprisingly, recomposing the future will return an updated value.
261     assertEquals(8, Futures.transform(immediateFuture, adder).get().intValue());
262 
263     // Repeating, with the timeout version
264     assertEquals(8, Futures.transform(immediateFuture, adder).get(
265         1000, TimeUnit.SECONDS).intValue());
266   }

1007     private boolean expectCall = false;
1008     private final CountDownLatch calledCountDown =
1009         new CountDownLatch(1);
1010 
1011     @Override
1012     public void run() {
1013       assertTrue("Listener called before it was expected", expectCall);
1014       assertFalse("Listener called more than once", wasCalled());
1015       calledCountDown.countDown();
1016     }
1017 
1018     public void expectCall() {
1019       assertFalse("expectCall is already true", expectCall);
1020       expectCall = true;
1021     }
1022 
1023     public boolean wasCalled() {
1024       return calledCountDown.getCount() == 0;
1025     }
1026 
1027     public void waitForCall() throws InterruptedException {
1028       assertTrue("expectCall is false", expectCall);
1029       calledCountDown.await();
1030     }
