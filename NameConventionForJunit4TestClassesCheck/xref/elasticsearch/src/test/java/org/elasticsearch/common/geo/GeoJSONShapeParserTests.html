888     private ShapeCollection<Shape> shapeCollection(Shape... shapes) {
889         return new ShapeCollection<>(Arrays.asList(shapes), SPATIAL_CONTEXT);
890     }
891 
892     private ShapeCollection<Shape> shapeCollection(Geometry... geoms) {
893         List<Shape> shapes = new ArrayList<>(geoms.length);
894         for (Geometry geom : geoms) {
895             shapes.add(jtsGeom(geom));
896         }
897         return new ShapeCollection<>(shapes, SPATIAL_CONTEXT);
898     }

900     private JtsGeometry jtsGeom(Geometry geom) {
901         return new JtsGeometry(geom, SPATIAL_CONTEXT, false, false);
902     }
