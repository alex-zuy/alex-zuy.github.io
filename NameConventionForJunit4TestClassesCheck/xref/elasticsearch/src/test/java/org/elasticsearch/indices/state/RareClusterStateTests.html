40  @ElasticsearchIntegrationTest.ClusterScope(scope = ElasticsearchIntegrationTest.Scope.SUITE, numDataNodes = 1, numClientNodes = 0, transportClientRatio = 0)
41  public class RareClusterStateTests extends ElasticsearchIntegrationTest {
42  
43      @Override
44      protected int numberOfShards() {
45          return 1;
46      }
47  
48      @Override
49      protected int numberOfReplicas() {
50          return 0;
51      }
52  
53      @Override
54      protected Settings nodeSettings(int nodeOrdinal) {
55          return ImmutableSettings.builder()
56                  .put(super.nodeSettings(nodeOrdinal))
57                  .put("gateway.type", "local")
58                  .build();
59      }
60  
61      @Test
62      public void testUnassignedShardAndEmptyNodesInRoutingTable() throws Exception {
63          createIndex("a");
64          ensureSearchable("a");
65          ClusterState current = clusterService().state();
66          GatewayAllocator allocator = internalCluster().getInstance(GatewayAllocator.class);
67  
68          AllocationDeciders allocationDeciders = new AllocationDeciders(ImmutableSettings.EMPTY, new AllocationDecider[0]);
69          RoutingNodes routingNodes = new RoutingNodes(
70                  ClusterState.builder(current)
71                          .routingTable(RoutingTable.builder(current.routingTable()).remove("a").addAsRecovery(current.metaData().index("a")))
72                          .nodes(DiscoveryNodes.EMPTY_NODES)
73                          .build()
74          );
75          ClusterInfo clusterInfo = new ClusterInfo(ImmutableMap.<String, DiskUsage>of(), ImmutableMap.<String, Long>of());
76  
77          RoutingAllocation routingAllocation =  new RoutingAllocation(allocationDeciders, routingNodes, current.nodes(), clusterInfo);
78          allocator.allocateUnassigned(routingAllocation);
79      }
