98      private void setupTribeNode(Settings settings) {
99          ImmutableMap<String,String> asMap = internalCluster().getDefaultSettings().getAsMap();
100         ImmutableSettings.Builder tribe1Defaults = ImmutableSettings.builder();
101         ImmutableSettings.Builder tribe2Defaults = ImmutableSettings.builder();
102         for (Map.Entry<String, String> entry : asMap.entrySet()) {
103             tribe1Defaults.put("tribe.t1." + entry.getKey(), entry.getValue());
104             tribe2Defaults.put("tribe.t2." + entry.getKey(), entry.getValue());
105         }
106         Settings merged = ImmutableSettings.builder()
107                 .put("tribe.t1.cluster.name", internalCluster().getClusterName())
108                 .put("tribe.t2.cluster.name", cluster2.getClusterName())
109                 .put("tribe.blocks.write", false)
110                 .put("tribe.blocks.read", false)
111                 .put(settings)
112                 .put(tribe1Defaults.build())
113                 .put(tribe2Defaults.build())
114                 .put(internalCluster().getDefaultSettings())
115                 .put("node.name", "tribe_node") // make sure we can identify threads from this node
116                 .build();
117 
118         tribeNode = NodeBuilder.nodeBuilder()
119                 .settings(merged)
120                 .node();
121         tribeClient = tribeNode.client();
122     }

124     @Test
125     public void testGlobalReadWriteBlocks() throws Exception {
126         logger.info("create 2 indices, test1 on t1, and test2 on t2");
127         internalCluster().client().admin().indices().prepareCreate("test1").get();
128         cluster2.client().admin().indices().prepareCreate("test2").get();
129 
130 
131         setupTribeNode(ImmutableSettings.builder()
132                 .put("tribe.blocks.write", true)
133                 .put("tribe.blocks.metadata", true)
134                 .build());
135 
136         logger.info("wait till tribe has the same nodes as the 2 clusters");
137         awaitSameNodeCounts();
138         // wait till the tribe node connected to the cluster, by checking if the index exists in the cluster state
139         logger.info("wait till test1 and test2 exists in the tribe node state");
140         awaitIndicesInClusterState("test1", "test2");
141 
142         try {
143             tribeClient.prepareIndex("test1", "type1", "1").setSource("field1", "value1").execute().actionGet();
144             fail("cluster block should be thrown");
145         } catch (ClusterBlockException e) {
146             // all is well!
147         }
148         try {
149             tribeClient.admin().indices().prepareOptimize("test1").execute().actionGet();
150             fail("cluster block should be thrown");
151         } catch (ClusterBlockException e) {
152             // all is well!
153         }
154         try {
155             tribeClient.admin().indices().prepareOptimize("test2").execute().actionGet();
156             fail("cluster block should be thrown");
157         } catch (ClusterBlockException e) {
158             // all is well!
159         }
160     }

196     @Test
197     public void testOnConflictDrop() throws Exception {
198         logger.info("create 2 indices, test1 on t1, and test2 on t2");
199         assertAcked(cluster().client().admin().indices().prepareCreate("conflict"));
200         assertAcked(cluster2.client().admin().indices().prepareCreate("conflict"));
201         assertAcked(cluster().client().admin().indices().prepareCreate("test1"));
202         assertAcked(cluster2.client().admin().indices().prepareCreate("test2"));
203 
204         setupTribeNode(ImmutableSettings.builder()
205                 .put("tribe.on_conflict", "drop")
206                 .build());
207 
208         logger.info("wait till tribe has the same nodes as the 2 clusters");
209         awaitSameNodeCounts();
210 
211         // wait till the tribe node connected to the cluster, by checking if the index exists in the cluster state
212         logger.info("wait till test1 and test2 exists in the tribe node state");
213         awaitIndicesInClusterState("test1", "test2");
214 
215         assertThat(tribeClient.admin().cluster().prepareState().get().getState().getMetaData().index("test1").getSettings().get(TribeService.TRIBE_NAME), equalTo("t1"));
216         assertThat(tribeClient.admin().cluster().prepareState().get().getState().getMetaData().index("test2").getSettings().get(TribeService.TRIBE_NAME), equalTo("t2"));
217         assertThat(tribeClient.admin().cluster().prepareState().get().getState().getMetaData().hasIndex("conflict"), equalTo(false));
218     }

248     @Test
249     public void testTribeOnOneCluster() throws Exception {
250         setupTribeNode(ImmutableSettings.EMPTY);
251         logger.info("create 2 indices, test1 on t1, and test2 on t2");
252         internalCluster().client().admin().indices().prepareCreate("test1").get();
253         cluster2.client().admin().indices().prepareCreate("test2").get();
254 
255 
256         // wait till the tribe node connected to the cluster, by checking if the index exists in the cluster state
257         logger.info("wait till test1 and test2 exists in the tribe node state");
258         awaitIndicesInClusterState("test1", "test2");
259 
260         logger.info("wait till tribe has the same nodes as the 2 clusters");
261         awaitSameNodeCounts();
262 
263         assertThat(tribeClient.admin().cluster().prepareHealth().setWaitForGreenStatus().get().getStatus(), equalTo(ClusterHealthStatus.GREEN));
264 
265         logger.info("create 2 docs through the tribe node");
266         tribeClient.prepareIndex("test1", "type1", "1").setSource("field1", "value1").get();
267         tribeClient.prepareIndex("test2", "type1", "1").setSource("field1", "value1").get();
268         tribeClient.admin().indices().prepareRefresh().get();
269 
270         logger.info("verify they are there");
271         assertHitCount(tribeClient.prepareCount().get(), 2l);
272         assertHitCount(tribeClient.prepareSearch().get(), 2l);
273         assertBusy(new Runnable() {
274             @Override
275             public void run() {
276                 ClusterState tribeState = tribeNode.client().admin().cluster().prepareState().get().getState();
277                 assertThat(tribeState.getMetaData().index("test1").mapping("type1"), notNullValue());
278                 assertThat(tribeState.getMetaData().index("test2").mapping("type1"), notNullValue());
279             }
280         });
281 
282 
283         logger.info("write to another type");
284         tribeClient.prepareIndex("test1", "type2", "1").setSource("field1", "value1").get();
285         tribeClient.prepareIndex("test2", "type2", "1").setSource("field1", "value1").get();
286         tribeClient.admin().indices().prepareRefresh().get();
287 
288 
289         logger.info("verify they are there");
290         assertHitCount(tribeClient.prepareCount().get(), 4l);
291         assertHitCount(tribeClient.prepareSearch().get(), 4l);
292         assertBusy(new Runnable() {
293             @Override
294             public void run() {
295                 ClusterState tribeState = tribeNode.client().admin().cluster().prepareState().get().getState();
296                 assertThat(tribeState.getMetaData().index("test1").mapping("type1"), notNullValue());
297                 assertThat(tribeState.getMetaData().index("test1").mapping("type2"), notNullValue());
298                 assertThat(tribeState.getMetaData().index("test2").mapping("type1"), notNullValue());
299                 assertThat(tribeState.getMetaData().index("test2").mapping("type2"), notNullValue());
300             }
301         });
302 
303         logger.info("make sure master level write operations fail... (we don't really have a master)");
304         try {
305             tribeClient.admin().indices().prepareCreate("tribe_index").setMasterNodeTimeout("10ms").get();
306             fail();
307         } catch (MasterNotDiscoveredException e) {
308             // all is well!
309         }
310 
311         logger.info("delete an index, and make sure its reflected");
312         cluster2.client().admin().indices().prepareDelete("test2").get();
313         assertBusy(new Runnable() {
314             @Override
315             public void run() {
316                 ClusterState tribeState = tribeNode.client().admin().cluster().prepareState().get().getState();
317                 assertTrue(tribeState.getMetaData().hasIndex("test1"));
318                 assertFalse(tribeState.getMetaData().hasIndex("test2"));
319                 assertTrue(tribeState.getRoutingTable().hasIndex("test1"));
320                 assertFalse(tribeState.getRoutingTable().hasIndex("test2"));
321             }
322         });
323 
324         logger.info("stop a node, make sure its reflected");
325         cluster2.stopRandomDataNode();
326         awaitSameNodeCounts();
327     }

329     private void awaitIndicesInClusterState(final String... indices) throws Exception {
330         assertBusy(new Runnable() {
331             @Override
332             public void run() {
333                 ClusterState tribeState = tribeNode.client().admin().cluster().prepareState().get().getState();
334                 for (String index : indices) {
335                     assertTrue(tribeState.getMetaData().hasIndex(index));
336                     assertTrue(tribeState.getRoutingTable().hasIndex(index));
337                 }
338             }
339         });
340     }
341 
342     private void awaitSameNodeCounts() throws Exception {
343         assertBusy(new Runnable() {
344             @Override
345             public void run() {
346                 DiscoveryNodes tribeNodes = tribeNode.client().admin().cluster().prepareState().get().getState().getNodes();
347                 assertThat(countDataNodesForTribe("t1", tribeNodes), equalTo(internalCluster().client().admin().cluster().prepareState().get().getState().getNodes().dataNodes().size()));
348                 assertThat(countDataNodesForTribe("t2", tribeNodes), equalTo(cluster2.client().admin().cluster().prepareState().get().getState().getNodes().dataNodes().size()));
349             }
350         });
351     }
352 
353     private int countDataNodesForTribe(String tribeName, DiscoveryNodes nodes) {
354         int count = 0;
355         for (DiscoveryNode node : nodes) {
356             if (!node.dataNode()) {
357                 continue;
358             }
359             if (tribeName.equals(node.getAttributes().get(TribeService.TRIBE_NAME))) {
360                 count++;
361             }
362         }
363         return count;
364     }
