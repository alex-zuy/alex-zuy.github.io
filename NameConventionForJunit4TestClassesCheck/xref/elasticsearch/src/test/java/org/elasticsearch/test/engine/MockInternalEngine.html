51  public class MockInternalEngine extends InternalEngine {
52  
53      public static final ConcurrentMap<AssertingSearcher, RuntimeException> INFLIGHT_ENGINE_SEARCHERS = new ConcurrentHashMap<>();
54  
55      private MockInternalEngineHolder.MockContext mockContext;
56  
57      public MockInternalEngine(MockInternalEngineHolder.MockContext mockContext, ShardId shardId, ESLogger logger, CodecService codecService,
58                                ThreadPool threadPool, ShardIndexingService indexingService,
59                                @Nullable IndicesWarmer warmer, Store store, SnapshotDeletionPolicy deletionPolicy, Translog translog,
60                                MergePolicyProvider mergePolicyProvider, MergeSchedulerProvider mergeScheduler, AnalysisService analysisService,
61                                SimilarityService similarityService, boolean enableGcDeletes, long gcDeletesInMillis, ByteSizeValue indexingBufferSize, String codecName,
62                                boolean compoundOnFlush, int indexConcurrency, boolean optimizeAutoGenerateId, boolean failEngineOnCorruption,
63                                FailedEngineListener failedEngineListener) throws EngineException {
64          super(shardId, logger, codecService, threadPool, indexingService, warmer, store, deletionPolicy, translog, mergePolicyProvider,
65                  mergeScheduler, analysisService, similarityService, enableGcDeletes, gcDeletesInMillis, indexingBufferSize, codecName,
66                  compoundOnFlush, indexConcurrency, optimizeAutoGenerateId, failEngineOnCorruption, failedEngineListener);
67          this.mockContext = mockContext;
68      }
69  
70      @Override
71      public void close() {
72          try {
73              super.close();
74          } finally {
75              if (logger.isTraceEnabled()) {
76                  // log debug if we have pending searchers
77                  for (Map.Entry<MockInternalEngine.AssertingSearcher, RuntimeException> entry : INFLIGHT_ENGINE_SEARCHERS.entrySet()) {
78                      logger.trace("Unreleased Searchers instance for shard [{}]", entry.getValue(), entry.getKey().shardId);
79                  }
80              }
81          }
82          logger.debug("Ongoing recoveries after engine close: " + onGoingRecoveries.get());
83      }
84  
85      @Override
86      protected Searcher newSearcher(String source, IndexSearcher searcher, SearcherManager manager) throws EngineException {
87  
88          IndexReader reader = searcher.getIndexReader();
89          IndexReader wrappedReader = reader;
90          if (reader instanceof DirectoryReader && mockContext.wrapReader) {
91              wrappedReader = wrapReader((DirectoryReader) reader);
92          }
93          // this executes basic query checks and asserts that weights are normalized only once etc.
94          final AssertingIndexSearcher assertingIndexSearcher = new AssertingIndexSearcher(mockContext.random, wrappedReader);
95          assertingIndexSearcher.setSimilarity(searcher.getSimilarity());
96          // pass the original searcher to the super.newSearcher() method to make sure this is the searcher that will
97          // be released later on. If we wrap an index reader here must not pass the wrapped version to the manager
98          // on release otherwise the reader will be closed too early. - good news, stuff will fail all over the place if we don't get this right here
99          return new AssertingSearcher(assertingIndexSearcher, super.newSearcher(source, searcher, manager), shardId);
100     }
101 
102     private DirectoryReader wrapReader(DirectoryReader reader) {
103         try {
104             Constructor<?>[] constructors = mockContext.wrapper.getConstructors();
105             Constructor<?> nonRandom = null;
106             for (Constructor<?> constructor : constructors) {
107                 Class<?>[] parameterTypes = constructor.getParameterTypes();
108                 if (parameterTypes.length > 0 && parameterTypes[0] == DirectoryReader.class) {
109                     if (parameterTypes.length == 1) {
110                         nonRandom = constructor;
111                     } else if (parameterTypes.length == 2 && parameterTypes[1] == Settings.class) {
112 
113                         return (DirectoryReader) constructor.newInstance(reader, mockContext.indexSettings);
114                     }
115                 }
116             }
117             if (nonRandom != null) {
118                 return (DirectoryReader) nonRandom.newInstance(reader);
119             }
120         } catch (Exception e) {
121             throw new ElasticsearchException("Can not wrap reader", e);
122         }
123         return reader;
124     }

126     public final class AssertingSearcher implements Searcher {
127         private final Searcher wrappedSearcher;
128         private final ShardId shardId;
129         private final IndexSearcher indexSearcher;
130         private RuntimeException firstReleaseStack;
131         private final Object lock = new Object();
132         private final int initialRefCount;
133 
134         public AssertingSearcher(IndexSearcher indexSearcher, Searcher wrappedSearcher, ShardId shardId) {
135             // we only use the given index searcher here instead of the IS of the wrapped searcher. the IS might be a wrapped searcher
136             // with a wrapped reader.
137             this.wrappedSearcher = wrappedSearcher;
138             this.shardId = shardId;
139             initialRefCount = wrappedSearcher.reader().getRefCount();
140             this.indexSearcher = indexSearcher;
141             assert initialRefCount > 0 : "IndexReader#getRefCount() was [" + initialRefCount + "] expected a value > [0] - reader is already closed";
142             INFLIGHT_ENGINE_SEARCHERS.put(this, new RuntimeException("Unreleased Searcher, source [" + wrappedSearcher.source() + "]"));
143         }
144 
145         @Override
146         public String source() {
147             return wrappedSearcher.source();
148         }
149 
150         @Override
151         public void close() throws ElasticsearchException {
152             RuntimeException remove = INFLIGHT_ENGINE_SEARCHERS.remove(this);
153             synchronized (lock) {
154                 // make sure we only get this once and store the stack of the first caller!
155                 if (remove == null) {
156                     assert firstReleaseStack != null;
157                     AssertionError error = new AssertionError("Released Searcher more than once, source [" + wrappedSearcher.source() + "]");
158                     error.initCause(firstReleaseStack);
159                     throw error;
160                 } else {
161                     assert firstReleaseStack == null;
162                     firstReleaseStack = new RuntimeException("Searcher Released first here, source [" + wrappedSearcher.source() + "]");
163                 }
164             }
165             final int refCount = wrappedSearcher.reader().getRefCount();
166             // this assert seems to be paranoid but given LUCENE-5362 we better add some assertions here to make sure we catch any potential
167             // problems.
168             assert refCount > 0 : "IndexReader#getRefCount() was [" + refCount + "] expected a value > [0] - reader is already closed. Initial refCount was: [" + initialRefCount + "]";
169             try {
170                 wrappedSearcher.close();
171             } catch (RuntimeException ex) {
172                 logger.debug("Failed to release searcher", ex);
173                 throw ex;
174             }
175         }
176 
177         @Override
178         public IndexReader reader() {
179             return indexSearcher.getIndexReader();
180         }
181 
182         @Override
183         public IndexSearcher searcher() {
184             return indexSearcher;
185         }
186 
187         public ShardId shardId() {
188             return shardId;
189         }
190     }
