108     @Inject
109     public InternalEngineHolder(ShardId shardId, @IndexSettings Settings indexSettings, ThreadPool threadPool,
110                                 IndexSettingsService indexSettingsService, ShardIndexingService indexingService, @Nullable IndicesWarmer warmer,
111                                 Store store, SnapshotDeletionPolicy deletionPolicy, Translog translog,
112                                 MergePolicyProvider mergePolicyProvider, MergeSchedulerProvider mergeScheduler,
113                                 AnalysisService analysisService, SimilarityService similarityService, CodecService codecService) throws EngineException {
114         super(shardId, indexSettings);
115         Preconditions.checkNotNull(store, "Store must be provided to the engine");
116         Preconditions.checkNotNull(deletionPolicy, "Snapshot deletion policy must be provided to the engine");
117         Preconditions.checkNotNull(translog, "Translog must be provided to the engine");
118 
119         this.gcDeletesInMillis = indexSettings.getAsTime(INDEX_GC_DELETES, TimeValue.timeValueSeconds(60)).millis();
120         this.indexingBufferSize = componentSettings.getAsBytesSize("index_buffer_size", new ByteSizeValue(64, ByteSizeUnit.MB)); // not really important, as it is set by the IndexingMemory manager
121         this.codecName = indexSettings.get(INDEX_CODEC, "default");
122 
123         this.threadPool = threadPool;
124         this.indexSettingsService = indexSettingsService;
125         this.indexingService = indexingService;
126         this.warmer = (InternalIndicesWarmer) warmer;
127         this.store = store;
128         this.deletionPolicy = deletionPolicy;
129         this.translog = translog;
130         this.mergePolicyProvider = mergePolicyProvider;
131         this.mergeScheduler = mergeScheduler;
132         this.analysisService = analysisService;
133         this.similarityService = similarityService;
134         this.codecService = codecService;
135         this.compoundOnFlush = indexSettings.getAsBoolean(INDEX_COMPOUND_ON_FLUSH, this.compoundOnFlush);
136         this.indexConcurrency = indexSettings.getAsInt(INDEX_INDEX_CONCURRENCY, Math.max(IndexWriterConfig.DEFAULT_MAX_THREAD_STATES, (int) (EsExecutors.boundedNumberOfProcessors(indexSettings) * 0.65)));
137         this.optimizeAutoGenerateId = indexSettings.getAsBoolean("index.optimize_auto_generated_id", true);
138 
139         this.failEngineOnCorruption = indexSettings.getAsBoolean(INDEX_FAIL_ON_CORRUPTION, true);
140         this.failOnMergeFailure = indexSettings.getAsBoolean(INDEX_FAIL_ON_MERGE_FAILURE, true);
141         this.mergeSchedulerFailureListener = new FailEngineOnMergeFailure();
142         this.mergeScheduler.addFailureListener(mergeSchedulerFailureListener);
143         this.mergeSchedulerListener = new MergeScheduleListener();
144         this.mergeScheduler.addListener(mergeSchedulerListener);
145 
146         this.settingsListener = new ApplySettings(logger, this);
147         this.indexSettingsService.addListener(this.settingsListener);
148         store.incRef();
149     }

188     @Override
189     public synchronized void start() throws EngineException {
190         if (closed) {
191             throw new EngineClosedException(shardId);
192         }
193         InternalEngine currentEngine = this.currentEngine.get();
194         if (currentEngine != null) {
195             throw new EngineAlreadyStartedException(shardId);
196         }
197         InternalEngine newEngine = createEngine();
198         store.incRef();
199         try {
200             newEngine.start();
201             boolean success = this.currentEngine.compareAndSet(null, newEngine);
202             assert success : "engine changes should be done under a synchronize";
203         } finally {
204             store.decRef();
205         }
206     }
207 
208     @Override
209     public synchronized void stop() throws EngineException {
210         InternalEngine currentEngine = this.currentEngine.getAndSet(null);
211         if (currentEngine != null) {
212             currentEngine.close();
213         }
214     }
215 
216     @Override
217     public synchronized void close() throws ElasticsearchException {
218         if (closed == false) {
219             closed = true;
220             try {
221                 InternalEngine currentEngine = this.currentEngine.getAndSet(null);
222                 if (currentEngine != null) {
223                     currentEngine.close();
224                 }
225                 mergeScheduler.removeFailureListener(mergeSchedulerFailureListener);
226                 mergeScheduler.removeListener(mergeSchedulerListener);
227                 indexSettingsService.removeListener(settingsListener);
228             } finally {
229                 store.decRef();
230             }
231         }
232     }

338     // called by the current engine
339     @Override
340     public void onFailedEngine(ShardId shardId, String reason, @Nullable Throwable failure) {
341         try {
342             for (FailedEngineListener listener : failedEngineListeners) {
343                 try {
344                     listener.onFailedEngine(shardId, reason, failure);
345                 } catch (Exception e) {
346                     logger.warn("exception while notifying engine failure", e);
347                 }
348             }
349         } finally {
350             close(); // we need to close ourself - we failed all bets are off
351         }
352     }

354     static class ApplySettings implements IndexSettingsService.Listener {
355 
356         private final ESLogger logger;
357         private final InternalEngineHolder holder;
358 
359         ApplySettings(ESLogger logger, InternalEngineHolder holder) {
360             this.logger = logger;
361             this.holder = holder;
362         }
363 
364         @Override
365         public void onRefreshSettings(Settings settings) {
366             boolean change = false;
367             long gcDeletesInMillis = settings.getAsTime(INDEX_GC_DELETES, TimeValue.timeValueMillis(holder.gcDeletesInMillis)).millis();
368             if (gcDeletesInMillis != holder.gcDeletesInMillis) {
369                 logger.info("updating index.gc_deletes from [{}] to [{}]", TimeValue.timeValueMillis(holder.gcDeletesInMillis), TimeValue.timeValueMillis(gcDeletesInMillis));
370                 holder.gcDeletesInMillis = gcDeletesInMillis;
371                 change = true;
372             }
373 
374             final boolean compoundOnFlush = settings.getAsBoolean(INDEX_COMPOUND_ON_FLUSH, holder.compoundOnFlush);
375             if (compoundOnFlush != holder.compoundOnFlush) {
376                 logger.info("updating {} from [{}] to [{}]", INDEX_COMPOUND_ON_FLUSH, holder.compoundOnFlush, compoundOnFlush);
377                 holder.compoundOnFlush = compoundOnFlush;
378                 change = true;
379             }
380 
381             final boolean failEngineOnCorruption = settings.getAsBoolean(INDEX_FAIL_ON_CORRUPTION, holder.failEngineOnCorruption);
382             if (failEngineOnCorruption != holder.failEngineOnCorruption) {
383                 logger.info("updating {} from [{}] to [{}]", INDEX_FAIL_ON_CORRUPTION, holder.failEngineOnCorruption, failEngineOnCorruption);
384                 holder.failEngineOnCorruption = failEngineOnCorruption;
385                 change = true;
386             }
387             int indexConcurrency = settings.getAsInt(INDEX_INDEX_CONCURRENCY, holder.indexConcurrency);
388             if (indexConcurrency != holder.indexConcurrency) {
389                 logger.info("updating index.index_concurrency from [{}] to [{}]", holder.indexConcurrency, indexConcurrency);
390                 holder.indexConcurrency = indexConcurrency;
391                 // we have to flush in this case, since it only applies on a new index writer
392                 change = true;
393             }
394             final String codecName = settings.get(INDEX_CODEC, holder.codecName);
395             if (!codecName.equals(holder.codecName)) {
396                 logger.info("updating index.codec from [{}] to [{}]", holder.codecName, codecName);
397                 holder.codecName = codecName;
398                 // we want to flush in this case, so the new codec will be reflected right away...
399                 change = true;
400             }
401             final boolean failOnMergeFailure = settings.getAsBoolean(INDEX_FAIL_ON_MERGE_FAILURE, holder.failOnMergeFailure);
402             if (failOnMergeFailure != holder.failOnMergeFailure) {
403                 logger.info("updating {} from [{}] to [{}]", INDEX_FAIL_ON_MERGE_FAILURE, holder.failOnMergeFailure, failOnMergeFailure);
404                 holder.failOnMergeFailure = failOnMergeFailure;
405             }
406 
407 
408             if (change) {
409                 holder.updateSettings();
410             }
411         }
412     }
413 
414      synchronized void updateSettings() {
415         // we need to make sure that we wait for the engine to be fully initialized
416         // the start method sets the current engine once it's done but samples the settings
417         // at construction time.
418         final InternalEngine engine = currentEngine.get();
419         if (engine != null) {
420             engine.updateSettings(gcDeletesInMillis, compoundOnFlush, failEngineOnCorruption, indexConcurrency, codecName);
421         }
422     }

424     class FailEngineOnMergeFailure implements MergeSchedulerProvider.FailureListener {
425         @Override
426         public void onFailedMerge(MergePolicy.MergeException e) {
427             if (Lucene.isCorruptionException(e)) {
428                 if (failEngineOnCorruption) {
429                     failEngine("corrupt file detected source: [merge]", e);
430                 } else {
431                     logger.warn("corrupt file detected source: [merge] but [{}] is set to [{}]", e, INDEX_FAIL_ON_CORRUPTION, failEngineOnCorruption);
432                 }
433             } else if (failOnMergeFailure) {
434                 failEngine("merge exception", e);
435             }
436         }
437     }

439     class MergeScheduleListener implements MergeSchedulerProvider.Listener {
440         private final AtomicInteger numMergesInFlight = new AtomicInteger(0);
441         private final AtomicBoolean isThrottling = new AtomicBoolean();
442 
443         @Override
444         public synchronized void beforeMerge(OnGoingMerge merge) {
445             int maxNumMerges = mergeScheduler.getMaxMerges();
446             InternalEngine currentEngineImpl = currentEngine.get();
447             if (numMergesInFlight.incrementAndGet() > maxNumMerges && currentEngineImpl != null) {
448                 if (isThrottling.getAndSet(true) == false) {
449                     logger.info("now throttling indexing: numMergesInFlight={}, maxNumMerges={}", numMergesInFlight, maxNumMerges);
450                     indexingService.throttlingActivated();
451                     currentEngineImpl.activateThrottling();
452                 }
453             }
454         }
455 
456         @Override
457         public synchronized void afterMerge(OnGoingMerge merge) {
458             int maxNumMerges = mergeScheduler.getMaxMerges();
459             InternalEngine currentEngineImpl = currentEngine.get();
460             if (numMergesInFlight.decrementAndGet() < maxNumMerges && currentEngineImpl != null) {
461                 if (isThrottling.getAndSet(false)) {
462                     logger.info("stop throttling indexing: numMergesInFlight={}, maxNumMerges={}", numMergesInFlight, maxNumMerges);
463                     indexingService.throttlingDeactivated();
464                     currentEngineImpl.deactivateThrottling();
465                 }
466             }
467         }
