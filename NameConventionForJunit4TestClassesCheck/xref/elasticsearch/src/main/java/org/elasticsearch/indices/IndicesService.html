125     @Override
126     protected void doStop() throws ElasticsearchException {
127         ImmutableSet<String> indices = ImmutableSet.copyOf(this.indices.keySet());
128         final CountDownLatch latch = new CountDownLatch(indices.size());
129 
130         final ExecutorService indicesStopExecutor = Executors.newFixedThreadPool(5, EsExecutors.daemonThreadFactory("indices_shutdown"));
131 
132         for (final String index : indices) {
133             indicesStopExecutor.execute(new Runnable() {
134                 @Override
135                 public void run() {
136                     try {
137                         removeIndex(index, "shutdown", false);
138                     } catch (Throwable e) {
139                         logger.warn("failed to delete index on stop [" + index + "]", e);
140                     } finally {
141                         latch.countDown();
142                     }
143                 }
144             });
145         }
146         try {
147             if (latch.await(30, TimeUnit.SECONDS) == false) {
148               logger.warn("Not all shards are closed yet, waited 30sec - stopping service");
149             }
150         } catch (InterruptedException e) {
151             // ignore
152         } finally {
153             indicesStopExecutor.shutdown();
154         }
155     }

157     @Override
158     protected void doClose() throws ElasticsearchException {
159         IOUtils.closeWhileHandlingException(injector.getInstance(RecoverySettings.class),

161     }

167     /**
168      * Returns the node stats indices stats. The <tt>includePrevious</tt> flag controls
169      * if old shards stats will be aggregated as well (only for relevant stats, such as
170      * refresh and indexing, not for docs/store).
171      */
172     public NodeIndicesStats stats(boolean includePrevious) {
173         return stats(true, new CommonStatsFlags().all());
174     }
175 
176     public NodeIndicesStats stats(boolean includePrevious, CommonStatsFlags flags) {
177         CommonStats oldStats = new CommonStats(flags);
178 
179         if (includePrevious) {
180             Flag[] setFlags = flags.getFlags();
181             for (Flag flag : setFlags) {
182                 switch (flag) {
183                     case Get:
184                         oldStats.get.add(oldShardsStats.getStats);
185                         break;
186                     case Indexing:
187                         oldStats.indexing.add(oldShardsStats.indexingStats);
188                         break;
189                     case Search:
190                         oldStats.search.add(oldShardsStats.searchStats);
191                         break;
192                     case Merge:
193                         oldStats.merge.add(oldShardsStats.mergeStats);
194                         break;
195                     case Refresh:
196                         oldStats.refresh.add(oldShardsStats.refreshStats);
197                         break;
198                     case Flush:
199                         oldStats.flush.add(oldShardsStats.flushStats);
200                         break;
201                 }
202             }
203         }
204 
205         Map<Index, List<IndexShardStats>> statsByShard = Maps.newHashMap();
206         for (IndexService indexService : indices.values()) {
207             for (IndexShard indexShard : indexService) {
208                 try {
209                     if (indexShard.routingEntry() == null) {
210                         continue;
211                     }
212                     IndexShardStats indexShardStats = new IndexShardStats(indexShard.shardId(), new ShardStats[] { new ShardStats(indexShard, indexShard.routingEntry(), flags) });
213                     if (!statsByShard.containsKey(indexService.index())) {
214                         statsByShard.put(indexService.index(), Lists.<IndexShardStats>newArrayList(indexShardStats));
215                     } else {
216                         statsByShard.get(indexService.index()).add(indexShardStats);
217                     }
218                 } catch (IllegalIndexShardStateException e) {
219                     // we can safely ignore illegal state on ones that are closing for example
220                 }
221             }
222         }
223         return new NodeIndicesStats(oldStats, statsByShard);
224     }

252     /**
253      * Returns an IndexService for the specified index if exists otherwise returns <code>null</code>.
254      *
255      * Even if the index name appeared in {@link #indices()} <code>null</code> can still be returned as an
256      * index maybe removed in the meantime, so preferable use the associated {@link IndexService} in order to prevent NPE.

260         return indices.get(index);
261     }

263     /**
264      * Returns an IndexService for the specified index if exists otherwise a {@link IndexMissingException} is thrown.
265      */
266     public IndexService indexServiceSafe(String index) throws IndexMissingException {
267         IndexService indexService = indexService(index);
268         if (indexService == null) {
269             throw new IndexMissingException(new Index(index));
270         }
271         return indexService;
272     }

274     public synchronized IndexService createIndex(String sIndexName, Settings settings, String localNodeId) throws ElasticsearchException {
275         if (!lifecycle.started()) {
276             throw new ElasticsearchIllegalStateException("Can't create an index [" + sIndexName + "], node is closed");
277         }
278         Index index = new Index(sIndexName);
279         if (indicesInjectors.containsKey(index.name())) {
280             throw new IndexAlreadyExistsException(index);
281         }
282 
283         indicesLifecycle.beforeIndexCreated(index);
284 
285         logger.debug("creating Index [{}], shards [{}]/[{}]", sIndexName, settings.get(SETTING_NUMBER_OF_SHARDS), settings.get(SETTING_NUMBER_OF_REPLICAS));
286 
287         Settings indexSettings = settingsBuilder()
288                 .put(this.settings)
289                 .put(settings)
290                 .classLoader(settings.getClassLoader())
291                 .build();
292 
293         ModulesBuilder modules = new ModulesBuilder();
294         modules.add(new IndexNameModule(index));
295         modules.add(new LocalNodeIdModule(localNodeId));
296         modules.add(new IndexSettingsModule(index, indexSettings));
297         modules.add(new IndexPluginsModule(indexSettings, pluginsService));
298         modules.add(new IndexStoreModule(indexSettings));
299         modules.add(new AnalysisModule(indexSettings, indicesAnalysisService));
300         modules.add(new SimilarityModule(indexSettings));
301         modules.add(new IndexCacheModule(indexSettings));
302         modules.add(new IndexFieldDataModule(indexSettings));
303         modules.add(new CodecModule(indexSettings));
304         modules.add(new MapperServiceModule());
305         modules.add(new IndexQueryParserModule(indexSettings));
306         modules.add(new IndexAliasesServiceModule());
307         modules.add(new IndexModule(indexSettings));
308 
309         Injector indexInjector;
310         try {
311             indexInjector = modules.createChildInjector(injector);
312         } catch (CreationException e) {
313             throw new IndexCreationException(index, Injectors.getFirstErrorFailure(e));
314         } catch (Throwable e) {
315             throw new IndexCreationException(index, e);
316         }
317 
318         indicesInjectors.put(index.name(), indexInjector);
319 
320         IndexService indexService = indexInjector.getInstance(IndexService.class);
321 
322         indicesLifecycle.afterIndexCreated(indexService);
323 
324         indices = newMapBuilder(indices).put(index.name(), indexService).immutableMap();
325 
326         return indexService;
327     }

329     /**
330      * Removes the given index from this service and releases all associated resources. Persistent parts of the index
331      * like the shards files, state and transaction logs are kept around in the case of a disaster recovery.
332      * @param index the index to remove
333      * @param reason  the high level reason causing this removal
334      */
335     public void removeIndex(String index, String reason) throws ElasticsearchException {
336         removeIndex(index, reason, false);
337     }
338 
339     /**
340      * Deletes the given index. Persistent parts of the index
341      * like the shards files, state and transaction logs are removed once all resources are released.
342      *
343      * Equivalent to {@link #removeIndex(String, String)} but fires
344      * different lifecycle events to ensure pending resources of this index are immediately removed.
345      * @param index the index to delete
346      * @param reason the high level reason causing this delete
347      */
348     public void deleteIndex(String index, String reason) throws ElasticsearchException {
349         removeIndex(index, reason, true);
350     }
351 
352     private void removeIndex(String index, String reason, boolean delete) throws ElasticsearchException {
353         try {
354             final IndexService indexService;
355             final Injector indexInjector;
356             synchronized (this) {
357                 indexInjector = indicesInjectors.remove(index);
358                 if (indexInjector == null) {
359                     return;
360                 }
361 
362                 logger.debug("[{}] closing ... (reason [{}])", index, reason);
363                 Map<String, IndexService> tmpMap = newHashMap(indices);
364                 indexService = tmpMap.remove(index);
365                 indices = ImmutableMap.copyOf(tmpMap);
366             }
367 
368             indicesLifecycle.beforeIndexClosed(indexService);
369             if (delete) {
370                 indicesLifecycle.beforeIndexDeleted(indexService);
371             }
372             IOUtils.close(Iterables.transform(pluginsService.indexServices(), new Function<Class<? extends Closeable>, Closeable>() {
373                 @Override
374                 public Closeable apply(Class<? extends Closeable> input) {
375                     return indexInjector.getInstance(input);
376                 }
377             }));
378 
379             logger.debug("[{}] closing index service (reason [{}])", index, reason);
380             indexService.close(reason);
381 
382             logger.debug("[{}] closing index cache (reason [{}])", index, reason);
383             indexInjector.getInstance(IndexCache.class).close();
384             logger.debug("[{}] clearing index field data (reason [{}])", index, reason);
385             indexInjector.getInstance(IndexFieldDataService.class).clear();
386             logger.debug("[{}] closing analysis service (reason [{}])", index, reason);
387             indexInjector.getInstance(AnalysisService.class).close();
388 
389             logger.debug("[{}] closing mapper service (reason [{}])", index, reason);
390             indexInjector.getInstance(MapperService.class).close();
391             logger.debug("[{}] closing index query parser service (reason [{}])", index, reason);
392             indexInjector.getInstance(IndexQueryParserService.class).close();
393 
394             logger.debug("[{}] closing index service (reason [{}])", index, reason);
395             indexInjector.getInstance(IndexStore.class).close();
396 
397             Injectors.close(injector);
398 
399             logger.debug("[{}] closed... (reason [{}])", index, reason);
400             indicesLifecycle.afterIndexClosed(indexService.index());
401             if (delete) {
402                 indicesLifecycle.afterIndexDeleted(indexService.index());
403             }
404         } catch (IOException ex) {
405             throw new ElasticsearchException("failed to remove index " + index, ex);
406         }
407     }

409     static class OldShardsStats extends IndicesLifecycle.Listener {
410 
411         final SearchStats searchStats = new SearchStats();
412         final GetStats getStats = new GetStats();
413         final IndexingStats indexingStats = new IndexingStats();
414         final MergeStats mergeStats = new MergeStats();
415         final RefreshStats refreshStats = new RefreshStats();
416         final FlushStats flushStats = new FlushStats();
417 
418         @Override
419         public synchronized void beforeIndexShardClosed(ShardId shardId, @Nullable IndexShard indexShard) {
420             if (indexShard != null) {
421                 getStats.add(indexShard.getStats());
422                 indexingStats.add(indexShard.indexingStats(), false);
423                 searchStats.add(indexShard.searchStats(), false);
424                 mergeStats.add(indexShard.mergeStats());
425                 refreshStats.add(indexShard.refreshStats());
426                 flushStats.add(indexShard.flushStats());
427             }
428         }
