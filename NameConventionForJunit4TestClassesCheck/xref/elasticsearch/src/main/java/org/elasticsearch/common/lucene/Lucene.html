107     public static void checkSegmentInfoIntegrity(final Directory directory) throws IOException {
108         new SegmentInfos.FindSegmentsFile(directory) {
109 
110             @Override
111             protected Object doBody(String segmentFileName) throws IOException {
112                 try (IndexInput input = directory.openInput(segmentFileName, IOContext.READ)) {
113                     final int format = input.readInt();
114                     final int actualFormat;
115                     if (format == CodecUtil.CODEC_MAGIC) {
116                         // 4.0+
117                         actualFormat = CodecUtil.checkHeaderNoMagic(input, "segments", SegmentInfos.VERSION_40, Integer.MAX_VALUE);
118                         if (actualFormat >= SegmentInfos.VERSION_48) {
119                             CodecUtil.checksumEntireFile(input);
120                         }
121                     }
122                     // legacy....
123                 }
124                 return null;
125             }
126         }.run();
127     }

129     public static long count(IndexSearcher searcher, Query query) throws IOException {
130         TotalHitCountCollector countCollector = new TotalHitCountCollector();
131         query = wrapCountQuery(query);
132         searcher.search(query, countCollector);
133         return countCollector.getTotalHits();
134     }

201     /**
202      * Performs a count on <code>query</code> and <code>filter</code> with early termination using <code>searcher</code>.
203      * The early termination threshold is specified by the provided <code>collector</code>
204      */
205     public static boolean countWithEarlyTermination(IndexSearcher searcher, Filter filter, Query query,
206                                                         EarlyTerminatingCollector collector) throws IOException {
207         query = wrapCountQuery(query);
208         try {
209             if (filter == null) {
210                 searcher.search(query, collector);
211             } else {
212                 searcher.search(query, filter, collector);
213             }
214         } catch (EarlyTerminationException e) {
215             // early termination
216             return true;
217         }
218         return false;
219     }

252     public static TopDocs readTopDocs(StreamInput in) throws IOException {
253         if (!in.readBoolean()) {
254             // no docs
255             return null;
256         }
257         if (in.readBoolean()) {
258             int totalHits = in.readVInt();
259             float maxScore = in.readFloat();
260 
261             SortField[] fields = new SortField[in.readVInt()];
262             for (int i = 0; i < fields.length; i++) {
263                 String field = null;
264                 if (in.readBoolean()) {
265                     field = in.readString();
266                 }
267                 fields[i] = new SortField(field, readSortType(in), in.readBoolean());
268             }
269 
270             FieldDoc[] fieldDocs = new FieldDoc[in.readVInt()];
271             for (int i = 0; i < fieldDocs.length; i++) {
272                 fieldDocs[i] = readFieldDoc(in);
273             }
274             return new TopFieldDocs(totalHits, fieldDocs, fields, maxScore);
275         } else {
276             int totalHits = in.readVInt();
277             float maxScore = in.readFloat();
278 
279             ScoreDoc[] scoreDocs = new ScoreDoc[in.readVInt()];
280             for (int i = 0; i < scoreDocs.length; i++) {
281                 scoreDocs[i] = new ScoreDoc(in.readVInt(), in.readFloat());
282             }
283             return new TopDocs(totalHits, scoreDocs, maxScore);
284         }
285     }

287     public static FieldDoc readFieldDoc(StreamInput in) throws IOException {
288         Comparable[] cFields = new Comparable[in.readVInt()];
289         for (int j = 0; j < cFields.length; j++) {
290             byte type = in.readByte();
291             if (type == 0) {
292                 cFields[j] = null;
293             } else if (type == 1) {
294                 cFields[j] = in.readString();
295             } else if (type == 2) {
296                 cFields[j] = in.readInt();
297             } else if (type == 3) {
298                 cFields[j] = in.readLong();
299             } else if (type == 4) {
300                 cFields[j] = in.readFloat();
301             } else if (type == 5) {
302                 cFields[j] = in.readDouble();
303             } else if (type == 6) {
304                 cFields[j] = in.readByte();
305             } else if (type == 7) {
306                 cFields[j] = in.readShort();
307             } else if (type == 8) {
308                 cFields[j] = in.readBoolean();
309             } else if (type == 9) {
310                 cFields[j] = in.readBytesRef();
311             } else {
312                 throw new IOException("Can't match type [" + type + "]");
313             }
314         }
315         return new FieldDoc(in.readVInt(), in.readFloat(), cFields);
316     }

322     public static void writeTopDocs(StreamOutput out, TopDocs topDocs, int from) throws IOException {
323         if (topDocs.scoreDocs.length - from < 0) {
324             out.writeBoolean(false);
325             return;
326         }
327         out.writeBoolean(true);
328         if (topDocs instanceof TopFieldDocs) {
329             out.writeBoolean(true);
330             TopFieldDocs topFieldDocs = (TopFieldDocs) topDocs;
331 
332             out.writeVInt(topDocs.totalHits);
333             out.writeFloat(topDocs.getMaxScore());
334 
335             out.writeVInt(topFieldDocs.fields.length);
336             for (SortField sortField : topFieldDocs.fields) {
337                 if (sortField.getField() == null) {
338                     out.writeBoolean(false);
339                 } else {
340                     out.writeBoolean(true);
341                     out.writeString(sortField.getField());
342                 }
343                 if (sortField.getComparatorSource() != null) {
344                     writeSortType(out, ((IndexFieldData.XFieldComparatorSource) sortField.getComparatorSource()).reducedType());
345                 } else {
346                     writeSortType(out, sortField.getType());
347                 }
348                 out.writeBoolean(sortField.getReverse());
349             }
350 
351             out.writeVInt(topDocs.scoreDocs.length - from);
352             int index = 0;
353             for (ScoreDoc doc : topFieldDocs.scoreDocs) {
354                 if (index++ < from) {
355                     continue;
356                 }
357                 writeFieldDoc(out, (FieldDoc) doc);
358             }
359         } else {
360             out.writeBoolean(false);
361             out.writeVInt(topDocs.totalHits);
362             out.writeFloat(topDocs.getMaxScore());
363 
364             out.writeVInt(topDocs.scoreDocs.length - from);
365             int index = 0;
366             for (ScoreDoc doc : topDocs.scoreDocs) {
367                 if (index++ < from) {
368                     continue;
369                 }
370                 writeScoreDoc(out, doc);
371             }
372         }
373     }

375     public static void writeFieldDoc(StreamOutput out, FieldDoc fieldDoc) throws IOException {
376         out.writeVInt(fieldDoc.fields.length);
377         for (Object field : fieldDoc.fields) {
378             if (field == null) {
379                 out.writeByte((byte) 0);
380             } else {
381                 Class type = field.getClass();
382                 if (type == String.class) {
383                     out.writeByte((byte) 1);
384                     out.writeString((String) field);
385                 } else if (type == Integer.class) {
386                     out.writeByte((byte) 2);
387                     out.writeInt((Integer) field);
388                 } else if (type == Long.class) {
389                     out.writeByte((byte) 3);
390                     out.writeLong((Long) field);
391                 } else if (type == Float.class) {
392                     out.writeByte((byte) 4);
393                     out.writeFloat((Float) field);
394                 } else if (type == Double.class) {
395                     out.writeByte((byte) 5);
396                     out.writeDouble((Double) field);
397                 } else if (type == Byte.class) {
398                     out.writeByte((byte) 6);
399                     out.writeByte((Byte) field);
400                 } else if (type == Short.class) {
401                     out.writeByte((byte) 7);
402                     out.writeShort((Short) field);
403                 } else if (type == Boolean.class) {
404                     out.writeByte((byte) 8);
405                     out.writeBoolean((Boolean) field);
406                 } else if (type == BytesRef.class) {
407                     out.writeByte((byte) 9);
408                     out.writeBytesRef((BytesRef) field);
409                 } else {
410                     throw new IOException("Can't handle sort field value of type [" + type + "]");
411                 }
412             }
413         }
414         out.writeVInt(fieldDoc.doc);
415         out.writeFloat(fieldDoc.score);
416     }

418     public static void writeScoreDoc(StreamOutput out, ScoreDoc scoreDoc) throws IOException {
419         if (!scoreDoc.getClass().equals(ScoreDoc.class)) {

435     public static Explanation readExplanation(StreamInput in) throws IOException {
436         Explanation explanation;
437         if (in.readBoolean()) {
438             Boolean match = in.readOptionalBoolean();
439             explanation = new ComplexExplanation();
440             ((ComplexExplanation) explanation).setMatch(match);
441 
442         } else {
443             explanation = new Explanation();
444         }
445         explanation.setValue(in.readFloat());
446         explanation.setDescription(in.readString());
447         if (in.readBoolean()) {
448             int size = in.readVInt();
449             for (int i = 0; i < size; i++) {
450                 explanation.addDetail(readExplanation(in));
451             }
452         }
453         return explanation;
454     }

456     public static void writeExplanation(StreamOutput out, Explanation explanation) throws IOException {
457 
458         if (explanation instanceof ComplexExplanation) {
459             out.writeBoolean(true);
460             out.writeOptionalBoolean(((ComplexExplanation) explanation).getMatch());
461         } else {
462             out.writeBoolean(false);
463         }
464         out.writeFloat(explanation.getValue());
465         out.writeString(explanation.getDescription());
466         Explanation[] subExplanations = explanation.getDetails();
467         if (subExplanations == null) {
468             out.writeBoolean(false);
469         } else {
470             out.writeBoolean(true);
471             out.writeVInt(subExplanations.length);
472             for (Explanation subExp : subExplanations) {
473                 writeExplanation(out, subExp);
474             }
475         }
476     }

489     /**
490      * A collector that terminates early by throwing {@link org.elasticsearch.common.lucene.Lucene.EarlyTerminationException}
491      * when count of matched documents has reached <code>maxCountHits</code>
492      */
493     public final static class EarlyTerminatingCollector extends SimpleCollector {
494 
495         private final int maxCountHits;
496         private final Collector delegate;
497 
498         private int count = 0;
499         private LeafCollector leafCollector;
500 
501         EarlyTerminatingCollector(int maxCountHits) {
502             this.maxCountHits = maxCountHits;
503             this.delegate = NOOP_COLLECTOR;
504         }
505 
506         EarlyTerminatingCollector(final Collector delegate, int maxCountHits) {
507             this.maxCountHits = maxCountHits;
508             this.delegate = (delegate == null) ? NOOP_COLLECTOR : delegate;
509         }
510 
511         public void reset() {
512             count = 0;
513         }
514         public int count() {
515             return count;
516         }
517 
518         public boolean exists() {
519             return count > 0;
520         }
521 
522         @Override
523         public void setScorer(Scorer scorer) throws IOException {
524             leafCollector.setScorer(scorer);
525         }
526 
527         @Override
528         public void collect(int doc) throws IOException {
529             leafCollector.collect(doc);
530 
531             if (++count >= maxCountHits) {
532                 throw new EarlyTerminationException("early termination [CountBased]");
533             }
534         }
535 
536         @Override
537         public void doSetNextReader(LeafReaderContext atomicReaderContext) throws IOException {
538             leafCollector = delegate.getLeafCollector(atomicReaderContext);
539         }
540 
541         @Override
542         public boolean acceptsDocsOutOfOrder() {
543             return leafCollector.acceptsDocsOutOfOrder();
544         }
545     }

571     private static final class LenientParser {
572         public static Version parse(String toParse, Version defaultValue) {
573             if (Strings.hasLength(toParse)) {
574                 try {
575                     return Version.parseLeniently(toParse);
576                 } catch (ParseException e) {
577                     // pass to default
578                 }
579             }
580             return defaultValue;
