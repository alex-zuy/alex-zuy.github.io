17  package com.google.common.util.concurrent;
18  
19  import static com.google.common.util.concurrent.MoreExecutors.directExecutor;
20  
21  import com.google.caliper.BeforeExperiment;
22  import com.google.caliper.Param;
23  import com.google.caliper.api.Footprint;
24  import com.google.caliper.api.SkipThisScenarioException;
25  import com.google.common.util.concurrent.AbstractFutureBenchmarks.Facade;
26  import com.google.common.util.concurrent.AbstractFutureBenchmarks.Impl;
27  
28  import java.util.HashSet;
29  import java.util.Set;
30  import java.util.concurrent.Executor;
31  
32  /**
33   * Measures the size of AbstractFuture implementations.
34   */
35  public class AbstractFutureFootprintBenchmark {
36  
37    enum State { NOT_DONE, FINISHED, CANCELLED, FAILED}
38  
39    @Param State state;
40    @Param Impl impl;
41    @Param({"0", "1", "5", "10"}) int numListeners;
42    @Param({"0", "1", "5", "10"}) int numThreads;
43  
44    private final Set<Thread> blockedThreads = new HashSet<>();
45  
46    @BeforeExperiment void setUp() throws Exception {
47      if (state != State.NOT_DONE && (numListeners != 0 || numThreads != 0)) {
48        throw new SkipThisScenarioException();
49      }
50    }
51  
52    // This exclusion doesn't exclude the TOMBSTONE objects we set. So 'done' NEW futures will look
53    // larger than they are.
54    @Footprint(exclude = {Runnable.class, Executor.class, Thread.class, Exception.class})
55    public Object measureSize() {
56      for (Thread thread : blockedThreads) {
57        thread.interrupt();
58      }
59      blockedThreads.clear();
60      final Facade<Object> f = impl.newFacade();
61      for (int i = 0; i < numThreads; i++) {
62        Thread thread = new Thread() {
63          @Override public void run() {
64            try {
65              f.get();
66            } catch (Throwable expected) {}
67          }
68        };
69        thread.start();
70        blockedThreads.add(thread);
71      }
72      for (int i = 0; i < numListeners; i++) {
73        f.addListener(Runnables.doNothing(), directExecutor());
74      }
75      for (Thread thread : blockedThreads) {
76        AbstractFutureBenchmarks.awaitWaiting(thread);
77      }
78      switch (state) {
79        case NOT_DONE:
80          break;
81        case FINISHED:
82          f.set(null);
83          break;
84        case CANCELLED:
85          f.cancel(false);
86          break;
87        case FAILED:
88          f.setException(new Exception());
89          break;
90        default:
91          throw new AssertionError();
92      }
93      return f;
94    }
95  }
