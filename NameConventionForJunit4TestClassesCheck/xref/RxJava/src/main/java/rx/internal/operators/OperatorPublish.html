40  public class OperatorPublish<T> extends ConnectableObservable<T> {
41      final Observable<? extends T> source;
42      private final RequestHandler<T> requestHandler;
43  
44      public static <T> ConnectableObservable<T> create(Observable<? extends T> source) {
45          return new OperatorPublish<T>(source);
46      }
47  
48      public static <T, R> Observable<R> create(final Observable<? extends T> source, final Func1<? super Observable<T>, ? extends Observable<R>> selector) {
49          return Observable.create(new OnSubscribe<R>() {
50  
51              @Override
52              public void call(final Subscriber<? super R> child) {
53                  OperatorPublish<T> op = new OperatorPublish<T>(source);
54                  selector.call(op).unsafeSubscribe(child);
55                  op.connect(new Action1<Subscription>() {
56  
57                      @Override
58                      public void call(Subscription sub) {
59                          child.add(sub);
60                      }
61  
62                  });
63              }
64  
65          });
66      }
67  
68      private OperatorPublish(Observable<? extends T> source) {
69          this(source, new Object(), new RequestHandler<T>());
70      }
71  
72      private OperatorPublish(Observable<? extends T> source, final Object guard, final RequestHandler<T> requestHandler) {
73          super(new OnSubscribe<T>() {
74              @Override
75              public void call(final Subscriber<? super T> subscriber) {
76                  subscriber.setProducer(new Producer() {
77  
78                      @Override
79                      public void request(long n) {
80                          requestHandler.requestFromChildSubscriber(subscriber, n);
81                      }
82  
83                  });
84                  subscriber.add(Subscriptions.create(new Action0() {
85  
86                      @Override
87                      public void call() {
88                          requestHandler.state.removeSubscriber(subscriber);
89                      }
90  
91                  }));
92              }
93          });
94          this.source = source;
95          this.requestHandler = requestHandler;
96      }
97  
98      @Override
99      public void connect(Action1<? super Subscription> connection) {
100         // each time we connect we create a new Subscription
101         boolean shouldSubscribe = false;
102         
103         // subscription is the state of whether we are connected or not
104         OriginSubscriber<T> origin = requestHandler.state.getOrigin();
105         if (origin == null) {
106             shouldSubscribe = true;
107             requestHandler.state.setOrigin(new OriginSubscriber<T>(requestHandler));
108         }
109 
110         // in the lock above we determined we should subscribe, do it now outside the lock
111         if (shouldSubscribe) {
112             // register a subscription that will shut this down
113             connection.call(Subscriptions.create(new Action0() {
114                 @Override
115                 public void call() {
116                     OriginSubscriber<T> s = requestHandler.state.getOrigin();
117                     requestHandler.state.setOrigin(null);
118                     if (s != null) {
119                         s.unsubscribe();
120                     }
121                 }
122             }));
123 
124             // now that everything is hooked up let's subscribe
125             // as long as the subscription is not null (which can happen if already unsubscribed)
126             OriginSubscriber<T> os = requestHandler.state.getOrigin();
127             if (os != null) {
128                 source.unsafeSubscribe(os);
129             }
130         }
131     }

133     private static class OriginSubscriber<T> extends Subscriber<T> {
134 
135         private final RequestHandler<T> requestHandler;
136         private final AtomicLong originOutstanding = new AtomicLong();
137         private final long THRESHOLD = RxRingBuffer.SIZE / 4;
138         private final RxRingBuffer buffer = RxRingBuffer.getSpmcInstance();
139 
140         OriginSubscriber(RequestHandler<T> requestHandler) {
141             this.requestHandler = requestHandler;
142             add(buffer);
143         }
144 
145         @Override
146         public void onStart() {
147             requestMore(RxRingBuffer.SIZE);
148         }
149 
150         private void requestMore(long r) {
151             originOutstanding.addAndGet(r);
152             request(r);
153         }
154 
155         @Override
156         public void onCompleted() {
157             try {
158                 requestHandler.emit(requestHandler.notifier.completed());
159             } catch (MissingBackpressureException e) {
160                 onError(e);
161             }
162         }
163 
164         @Override
165         public void onError(Throwable e) {
166             List<Throwable> errors = null;
167             for (Subscriber<? super T> subscriber : requestHandler.state.getSubscribers()) {
168                 try {
169                     subscriber.onError(e);
170                 } catch (Throwable e2) {
171                     if (errors == null) {
172                         errors = new ArrayList<Throwable>();
173                     }
174                     errors.add(e2);
175                 }
176             }
177             if (errors != null) {
178                 if (errors.size() == 1) {
179                     Exceptions.propagate(errors.get(0));
180                 } else {
181                     throw new CompositeException("Errors while emitting onError", errors);
182                 }
183             }
184         }
185 
186         @Override
187         public void onNext(T t) {
188             try {
189                 requestHandler.emit(requestHandler.notifier.next(t));
190             } catch (MissingBackpressureException e) {
191                 onError(e);
192             }
193         }
194 
195     }

197     /**
198      * Synchronized mutable state.
199      * 
200      * benjchristensen => I have not figured out a non-blocking approach to this that doesn't involve massive object allocation overhead
201      * with a complicated state machine so I'm sticking with mutex locks and just trying to make sure the work done while holding the
202      * lock is small (such as never emitting data).
203      * 
204      * This does however mean we can't rely on a reference to State being consistent. For example, it can end up with a null OriginSubscriber. 
205      * 
206      * @param <T>
207      */
208     private static class State<T> {
209         private long outstandingRequests = -1;
210         private long emittedSinceRequest = 0;
211         private OriginSubscriber<T> origin;
212         // using AtomicLong to simplify mutating it, not for thread-safety since we're synchronizing access to this class
213         // using LinkedHashMap so the order of Subscribers having onNext invoked is deterministic (same each time the code is run)
214         private final Map<Subscriber<? super T>, AtomicLong> ss = new LinkedHashMap<Subscriber<? super T>, AtomicLong>();
215         @SuppressWarnings("unchecked")
216         private Subscriber<? super T>[] subscribers = new Subscriber[0];
217 
218         public synchronized OriginSubscriber<T> getOrigin() {
219             return origin;
220         }
221 
222         public synchronized void setOrigin(OriginSubscriber<T> o) {
223             this.origin = o;
224         }
225 
226         public synchronized boolean canEmitWithDecrement() {
227             if (outstandingRequests > 0) {
228                 outstandingRequests--;
229                 emittedSinceRequest++;
230                 return true;
231             }
232             return false;
233         }
234 
235         public synchronized void incrementOutstandingAfterFailedEmit() {
236             outstandingRequests++;
237             emittedSinceRequest--;
238         }
239 
240         public synchronized Subscriber<? super T>[] getSubscribers() {
241             return subscribers;
242         }
243 
244         /**
245          * @return long outstandingRequests
246          */
247         public synchronized long requestFromSubscriber(Subscriber<? super T> subscriber, Long request) {
248             AtomicLong r = ss.get(subscriber);
249             if (r == null) {
250                 ss.put(subscriber, new AtomicLong(request));
251             } else {
252                 if (r.get() != Long.MAX_VALUE) {
253                     if (request == Long.MAX_VALUE) {
254                         r.set(Long.MAX_VALUE);
255                     } else {
256                         r.addAndGet(request.longValue());
257                     }
258                 }
259             }
260 
261             return resetAfterSubscriberUpdate();
262         }
263 
264         public synchronized void removeSubscriber(Subscriber<? super T> subscriber) {
265             ss.remove(subscriber);
266             resetAfterSubscriberUpdate();
267         }
268 
269         private long resetAfterSubscriberUpdate() {
270             subscribers = new Subscriber[ss.size()];
271             int i = 0;
272             for (Subscriber<? super T> s : ss.keySet()) {
273                 subscribers[i++] = s;
274             }
275 
276             long lowest = -1;
277             for (AtomicLong l : ss.values()) {
278                 // decrement all we have emitted since last request
279                 long c = l.addAndGet(-emittedSinceRequest);
280                 if (lowest == -1 || c < lowest) {
281                     lowest = c;
282                 }
283             }
284             /*
285              * when receiving a request from a subscriber we reset 'outstanding' to the lowest of all subscribers
286              */
287             outstandingRequests = lowest;
288             emittedSinceRequest = 0;
289             return outstandingRequests;
290         }
291     }

293     private static class RequestHandler<T> {
294         private final NotificationLite<T> notifier = NotificationLite.instance();
295         
296         private final State<T> state = new State<T>();
297         @SuppressWarnings("unused")
298         volatile long wip;
299         @SuppressWarnings("rawtypes")
300         static final AtomicLongFieldUpdater<RequestHandler> WIP = AtomicLongFieldUpdater.newUpdater(RequestHandler.class, "wip");
301 
302         public void requestFromChildSubscriber(Subscriber<? super T> subscriber, Long request) {
303             state.requestFromSubscriber(subscriber, request);
304             OriginSubscriber<T> originSubscriber = state.getOrigin();
305             if(originSubscriber != null) {
306                 drainQueue(originSubscriber);
307             }
308         }
309 
310         public void emit(Object t) throws MissingBackpressureException {
311             OriginSubscriber<T> originSubscriber = state.getOrigin();
312             if(originSubscriber == null) {
313                 // unsubscribed so break ... we are done
314                 return;
315             }
316             if (notifier.isCompleted(t)) {
317                 originSubscriber.buffer.onCompleted();
318             } else {
319                 originSubscriber.buffer.onNext(notifier.getValue(t));
320             }
321             drainQueue(originSubscriber);
322         }
323 
324         private void requestMoreAfterEmission(int emitted) {
325             OriginSubscriber<T> origin = state.getOrigin();
326             if (emitted > 0 && origin != null) {
327                 long r = origin.originOutstanding.addAndGet(-emitted);
328                 if (r <= origin.THRESHOLD) {
329                     origin.requestMore(RxRingBuffer.SIZE - origin.THRESHOLD);
330                 }
331             }
332         }
333 
334         public void drainQueue(OriginSubscriber<T> originSubscriber) {
335             if (WIP.getAndIncrement(this) == 0) {
336                 int emitted = 0;
337                 do {
338                     /*
339                      * Set to 1 otherwise it could have grown very large while in the last poll loop
340                      * and then we can end up looping all those times again here before exiting even once we've drained
341                      */
342                     WIP.set(this, 1);
343                     /**
344                      * This is done in the most inefficient possible way right now and we can revisit the approach.
345                      * If we want to batch this then we need to account for new subscribers arriving with a lower request count
346                      * concurrently while iterating the batch ... or accept that they won't
347                      */
348                     while (true) {
349                         boolean shouldEmit = state.canEmitWithDecrement();
350                         if (!shouldEmit) {
351                             break;
352                         }
353                         Object o = originSubscriber.buffer.poll();
354                         if (o == null) {
355                             // nothing in buffer so increment outstanding back again
356                             state.incrementOutstandingAfterFailedEmit();
357                             break;
358                         }
359 
360                         if (notifier.isCompleted(o)) {
361                             for (Subscriber<? super T> s : state.getSubscribers()) {
362                                 notifier.accept(s, o);
363                             }
364 
365                         } else {
366                             for (Subscriber<? super T> s : state.getSubscribers()) {
367                                 notifier.accept(s, o);
368                             }
369                         }
370                         emitted++;
371                     }
372                 } while (WIP.decrementAndGet(this) > 0);
373                 requestMoreAfterEmission(emitted);
374             }
375         }
376     }
