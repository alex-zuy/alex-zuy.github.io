3   import android.content.Context;
4   import android.graphics.Canvas;
5   import android.graphics.Color;
6   import android.graphics.Paint;
7   import android.util.AttributeSet;
8   import android.widget.TextView;
9   
10  import com.gc.materialdesign.R;
11  
12  public class ButtonFlat extends ButtonRectangle {
13  	
14  	public ButtonFlat(Context context, AttributeSet attrs) {
15  		super(context, attrs);
16  	}
17  	
18  	@Override
19  	protected void onInitDefaultValues(){
20  		textButton = new TextView(getContext());
21  		minHeight = 36;
22  		minWidth = 88;
23  		rippleSpeed = 6f;
24  		defaultTextColor =  Color.parseColor("#1E88E5");
25  		backgroundResId = R.drawable.background_transparent;
26  		rippleColor = Color.parseColor("#88DDDDDD");
27  		//setBackgroundResource(R.drawable.background_transparent);
28  	}
29  	
30  	@Override
31  	protected void onDraw(Canvas canvas) {
32  		//super.onDraw(canvas);//不调用父类的onDraw()方法。因为这会用ButtonRectangle的onDraw()
33  		if (x != -1) {
34  			
35  			Paint paint = new Paint();
36  			paint.setAntiAlias(true);
37  			if (rippleColor == null) {
38  				paint.setColor(Color.parseColor("#88DDDDDD"));
39  			}else {
40  				paint.setColor(rippleColor);
41  			}
42  			canvas.drawCircle(x, y, radius, paint);
43  			if(radius > getHeight()/rippleSize)
44  				radius += rippleSpeed;
45  			if(radius >= getWidth()){
46  				x = -1;
47  				y = -1;
48  				radius = getHeight()/rippleSize;
49  				if (isEnabled() && clickAfterRipple == true && onClickListener != null) {
50  					onClickListener.onClick(this);
51  				}
52  			}
53  		}		
54  		invalidate();
55  	}
56  	
57  	@Override
58  	public void setBackgroundColor(int color) {
59  		super.setBackgroundColor(color);
60  		if (!settedRippleColor) {
61  			// 如果之前没有设置过涟漪颜色，那么就用默认的
62  			rippleColor = Color.parseColor("#88DDDDDD");
63  		}
64  	}
65  	
66  	
67  }
