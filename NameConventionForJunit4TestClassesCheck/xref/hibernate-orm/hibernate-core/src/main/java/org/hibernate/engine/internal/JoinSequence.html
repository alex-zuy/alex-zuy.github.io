42  /**
43   * A sequence of {@link Join} delegates to make it "easier" to work with joins.  The "easier" part is obviously
44   * subjective ;)
45   * <p/>
46   * Additionally JoinSequence is a directed graph of other JoinSequence instances, as represented by the
47   * {@link #next} ({@link #setNext(JoinSequence)}) pointer.
48   *
49   * @author Gavin King
50   * @author Steve Ebersole
51   *
52   * @see JoinFragment
53   */
54  public class JoinSequence {
55  	private final SessionFactoryImplementor factory;
56  
57  	private final StringBuilder conditions = new StringBuilder();
58  	private final List<Join> joins = new ArrayList<Join>();
59  
60  	private boolean useThetaStyle;
61  	private String rootAlias;
62  	private Joinable rootJoinable;
63  	private Selector selector;
64  	private JoinSequence next;
65  	private boolean isFromPart;

76  	/**
77  	 * Retrieve a JoinSequence that represents just the FROM clause parts
78  	 *
79  	 * @return The JoinSequence that represents just the FROM clause parts
80  	 */
81  	public JoinSequence getFromPart() {
82  		final JoinSequence fromPart = new JoinSequence( factory );
83  		fromPart.joins.addAll( this.joins );
84  		fromPart.useThetaStyle = this.useThetaStyle;
85  		fromPart.rootAlias = this.rootAlias;
86  		fromPart.rootJoinable = this.rootJoinable;
87  		fromPart.selector = this.selector;
88  		fromPart.next = this.next == null ? null : this.next.getFromPart();
89  		fromPart.isFromPart = true;
90  		return fromPart;
91  	}

93  	private Set<String> treatAsDeclarations;
94  
95  	public void applyTreatAsDeclarations(Set<String> treatAsDeclarations) {
96  		if ( treatAsDeclarations == null || treatAsDeclarations.isEmpty() ) {
97  			return;
98  		}
99  
100 		if ( this.treatAsDeclarations == null ) {
101 			this.treatAsDeclarations = new HashSet<String>();
102 		}
103 
104 		this.treatAsDeclarations.addAll( treatAsDeclarations );
105 	}

109 	 * Create a full, although shallow, copy.
110 	 *
111 	 * @return The copy
112 	 */
113 	public JoinSequence copy() {
114 		final JoinSequence copy = new JoinSequence( factory );
115 		copy.joins.addAll( this.joins );
116 		copy.useThetaStyle = this.useThetaStyle;
117 		copy.rootAlias = this.rootAlias;
118 		copy.rootJoinable = this.rootJoinable;
119 		copy.selector = this.selector;
120 		copy.next = this.next == null ? null : this.next.copy();
121 		copy.isFromPart = this.isFromPart;
122 		copy.conditions.append( this.conditions.toString() );
123 		return copy;
124 	}

172 	/**
173 	 * Generate a JoinFragment
174 	 *
175 	 * @param enabledFilters The filters associated with the originating session to properly define join conditions
176 	 * @param includeAllSubclassJoins Should all subclass joins be added to the rendered JoinFragment?
177 	 * @param withClauseFragment The with clause (which represents additional join restrictions) fragment
178 	 * @param withClauseJoinAlias The
179 	 *
180 	 * @return The JoinFragment
181 	 *
182 	 * @throws MappingException Indicates a problem access the provided metadata, or incorrect metadata
183 	 */
184 	public JoinFragment toJoinFragment(
185 			Map enabledFilters,
186 			boolean includeAllSubclassJoins,
187 			String withClauseFragment,
188 			String withClauseJoinAlias) throws MappingException {
189 		final QueryJoinFragment joinFragment = new QueryJoinFragment( factory.getDialect(), useThetaStyle );
190 		if ( rootJoinable != null ) {
191 			joinFragment.addCrossJoin( rootJoinable.getTableName(), rootAlias );
192 			final String filterCondition = rootJoinable.filterFragment( rootAlias, enabledFilters, treatAsDeclarations );
193 			// JoinProcessor needs to know if the where clause fragment came from a dynamic filter or not so it
194 			// can put the where clause fragment in the right place in the SQL AST.   'hasFilterCondition' keeps track
195 			// of that fact.
196 			joinFragment.setHasFilterCondition( joinFragment.addCondition( filterCondition ) );
197 			addSubclassJoins( joinFragment, rootAlias, rootJoinable, true, includeAllSubclassJoins, treatAsDeclarations );
198 		}
199 
200 		Joinable last = rootJoinable;
201 
202 		for ( Join join : joins ) {
203 			// technically the treatAsDeclarations should only apply to rootJoinable or to a single Join,
204 			// but that is not possible atm given how these JoinSequence and Join objects are built.
205 			// However, it is generally ok given how the HQL parser builds these JoinSequences (a HQL join
206 			// results in a JoinSequence with an empty rootJoinable and a single Join).  So we use that here
207 			// as an assumption
208 			final String on = join.getAssociationType().getOnCondition( join.getAlias(), factory, enabledFilters, treatAsDeclarations );
209 			String condition;
210 			if ( last != null
211 					&& isManyToManyRoot( last )
212 					&& ((QueryableCollection) last).getElementType() == join.getAssociationType() ) {
213 				// the current join represents the join between a many-to-many association table
214 				// and its "target" table.  Here we need to apply any additional filters
215 				// defined specifically on the many-to-many
216 				final String manyToManyFilter = ( (QueryableCollection) last ).getManyToManyFilterFragment(
217 						join.getAlias(),
218 						enabledFilters
219 				);
220 				condition = "".equals( manyToManyFilter )
221 						? on
222 						: "".equals( on ) ? manyToManyFilter : on + " and " + manyToManyFilter;
223 			}
224 			else {
225 				condition = on;
226 			}
227 
228 			if ( withClauseFragment != null && !isManyToManyRoot( join.joinable )) {
229 				condition += " and " + withClauseFragment;
230 			}
231 
232 			joinFragment.addJoin(
233 					join.getJoinable().getTableName(),
234 					join.getAlias(),
235 					join.getLHSColumns(),
236 					JoinHelper.getRHSColumnNames( join.getAssociationType(), factory ),
237 					join.joinType,
238 					condition
239 			);
240 
241 			addSubclassJoins(
242 					joinFragment,
243 					join.getAlias(),
244 					join.getJoinable(),
245 					join.joinType == JoinType.INNER_JOIN,
246 					includeAllSubclassJoins,
247 					// ugh.. this is needed because of how HQL parser (FromElementFactory/SessionFactoryHelper)
248 					// builds the JoinSequence for HQL joins
249 					treatAsDeclarations
250 			);
251 			last = join.getJoinable();
252 		}
253 
254 		if ( next != null ) {
255 			joinFragment.addFragment( next.toJoinFragment( enabledFilters, includeAllSubclassJoins ) );
256 		}
257 
258 		joinFragment.addCondition( conditions.toString() );
259 
260 		if ( isFromPart ) {
261 			joinFragment.clearWherePart();
262 		}
263 
264 		return joinFragment;
265 	}
266 
267 	@SuppressWarnings("SimplifiableIfStatement")
268 	private boolean isManyToManyRoot(Joinable joinable) {
269 		if ( joinable != null && joinable.isCollection() ) {
270 			return ( (QueryableCollection) joinable ).isManyToMany();
271 		}
272 		return false;
273 	}
274 
275 	private void addSubclassJoins(
276 			JoinFragment joinFragment,
277 			String alias,
278 			Joinable joinable,
279 			boolean innerJoin,
280 			boolean includeSubclassJoins,
281 			Set<String> treatAsDeclarations) {
282 		final boolean include = includeSubclassJoins && isIncluded( alias );
283 		joinFragment.addJoins(
284 				joinable.fromJoinFragment( alias, innerJoin, include, treatAsDeclarations ),
285 				joinable.whereJoinFragment( alias, innerJoin, include, treatAsDeclarations )
286 		);
287 	}
288 
289 	private boolean isIncluded(String alias) {
290 		return selector != null && selector.includeSubclasses( alias );
291 	}

293 	/**
294 	 * Add a condition to this sequence.
295 	 *
296 	 * @param condition The condition
297 	 *
298 	 * @return {@link this}, for method chaining
299 	 */
300 	public JoinSequence addCondition(String condition) {
301 		if ( condition.trim().length() != 0 ) {
302 			if ( !condition.startsWith( " and " ) ) {
303 				conditions.append( " and " );
304 			}
305 			conditions.append( condition );
306 		}
307 		return this;
308 	}
309 
310 	/**
311 	 * Add a condition to this sequence.  Typical usage here might be:
312 	 * <pre>
313 	 *     addCondition( "a", {"c1", "c2"}, "?" )
314 	 * </pre>
315 	 * to represent:
316 	 * <pre>
317 	 *     "... a.c1 = ? and a.c2 = ? ..."
318 	 * </pre>
319 	 *
320 	 * @param alias The alias to apply to the columns
321 	 * @param columns The columns to add checks for
322 	 * @param condition The conditions to check against the columns
323 	 *
324 	 * @return {@link this}, for method chaining
325 	 */
326 	public JoinSequence addCondition(String alias, String[] columns, String condition) {
327 		for ( String column : columns ) {
328 			conditions.append( " and " )
329 					.append( alias )
330 					.append( '.' )
331 					.append( column )
332 					.append( condition );
333 		}
334 		return this;
335 	}

409 	/**
410 	 * Represents a join
411 	 */
412 	public static final class Join {
413 		private final AssociationType associationType;
414 		private final Joinable joinable;
415 		private final JoinType joinType;
416 		private final String alias;
417 		private final String[] lhsColumns;
418 
419 		Join(
420 				SessionFactoryImplementor factory,
421 				AssociationType associationType,
422 				String alias,
423 				JoinType joinType,
424 				String[] lhsColumns) throws MappingException {
425 			this.associationType = associationType;
426 			this.joinable = associationType.getAssociatedJoinable( factory );
427 			this.alias = alias;
428 			this.joinType = joinType;
429 			this.lhsColumns = lhsColumns;
430 		}
431 
432 		public String getAlias() {
433 			return alias;
434 		}
435 
436 		public AssociationType getAssociationType() {
437 			return associationType;
438 		}
439 
440 		public Joinable getJoinable() {
441 			return joinable;
442 		}
443 
444 		public JoinType getJoinType() {
445 			return joinType;
446 		}
447 
448 		public String[] getLHSColumns() {
449 			return lhsColumns;
450 		}
451 
452 		@Override
453 		public String toString() {
454 			return joinable.toString() + '[' + alias + ']';
455 		}
456 	}
457 
458 	@Override
459 	public String toString() {
460 		final StringBuilder buf = new StringBuilder();
461 		buf.append( "JoinSequence{" );
462 		if ( rootJoinable != null ) {
463 			buf.append( rootJoinable )
464 					.append( '[' )
465 					.append( rootAlias )
466 					.append( ']' );
467 		}
468 		for ( Join join : joins ) {
469 			buf.append( "->" ).append( join );
470 		}
471 		return buf.append( '}' ).toString();
472 	}
