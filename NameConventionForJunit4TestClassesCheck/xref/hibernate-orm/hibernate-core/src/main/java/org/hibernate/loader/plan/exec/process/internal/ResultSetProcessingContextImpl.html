74  	/**
75  	 * Builds a ResultSetProcessingContextImpl
76  	 *
77  	 * @param resultSet
78  	 * @param session
79  	 * @param loadPlan
80  	 * @param readOnly
81  	 * @param shouldUseOptionalEntityInformation There are times when the "optional entity information" on
82  	 * QueryParameters should be used and times when they should not.  Collection initializers, batch loaders, etc
83  	 * are times when it should NOT be used.
84  	 * @param forceFetchLazyAttributes
85  	 * @param shouldReturnProxies
86  	 * @param queryParameters
87  	 * @param namedParameterContext
88  	 * @param hadSubselectFetches
89  	 */
90  	public ResultSetProcessingContextImpl(
91  			final ResultSet resultSet,
92  			final SessionImplementor session,
93  			final LoadPlan loadPlan,
94  			final boolean readOnly,
95  			final boolean shouldUseOptionalEntityInformation,
96  			final boolean forceFetchLazyAttributes,
97  			final boolean shouldReturnProxies,
98  			final QueryParameters queryParameters,
99  			final NamedParameterContext namedParameterContext,
100 			final boolean hadSubselectFetches) {
101 		this.resultSet = resultSet;
102 		this.session = session;
103 		this.loadPlan = loadPlan;
104 		this.readOnly = readOnly;
105 		this.shouldUseOptionalEntityInformation = shouldUseOptionalEntityInformation;
106 		this.forceFetchLazyAttributes = forceFetchLazyAttributes;
107 		this.shouldReturnProxies = shouldReturnProxies;
108 		this.queryParameters = queryParameters;
109 		this.namedParameterContext = namedParameterContext;
110 		this.hadSubselectFetches = hadSubselectFetches;
111 
112 		if ( shouldUseOptionalEntityInformation ) {
113 			if ( queryParameters.getOptionalId() != null ) {
114 				// make sure we have only one return
115 				if ( loadPlan.getReturns().size() > 1 ) {
116 					throw new IllegalStateException( "Cannot specify 'optional entity' values with multi-return load plans" );
117 				}
118 			}
119 		}
120 	}

160 	private Map<EntityReference,EntityReferenceProcessingState> identifierResolutionContextMap;

162 	@Override
163 	public EntityReferenceProcessingState getProcessingState(final EntityReference entityReference) {
164 		if ( identifierResolutionContextMap == null ) {
165 			identifierResolutionContextMap = new IdentityHashMap<EntityReference, EntityReferenceProcessingState>();
166 		}
167 
168 		EntityReferenceProcessingState context = identifierResolutionContextMap.get( entityReference );
169 		if ( context == null ) {
170 			context = new EntityReferenceProcessingState() {
171 				private boolean wasMissingIdentifier;
172 				private Object identifierHydratedForm;
173 				private EntityKey entityKey;
174 				private Object[] hydratedState;
175 				private Object entityInstance;
176 
177 				@Override
178 				public EntityReference getEntityReference() {
179 					return entityReference;
180 				}
181 
182 				@Override
183 				public void registerMissingIdentifier() {
184 					if ( !EntityFetch.class.isInstance( entityReference ) ) {
185 						throw new IllegalStateException( "Missing return row identifier" );
186 					}
187 					ResultSetProcessingContextImpl.this.registerNonExists( (EntityFetch) entityReference );
188 					wasMissingIdentifier = true;
189 				}
190 
191 				@Override
192 				public boolean isMissingIdentifier() {
193 					return wasMissingIdentifier;
194 				}
195 
196 				@Override
197 				public void registerIdentifierHydratedForm(Object identifierHydratedForm) {
198 					this.identifierHydratedForm = identifierHydratedForm;
199 				}
200 
201 				@Override
202 				public Object getIdentifierHydratedForm() {
203 					return identifierHydratedForm;
204 				}
205 
206 				@Override
207 				public void registerEntityKey(EntityKey entityKey) {
208 					this.entityKey = entityKey;
209 				}
210 
211 				@Override
212 				public EntityKey getEntityKey() {
213 					return entityKey;
214 				}
215 
216 				@Override
217 				public void registerHydratedState(Object[] hydratedState) {
218 					this.hydratedState = hydratedState;
219 				}
220 
221 				@Override
222 				public Object[] getHydratedState() {
223 					return hydratedState;
224 				}
225 
226 				@Override
227 				public void registerEntityInstance(Object entityInstance) {
228 					this.entityInstance = entityInstance;
229 				}
230 
231 				@Override
232 				public Object getEntityInstance() {
233 					return entityInstance;
234 				}
235 			};
236 			identifierResolutionContextMap.put( entityReference, context );
237 		}
238 
239 		return context;
240 	}

242 	private void registerNonExists(EntityFetch fetch) {
243 		final EntityType fetchedType = fetch.getFetchedType();
244 		if ( ! fetchedType.isOneToOne() ) {
245 			return;
246 		}
247 
248 		final EntityReferenceProcessingState fetchOwnerState = getOwnerProcessingState( fetch );
249 		if ( fetchOwnerState == null ) {
250 			throw new IllegalStateException( "Could not locate fetch owner state" );
251 		}
252 
253 		final EntityKey ownerEntityKey = fetchOwnerState.getEntityKey();
254 		if ( ownerEntityKey == null ) {
255 			throw new IllegalStateException( "Could not locate fetch owner EntityKey" );
256 		}
257 
258 		session.getPersistenceContext().addNullProperty(
259 				ownerEntityKey,
260 				fetchedType.getPropertyName()
261 		);
262 	}
263 
264 	@Override
265 	public EntityReferenceProcessingState getOwnerProcessingState(Fetch fetch) {
266 		return getProcessingState( fetch.getSource().resolveEntityReference() );
267 	}

269 	@Override
270 	public void registerHydratedEntity(EntityReference entityReference, EntityKey entityKey, Object entityInstance) {
271 		if ( currentRowHydratedEntityRegistrationList == null ) {
272 			currentRowHydratedEntityRegistrationList = new ArrayList<HydratedEntityRegistration>();
273 		}
274 		currentRowHydratedEntityRegistrationList.add(
275 				new HydratedEntityRegistration(
276 						entityReference,
277 						entityKey,
278 						entityInstance
279 				)
280 		);
281 	}

283 	/**
284 	 * Package-protected
285 	 */
286 	void finishUpRow() {
287 		if ( currentRowHydratedEntityRegistrationList == null ) {
288 			if ( identifierResolutionContextMap != null ) {
289 				identifierResolutionContextMap.clear();
290 			}
291 			return;
292 		}
293 
294 
295 		// managing the running list of registrations ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
296 		if ( hydratedEntityRegistrationList == null ) {
297 			hydratedEntityRegistrationList = new ArrayList<HydratedEntityRegistration>();
298 		}
299 		hydratedEntityRegistrationList.addAll( currentRowHydratedEntityRegistrationList );
300 
301 
302 		// managing the map forms needed for subselect fetch generation ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
303 		if ( hadSubselectFetches ) {
304 			if ( subselectLoadableEntityKeyMap == null ) {
305 				subselectLoadableEntityKeyMap = new HashMap<EntityPersister, Set<EntityKey>>();
306 			}
307 			for ( HydratedEntityRegistration registration : currentRowHydratedEntityRegistrationList ) {
308 				Set<EntityKey> entityKeys = subselectLoadableEntityKeyMap.get( registration.getEntityReference()
309 																					   .getEntityPersister() );
310 				if ( entityKeys == null ) {
311 					entityKeys = new HashSet<EntityKey>();
312 					subselectLoadableEntityKeyMap.put( registration.getEntityReference().getEntityPersister(), entityKeys );
313 				}
314 				entityKeys.add( registration.getKey() );
315 			}
316 		}
317 
318 		// release the currentRowHydratedEntityRegistrationList entries
319 		currentRowHydratedEntityRegistrationList.clear();
320 
321 		identifierResolutionContextMap.clear();
322 	}
323 
324 	public List<HydratedEntityRegistration> getHydratedEntityRegistrationList() {
325 		return hydratedEntityRegistrationList;
326 	}

328 	/**
329 	 * Package-protected
330 	 */
331 	void wrapUp() {
332 		createSubselects();
333 
334 		if ( hydratedEntityRegistrationList != null ) {
335 			hydratedEntityRegistrationList.clear();
336 			hydratedEntityRegistrationList = null;
337 		}
338 
339 		if ( subselectLoadableEntityKeyMap != null ) {
340 			subselectLoadableEntityKeyMap.clear();
341 			subselectLoadableEntityKeyMap = null;
342 		}
343 	}
344 
345 	private void createSubselects() {
346 		if ( subselectLoadableEntityKeyMap == null || subselectLoadableEntityKeyMap.size() <= 1 ) {
347 			// if we only returned one entity, query by key is more efficient; so do nothing here
348 			return;
349 		}
350 
351 		final Map<String, int[]> namedParameterLocMap =
352 				ResultSetProcessorHelper.buildNamedParameterLocMap( queryParameters, namedParameterContext );
353 
354 		for ( Map.Entry<EntityPersister, Set<EntityKey>> entry : subselectLoadableEntityKeyMap.entrySet() ) {
355 			if ( ! entry.getKey().hasSubselectLoadableCollections() ) {
356 				continue;
357 			}
358 
359 			SubselectFetch subselectFetch = new SubselectFetch(
360 					//getSQLString(),
361 					null, // aliases[i],
362 					(Loadable) entry.getKey(),
363 					queryParameters,
364 					entry.getValue(),
365 					namedParameterLocMap
366 			);
367 
368 			for ( EntityKey key : entry.getValue() ) {
369 				session.getPersistenceContext().getBatchFetchQueue().addSubselect( key, subselectFetch );
370 			}
371 
372 		}
373 	}
