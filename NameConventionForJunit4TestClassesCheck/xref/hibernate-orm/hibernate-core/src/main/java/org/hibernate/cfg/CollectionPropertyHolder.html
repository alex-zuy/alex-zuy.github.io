87  	private void buildAttributeConversionInfoMaps(
88  			XProperty collectionProperty,
89  			Map<String,AttributeConversionInfo> elementAttributeConversionInfoMap,
90  			Map<String,AttributeConversionInfo> keyAttributeConversionInfoMap) {
91  		if ( collectionProperty == null ) {
92  			// not sure this is valid condition
93  			return;
94  		}
95  
96  		{
97  			final Convert convertAnnotation = collectionProperty.getAnnotation( Convert.class );
98  			if ( convertAnnotation != null ) {
99  				applyLocalConvert( convertAnnotation, collectionProperty, elementAttributeConversionInfoMap, keyAttributeConversionInfoMap );
100 			}
101 		}
102 
103 		{
104 			final Converts convertsAnnotation = collectionProperty.getAnnotation( Converts.class );
105 			if ( convertsAnnotation != null ) {
106 				for ( Convert convertAnnotation : convertsAnnotation.value() ) {
107 					applyLocalConvert(
108 							convertAnnotation,
109 							collectionProperty,
110 							elementAttributeConversionInfoMap,
111 							keyAttributeConversionInfoMap
112 					);
113 				}
114 			}
115 		}
116 	}

118 	private void applyLocalConvert(
119 			Convert convertAnnotation,
120 			XProperty collectionProperty,
121 			Map<String,AttributeConversionInfo> elementAttributeConversionInfoMap,
122 			Map<String,AttributeConversionInfo> keyAttributeConversionInfoMap) {
123 
124 		// IMPL NOTE : the rules here are quite more lenient than what JPA says.  For example, JPA says
125 		// that @Convert on a Map always needs to specify attributeName of key/value (or prefixed with
126 		// key./value. for embedded paths).  However, we try to see if conversion of either is disabled
127 		// for whatever reason.  For example, if the Map is annotated with @Enumerated the elements cannot
128 		// be converted so any @Convert likely meant the key, so we apply it to the key
129 
130 		final AttributeConversionInfo info = new AttributeConversionInfo( convertAnnotation, collectionProperty );
131 		if ( collection.isMap() ) {
132 			boolean specCompliant = StringHelper.isNotEmpty( info.getAttributeName() )
133 					&& ( info.getAttributeName().startsWith( "key" )
134 					|| info.getAttributeName().startsWith( "value" ) );
135 			if ( !specCompliant ) {
136 				log.nonCompliantMapConversion( collection.getRole() );
137 			}
138 		}
139 
140 		if ( StringHelper.isEmpty( info.getAttributeName() ) ) {
141 			if ( canElementBeConverted && canKeyBeConverted ) {
142 				throw new IllegalStateException(
143 						"@Convert placed on Map attribute [" + collection.getRole()
144 								+ "] must define attributeName of 'key' or 'value'"
145 				);
146 			}
147 			else if ( canKeyBeConverted ) {
148 				keyAttributeConversionInfoMap.put( "", info );
149 			}
150 			else if ( canElementBeConverted ) {
151 				elementAttributeConversionInfoMap.put( "", info );
152 			}
153 			// if neither, we should not be here...
154 		}
155 		else {
156 			if ( canElementBeConverted && canKeyBeConverted ) {
157 				// specified attributeName needs to have 'key.' or 'value.' prefix
158 				if ( info.getAttributeName().startsWith( "key." ) ) {
159 					keyAttributeConversionInfoMap.put(
160 							info.getAttributeName().substring( 4 ),
161 							info
162 					);
163 				}
164 				else if ( info.getAttributeName().startsWith( "value." ) ) {
165 					elementAttributeConversionInfoMap.put(
166 							info.getAttributeName().substring( 6 ),
167 							info
168 					);
169 				}
170 				else {
171 					throw new IllegalStateException(
172 							"@Convert placed on Map attribute [" + collection.getRole()
173 									+ "] must define attributeName of 'key' or 'value'"
174 					);
175 				}
176 			}
177 		}
178 	}

277 	public void prepare(XProperty collectionProperty) {
278 		// fugly
279 		if ( prepared ) {
280 			return;
281 		}
282 
283 		if ( collectionProperty == null ) {
284 			return;
285 		}
286 
287 		prepared = true;
288 
289 		if ( collection.isMap() ) {
290 			if ( collectionProperty.isAnnotationPresent( MapKeyEnumerated.class ) ) {
291 				canKeyBeConverted = false;
292 			}
293 			else if ( collectionProperty.isAnnotationPresent( MapKeyTemporal.class ) ) {
294 				canKeyBeConverted = false;
295 			}
296 			else if ( collectionProperty.isAnnotationPresent( MapKeyClass.class ) ) {
297 				canKeyBeConverted = false;
298 			}
299 			else if ( collectionProperty.isAnnotationPresent( MapKeyType.class ) ) {
300 				canKeyBeConverted = false;
301 			}
302 		}
303 		else {
304 			canKeyBeConverted = false;
305 		}
306 
307 		if ( collectionProperty.isAnnotationPresent( ManyToAny.class ) ) {
308 			canElementBeConverted = false;
309 		}
310 		else if ( collectionProperty.isAnnotationPresent( OneToMany.class ) ) {
311 			canElementBeConverted = false;
312 		}
313 		else if ( collectionProperty.isAnnotationPresent( ManyToMany.class ) ) {
314 			canElementBeConverted = false;
315 		}
316 		else if ( collectionProperty.isAnnotationPresent( Enumerated.class ) ) {
317 			canElementBeConverted = false;
318 		}
319 		else if ( collectionProperty.isAnnotationPresent( Temporal.class ) ) {
320 			canElementBeConverted = false;
321 		}
322 		else if ( collectionProperty.isAnnotationPresent( CollectionType.class ) ) {
323 			canElementBeConverted = false;
324 		}
325 
326 		// Is it valid to reference a collection attribute in a @Convert attached to the owner (entity) by path?
327 		// if so we should pass in 'clazzToProcess' also
328 		if ( canKeyBeConverted || canElementBeConverted ) {
329 			buildAttributeConversionInfoMaps( collectionProperty, elementAttributeConversionInfoMap, keyAttributeConversionInfoMap );
330 		}
331 	}

333 	public AttributeConverterDefinition resolveElementAttributeConverterDefinition(XClass elementXClass) {
334 		AttributeConversionInfo info = locateAttributeConversionInfo( "element" );
335 		if ( info != null ) {
336 			if ( info.isConversionDisabled() ) {
337 				return null;
338 			}
339 			else {
340 				try {
341 					return makeAttributeConverterDefinition( info );
342 				}
343 				catch (Exception e) {
344 					throw new IllegalStateException(
345 							String.format( "Unable to instantiate AttributeConverter [%s", info.getConverterClass().getName() ),
346 							e
347 					);
348 				}
349 			}
350 		}
351 
352 		log.debugf(
353 				"Attempting to locate auto-apply AttributeConverter for collection element [%s]",
354 				collection.getRole()
355 		);
356 
357 		final Class elementClass = determineElementClass( elementXClass );
358 		if ( elementClass != null ) {
359 			for ( AttributeConverterDefinition attributeConverterDefinition : getMappings().getAttributeConverters() ) {
360 				if ( ! attributeConverterDefinition.isAutoApply() ) {
361 					continue;
362 				}
363 				log.debugf(
364 						"Checking auto-apply AttributeConverter [%s] type [%s] for match [%s]",
365 						attributeConverterDefinition.toString(),
366 						attributeConverterDefinition.getEntityAttributeType().getSimpleName(),
367 						elementClass.getSimpleName()
368 				);
369 				if ( areTypeMatch( attributeConverterDefinition.getEntityAttributeType(), elementClass ) ) {
370 					return attributeConverterDefinition;
371 				}
372 			}
373 		}
374 
375 		return null;
376 	}

378 	private Class determineElementClass(XClass elementXClass) {
379 		if ( elementXClass != null ) {
380 			try {
381 				return getMappings().getReflectionManager().toClass( elementXClass );
382 			}
383 			catch (Exception e) {
384 				log.debugf(
385 						"Unable to resolve XClass [%s] to Class for collection elements [%s]",
386 						elementXClass.getName(),
387 						collection.getRole()
388 				);
389 			}
390 		}
391 
392 		if ( collection.getElement() != null ) {
393 			if ( collection.getElement().getType() != null ) {
394 				return collection.getElement().getType().getReturnedClass();
395 			}
396 		}
397 
398 		// currently this is called from paths where the element type really should be known,
399 		// so log the fact that we could not resolve the collection element info
400 		log.debugf(
401 				"Unable to resolve element information for collection [%s]",
402 				collection.getRole()
403 		);
404 		return null;
405 	}

408 		AttributeConversionInfo info = locateAttributeConversionInfo( "key" );
409 		if ( info != null ) {
410 			if ( info.isConversionDisabled() ) {
411 				return null;

414 				try {
415 					return makeAttributeConverterDefinition( info );

431 		final Class elementClass = determineKeyClass( keyXClass );
432 		if ( elementClass != null ) {
433 			for ( AttributeConverterDefinition attributeConverterDefinition : getMappings().getAttributeConverters() ) {
434 				if ( ! attributeConverterDefinition.isAutoApply() ) {

439 						attributeConverterDefinition.toString(),

444 					return attributeConverterDefinition;
445 				}
446 			}

452 	private Class determineKeyClass(XClass keyXClass) {
453 		if ( keyXClass != null ) {
454 			try {
455 				return getMappings().getReflectionManager().toClass( keyXClass );
456 			}
457 			catch (Exception e) {
458 				log.debugf(
459 						"Unable to resolve XClass [%s] to Class for collection key [%s]",
460 						keyXClass.getName(),
461 						collection.getRole()
462 				);
463 			}
464 		}
465 
466 		final IndexedCollection indexedCollection = (IndexedCollection) collection;
467 		if ( indexedCollection.getIndex() != null ) {
468 			if ( indexedCollection.getIndex().getType() != null ) {
469 				return indexedCollection.getIndex().getType().getReturnedClass();
470 			}
471 		}
472 
473 		// currently this is called from paths where the element type really should be known,
474 		// so log the fact that we could not resolve the collection element info
475 		log.debugf(
476 				"Unable to resolve key information for collection [%s]",
477 				collection.getRole()
478 		);
479 		return null;
480 	}
