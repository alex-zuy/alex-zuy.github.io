64  public class LoadPlanBuilderTest extends BaseCoreFunctionalTestCase {
65  	@Override
66  	protected Class<?>[] getAnnotatedClasses() {
67  		return new Class[] { Message.class, Poster.class };
68  	}
69  
70  	@Test
71  	public void testSimpleBuild() {
72  		EntityPersister ep = (EntityPersister) sessionFactory().getClassMetadata(Message.class);
73  		FetchStyleLoadPlanBuildingAssociationVisitationStrategy strategy = new FetchStyleLoadPlanBuildingAssociationVisitationStrategy(
74  				sessionFactory(),
75  				LoadQueryInfluencers.NONE,
76  				LockMode.NONE
77  		);
78  		LoadPlan plan = MetamodelDrivenLoadPlanBuilder.buildRootEntityLoadPlan( strategy, ep );
79  		assertFalse( plan.hasAnyScalarReturns() );
80  		assertEquals( 1, plan.getReturns().size() );
81  		Return rtn = plan.getReturns().get( 0 );
82  		EntityReturn entityReturn = ExtraAssertions.assertTyping( EntityReturn.class, rtn );
83  		assertNotNull( entityReturn.getFetches() );
84  		assertEquals( 1, entityReturn.getFetches().length );
85  		Fetch fetch = entityReturn.getFetches()[0];
86  		EntityFetch entityFetch = ExtraAssertions.assertTyping( EntityFetch.class, fetch );
87  		assertNotNull( entityFetch.getFetches() );
88  		assertEquals( 0, entityFetch.getFetches().length );
89  
90  		LoadPlanTreePrinter.INSTANCE.logTree( plan, new AliasResolutionContextImpl( sessionFactory() ) );
91  	}
92  
93  	@Test
94  	public void testCascadeBasedBuild() {
95  		EntityPersister ep = (EntityPersister) sessionFactory().getClassMetadata(Message.class);
96  		CascadeStyleLoadPlanBuildingAssociationVisitationStrategy strategy = new CascadeStyleLoadPlanBuildingAssociationVisitationStrategy(
97  				CascadingActions.MERGE,
98  				sessionFactory(),
99  				LoadQueryInfluencers.NONE,
100 				LockMode.NONE
101 		);
102 		LoadPlan plan = MetamodelDrivenLoadPlanBuilder.buildRootEntityLoadPlan( strategy, ep );
103 		assertFalse( plan.hasAnyScalarReturns() );
104 		assertEquals( 1, plan.getReturns().size() );
105 		Return rtn = plan.getReturns().get( 0 );
106 		EntityReturn entityReturn = ExtraAssertions.assertTyping( EntityReturn.class, rtn );
107 		assertNotNull( entityReturn.getFetches() );
108 		assertEquals( 1, entityReturn.getFetches().length );
109 		Fetch fetch = entityReturn.getFetches()[0];
110 		EntityFetch entityFetch = ExtraAssertions.assertTyping( EntityFetch.class, fetch );
111 		assertNotNull( entityFetch.getFetches() );
112 		assertEquals( 0, entityFetch.getFetches().length );
113 
114 		LoadPlanTreePrinter.INSTANCE.logTree( plan, new AliasResolutionContextImpl( sessionFactory() ) );
115 	}
116 
117 	@Test
118 	public void testCollectionInitializerCase() {
119 		CollectionPersister cp = sessionFactory().getCollectionPersister( Poster.class.getName() + ".messages" );
120 		FetchStyleLoadPlanBuildingAssociationVisitationStrategy strategy = new FetchStyleLoadPlanBuildingAssociationVisitationStrategy(
121 				sessionFactory(),
122 				LoadQueryInfluencers.NONE,
123 				LockMode.NONE
124 		);
125 		LoadPlan plan = MetamodelDrivenLoadPlanBuilder.buildRootCollectionLoadPlan( strategy, cp );
126 		assertFalse( plan.hasAnyScalarReturns() );
127 		assertEquals( 1, plan.getReturns().size() );
128 		Return rtn = plan.getReturns().get( 0 );
129 		CollectionReturn collectionReturn = ExtraAssertions.assertTyping( CollectionReturn.class, rtn );
130 
131 		assertNotNull( collectionReturn.getElementGraph() );
132 		assertNotNull( collectionReturn.getElementGraph().getFetches() );
133 		// the collection Message elements are fetched, but Message.poster is not fetched
134 		// (because that collection is owned by that Poster)
135 		assertEquals( 0, collectionReturn.getElementGraph().getFetches().length );
136 		EntityReference entityReference = ExtraAssertions.assertTyping( EntityReference.class, collectionReturn.getElementGraph() );
137 		assertNotNull( entityReference.getFetches() );
138 		assertEquals( 0, entityReference.getFetches().length );
139 
140 		LoadPlanTreePrinter.INSTANCE.logTree( plan, new AliasResolutionContextImpl( sessionFactory() ) );
141 	}

154 	public static class Poster {
155 		@Id

157 		private String name;
158 		@OneToMany(mappedBy = "poster")
159 		private List<Message> messages;
160 	}
