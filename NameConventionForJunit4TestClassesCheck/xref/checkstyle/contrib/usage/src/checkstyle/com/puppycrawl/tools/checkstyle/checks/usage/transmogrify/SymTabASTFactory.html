29  package com.puppycrawl.tools.checkstyle.checks.usage.transmogrify;
30  
31  import antlr.ASTFactory;
32  import antlr.collections.AST;
33  
34  /**
35   * Factory for <code>SymTabAST</code> objects.
36   * @author Rick Giles
37   */
38  public class SymTabASTFactory
39  {
40      /** singleton factory */
41      private static ASTFactory factory;
42      
43      static
44      {
45          factory = new ASTFactory();
46          factory.setASTNodeClass(SymTabAST.class.getName());
47      }
48      
49      /**
50       * Creates a <code>SymTabAST</code> with a given type and text.
51       * @param aType the type for the new <code>SymTabAST</code>.
52       * @param aText the text for the new <code>SymTabAST</code>.
53       * @return the new <code>SymTabAST</code>.
54       */
55      public static SymTabAST create(int aType, String aText)
56      {
57          return (SymTabAST) factory.create(aType, aText);
58      }
59  
60      /**
61       * Creates an <code>SymTabAST</code> from a given <code>AST</code>.
62       * @param aAST the <code>AST</code> for the new <code>SymTabAST</code>.
63       * @return the new <code>SymTabAST</code>.
64       */    
65      public static SymTabAST create(AST aAST)
66      {
67          return (SymTabAST) factory.create(aAST);
68      }
69      
70      ///CLOVER:OFF
71      /** prevent instantiation */
72      private SymTabASTFactory()
73      {
74      }
75      ///CLOVER:ON
76  }
