50  /**
51   * A builder used to create {@link ObjectMapper} instances with a fluent API.
52   *
53   * <p>It customizes Jackson's default properties with the following ones:
54   * <ul>
55   * <li>{@link MapperFeature#DEFAULT_VIEW_INCLUSION} is disabled</li>
56   * <li>{@link DeserializationFeature#FAIL_ON_UNKNOWN_PROPERTIES} is disabled</li>
57   * </ul>
58   *
59   * <p>Note that Jackson's JSR-310 and Joda-Time support modules will be registered automatically
60   * when available (and when Java 8 and Joda-Time themselves are available, respectively).
61   *
62   * <p>Tested against Jackson 2.2, 2.3 and 2.4; compatible with Jackson 2.0 and higher.
63   *
64   * @author Sebastien Deleuze
65   * @author Juergen Hoeller
66   * @author Tadaya Tsuyukubo
67   * @since 4.1.1
68   * @see #build()
69   * @see #configure(ObjectMapper)
70   * @see Jackson2ObjectMapperFactoryBean
71   */

163 	/**
164 	 * Configure custom serializers. Each serializer is registered for the type
165 	 * returned by {@link JsonSerializer#handledType()}, which must not be
166 	 * {@code null}.
167 	 * @see #serializersByType(Map)
168 	 */
169 	public Jackson2ObjectMapperBuilder serializers(JsonSerializer<?>... serializers) {
170 		if (serializers != null) {
171 			for (JsonSerializer<?> serializer : serializers) {
172 				Class<?> handledType = serializer.handledType();
173 				if (handledType == null || handledType == Object.class) {
174 					throw new IllegalArgumentException("Unknown handled type in " + serializer.getClass().getName());
175 				}
176 				this.serializers.put(serializer.handledType(), serializer);
177 			}
178 		}
179 		return this;
180 	}

411 	/**
412 	 * Build a new {@link ObjectMapper} instance.
413 	 * <p>Each build operation produces an independent {@link ObjectMapper} instance.
414 	 * The builder's settings can get modified, with a subsequent build operation
415 	 * then producing a new {@link ObjectMapper} based on the most recent settings.
416 	 * @return the newly built ObjectMapper
417 	 */
418 	@SuppressWarnings("unchecked")
419 	public <T extends ObjectMapper> T build() {
420 		ObjectMapper objectMapper;
421 		if (this.createXmlMapper) {
422 			try {
423 				Class<? extends ObjectMapper> xmlMapper = (Class<? extends ObjectMapper>)
424 						ClassUtils.forName("com.fasterxml.jackson.dataformat.xml.XmlMapper", this.moduleClassLoader);
425 				objectMapper = BeanUtils.instantiate(xmlMapper);
426 			}
427 			catch (ClassNotFoundException ex) {
428 				throw new IllegalStateException("Could not instantiate XmlMapper - not found on classpath");
429 			}
430 		}
431 		else {
432 			objectMapper = new ObjectMapper();
433 		}
434 		configure(objectMapper);
435 		return (T) objectMapper;
436 	}

438 	/**
439 	 * Configure an existing {@link ObjectMapper} instance with this builder's
440 	 * settings. This can be applied to any number of {@code ObjectMappers}.
441 	 * @param objectMapper the ObjectMapper to configure
442 	 */
443 	@SuppressWarnings("deprecation")
444 	public void configure(ObjectMapper objectMapper) {
445 		Assert.notNull(objectMapper, "ObjectMapper must not be null");
446 
447 		if (this.dateFormat != null) {
448 			objectMapper.setDateFormat(this.dateFormat);
449 		}
450 
451 		if (this.annotationIntrospector != null) {
452 			objectMapper.setAnnotationIntrospector(this.annotationIntrospector);
453 		}
454 
455 		if (this.serializationInclusion != null) {
456 			objectMapper.setSerializationInclusion(this.serializationInclusion);
457 		}
458 
459 		if (!this.serializers.isEmpty() || !this.deserializers.isEmpty()) {
460 			SimpleModule module = new SimpleModule();
461 			addSerializers(module);
462 			addDeserializers(module);
463 			objectMapper.registerModule(module);
464 		}
465 
466 		customizeDefaultFeatures(objectMapper);
467 		for (Object feature : this.features.keySet()) {
468 			configureFeature(objectMapper, feature, this.features.get(feature));
469 		}
470 
471 		if (this.modules != null) {
472 			// Complete list of modules given
473 			for (Module module : this.modules) {
474 				// Using Jackson 2.0+ registerModule method, not Jackson 2.2+ registerModules
475 				objectMapper.registerModule(module);
476 			}
477 		}
478 		else {
479 			// Combination of modules by class names specified and class presence in the classpath
480 			if (this.modulesToInstall != null) {
481 				for (Class<? extends Module> module : this.modulesToInstall) {
482 					objectMapper.registerModule(BeanUtils.instantiate(module));
483 				}
484 			}
485 			if (this.findModulesViaServiceLoader) {
486 				// Jackson 2.2+
487 				objectMapper.registerModules(ObjectMapper.findModules(this.moduleClassLoader));
488 			}
489 			else {
490 				registerWellKnownModulesIfAvailable(objectMapper);
491 			}
492 		}
493 
494 		if (this.propertyNamingStrategy != null) {
495 			objectMapper.setPropertyNamingStrategy(this.propertyNamingStrategy);
496 		}
497 		for (Class<?> target : this.mixIns.keySet()) {
498 			// Deprecated as of Jackson 2.5, but just in favor of a fluent variant.
499 			objectMapper.addMixInAnnotations(target, this.mixIns.get(target));
500 		}
501 		if (this.handlerInstantiator != null) {
502 			objectMapper.setHandlerInstantiator(this.handlerInstantiator);
503 		}
504 		else if (this.applicationContext != null) {
505 			objectMapper.setHandlerInstantiator(
506 					new SpringHandlerInstantiator(this.applicationContext.getAutowireCapableBeanFactory()));
507 		}
508 	}

510 	// Any change to this method should be also applied to spring-jms and spring-messaging
511 	// MappingJackson2MessageConverter default constructors
512 	private void customizeDefaultFeatures(ObjectMapper objectMapper) {
513 		if (!this.features.containsKey(MapperFeature.DEFAULT_VIEW_INCLUSION)) {
514 			configureFeature(objectMapper, MapperFeature.DEFAULT_VIEW_INCLUSION, false);
515 		}
516 		if (!this.features.containsKey(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES)) {
517 			configureFeature(objectMapper, DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);
518 		}
519 	}

521 	@SuppressWarnings("unchecked")
522 	private <T> void addSerializers(SimpleModule module) {
523 		for (Class<?> type : this.serializers.keySet()) {
524 			module.addSerializer((Class<? extends T>) type, (JsonSerializer<T>) this.serializers.get(type));
525 		}
526 	}
527 
528 	@SuppressWarnings("unchecked")
529 	private <T> void addDeserializers(SimpleModule module) {
530 		for (Class<?> type : this.deserializers.keySet()) {
531 			module.addDeserializer((Class<T>) type, (JsonDeserializer<? extends T>) this.deserializers.get(type));
532 		}
533 	}
534 
535 	private void configureFeature(ObjectMapper objectMapper, Object feature, boolean enabled) {
536 		if (feature instanceof JsonParser.Feature) {
537 			objectMapper.configure((JsonParser.Feature) feature, enabled);
538 		}
539 		else if (feature instanceof JsonGenerator.Feature) {
540 			objectMapper.configure((JsonGenerator.Feature) feature, enabled);
541 		}
542 		else if (feature instanceof SerializationFeature) {
543 			objectMapper.configure((SerializationFeature) feature, enabled);
544 		}
545 		else if (feature instanceof DeserializationFeature) {
546 			objectMapper.configure((DeserializationFeature) feature, enabled);
547 		}
548 		else if (feature instanceof MapperFeature) {
549 			objectMapper.configure((MapperFeature) feature, enabled);
550 		}
551 		else {
552 			throw new FatalBeanException("Unknown feature class: " + feature.getClass().getName());
553 		}
554 	}

556 	@SuppressWarnings("unchecked")
557 	private void registerWellKnownModulesIfAvailable(ObjectMapper objectMapper) {
558 		// Java 8 java.time package present?
559 		if (ClassUtils.isPresent("java.time.LocalDate", this.moduleClassLoader)) {
560 			try {
561 				Class<? extends Module> jsr310Module = (Class<? extends Module>)
562 						ClassUtils.forName("com.fasterxml.jackson.datatype.jsr310.JSR310Module", this.moduleClassLoader);
563 				objectMapper.registerModule(BeanUtils.instantiate(jsr310Module));
564 			}
565 			catch (ClassNotFoundException ex) {
566 				// jackson-datatype-jsr310 not available
567 			}
568 		}
569 		// Joda-Time present?
570 		if (ClassUtils.isPresent("org.joda.time.LocalDate", this.moduleClassLoader)) {
571 			try {
572 				Class<? extends Module> jodaModule = (Class<? extends Module>)
573 						ClassUtils.forName("com.fasterxml.jackson.datatype.joda.JodaModule", this.moduleClassLoader);
574 				objectMapper.registerModule(BeanUtils.instantiate(jodaModule));
575 			}
576 			catch (ClassNotFoundException ex) {
577 				// jackson-datatype-joda not available
578 			}
579 		}
580 	}
