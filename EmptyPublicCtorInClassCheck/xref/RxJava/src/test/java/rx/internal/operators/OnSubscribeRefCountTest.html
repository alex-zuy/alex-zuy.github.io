61      @Test
62      public void testRefCountAsync() {
63          final AtomicInteger subscribeCount = new AtomicInteger();
64          final AtomicInteger nextCount = new AtomicInteger();
65          Observable<Long> r = Observable.timer(0, 1, TimeUnit.MILLISECONDS)
66                  .doOnSubscribe(new Action0() {
67  
68                      @Override
69                      public void call() {
70                          subscribeCount.incrementAndGet();
71                      }
72  
73                  })
74                  .doOnNext(new Action1<Long>() {
75  
76                      @Override
77                      public void call(Long l) {
78                          nextCount.incrementAndGet();
79                      }
80  
81                  })
82                  .publish().refCount();
83  
84          final AtomicInteger receivedCount = new AtomicInteger();
85          Subscription s1 = r.subscribe(new Action1<Long>() {
86  
87              @Override
88              public void call(Long l) {
89                  receivedCount.incrementAndGet();
90              }
91  
92          });
93          Subscription s2 = r.subscribe();
94  
95          // give time to emit
96          try {
97              Thread.sleep(50);
98          } catch (InterruptedException e) {
99          }
100 
101         // now unsubscribe
102         s2.unsubscribe(); // unsubscribe s2 first as we're counting in 1 and there can be a race between unsubscribe and one subscriber getting a value but not the other
103         s1.unsubscribe();
104 
105         System.out.println("onNext: " + nextCount.get());
106 
107         // should emit once for both subscribers
108         assertEquals(nextCount.get(), receivedCount.get());
109         // only 1 subscribe
110         assertEquals(1, subscribeCount.get());
111     }

116         final AtomicInteger nextCount = new AtomicInteger();

130                         nextCount.incrementAndGet();
131                     }

134                 .publish().refCount();
135 
136         final AtomicInteger receivedCount = new AtomicInteger();
137         Subscription s1 = r.subscribe(new Action1<Integer>() {
138 
139             @Override
140             public void call(Integer l) {
141                 receivedCount.incrementAndGet();
142             }
143 
144         });
145         Subscription s2 = r.subscribe();

150         } catch (InterruptedException e) {

197     @Test
198     public void testRepeat() {
199         final AtomicInteger subscribeCount = new AtomicInteger();
200         final AtomicInteger unsubscribeCount = new AtomicInteger();
201         Observable<Long> r = Observable.timer(0, 1, TimeUnit.MILLISECONDS)
202                 .doOnSubscribe(new Action0() {
203 
204                     @Override
205                     public void call() {
206                         System.out.println("******************************* Subscribe received");
207                         // when we are subscribed
208                         subscribeCount.incrementAndGet();
209                     }
210 
211                 })
212                 .doOnUnsubscribe(new Action0() {
213 
214                     @Override
215                     public void call() {
216                         System.out.println("******************************* Unsubscribe received");
217                         // when we are unsubscribed
218                         unsubscribeCount.incrementAndGet();
219                     }
220 
221                 })
222                 .publish().refCount();
223 
224         for (int i = 0; i < 10; i++) {
225             TestSubscriber<Long> ts1 = new TestSubscriber<Long>();
226             TestSubscriber<Long> ts2 = new TestSubscriber<Long>();
227             r.subscribe(ts1);
228             r.subscribe(ts2);
229             try {
230                 Thread.sleep(50);
231             } catch (InterruptedException e) {
232             }
233             ts1.unsubscribe();
234             ts2.unsubscribe();
235             ts1.assertNoErrors();
236             ts2.assertNoErrors();
237             assertTrue(ts1.getOnNextEvents().size() > 0);
238             assertTrue(ts2.getOnNextEvents().size() > 0);
239         }
240 
241         assertEquals(10, subscribeCount.get());
242         assertEquals(10, unsubscribeCount.get());
243     }

245     @Test
246     public void testConnectUnsubscribe() throws InterruptedException {
247         final CountDownLatch unsubscribeLatch = new CountDownLatch(1);
248         final CountDownLatch subscribeLatch = new CountDownLatch(1);
249         Observable<Long> o = synchronousInterval()
250                 .doOnSubscribe(new Action0() {
251 
252                     @Override
253                     public void call() {
254                         System.out.println("******************************* Subscribe received");
255                         // when we are subscribed
256                         subscribeLatch.countDown();
257                     }
258 
259                 })
260                 .doOnUnsubscribe(new Action0() {
261 
262                     @Override
263                     public void call() {
264                         System.out.println("******************************* Unsubscribe received");
265                         // when we are unsubscribed
266                         unsubscribeLatch.countDown();
267                     }
268 
269                 });
270         TestSubscriber<Long> s = new TestSubscriber<Long>();
271         o.publish().refCount().subscribeOn(Schedulers.newThread()).subscribe(s);
272         System.out.println("send unsubscribe");
273         // wait until connected
274         subscribeLatch.await();
275         // now unsubscribe
276         s.unsubscribe();
277         System.out.println("DONE sending unsubscribe ... now waiting");
278         if (!unsubscribeLatch.await(3000, TimeUnit.MILLISECONDS)) {
279             System.out.println("Errors: " + s.getOnErrorEvents());
280             if (s.getOnErrorEvents().size() > 0) {
281                 s.getOnErrorEvents().get(0).printStackTrace();
282             }
283             fail("timed out waiting for unsubscribe");
284         }
285         s.assertNoErrors();
286     }

288     @Test
289     public void testConnectUnsubscribeRaceCondition() throws InterruptedException {
290         final AtomicInteger subUnsubCount = new AtomicInteger();
291         Observable<Long> o = synchronousInterval()
292                 .doOnUnsubscribe(new Action0() {
293 
294                     @Override
295                     public void call() {
296                         System.out.println("******************************* Unsubscribe received");
297                         // when we are unsubscribed
298                         subUnsubCount.decrementAndGet();
299                     }
300 
301                 })
302                 .doOnSubscribe(new Action0() {
303 
304                     @Override
305                     public void call() {
306                         System.out.println("******************************* SUBSCRIBE received");
307                         subUnsubCount.incrementAndGet();
308                     }
309 
310                 });
311 
312         TestSubscriber<Long> s = new TestSubscriber<Long>();
313         o.publish().refCount().subscribeOn(Schedulers.newThread()).subscribe(s);
314         System.out.println("send unsubscribe");
315         // now immediately unsubscribe while subscribeOn is racing to subscribe
316         s.unsubscribe();
317         // this generally will mean it won't even subscribe as it is already unsubscribed by the time connect() gets scheduled
318 
319         // either we subscribed and then unsubscribed, or we didn't ever even subscribe
320         assertEquals(0, subUnsubCount.get());
321 
322         System.out.println("DONE sending unsubscribe ... now waiting");
323         System.out.println("Errors: " + s.getOnErrorEvents());
324         if (s.getOnErrorEvents().size() > 0) {
325             s.getOnErrorEvents().get(0).printStackTrace();
326         }
327         s.assertNoErrors();
328     }

330     private Observable<Long> synchronousInterval() {
331         return Observable.create(new OnSubscribe<Long>() {
332 
333             @Override
334             public void call(Subscriber<? super Long> subscriber) {
335                 while (!subscriber.isUnsubscribed()) {
336                     try {
337                         Thread.sleep(100);
338                     } catch (InterruptedException e) {
339                     }
340                     subscriber.onNext(1L);
341                 }
342             }
343         });
344     }

346     @Test
347     public void onlyFirstShouldSubscribeAndLastUnsubscribe() {
348         final AtomicInteger subscriptionCount = new AtomicInteger();
349         final AtomicInteger unsubscriptionCount = new AtomicInteger();
350         Observable<Integer> observable = Observable.create(new OnSubscribe<Integer>() {
351             @Override
352             public void call(Subscriber<? super Integer> observer) {
353                 subscriptionCount.incrementAndGet();
354                 observer.add(Subscriptions.create(new Action0() {
355                     @Override
356                     public void call() {
357                         unsubscriptionCount.incrementAndGet();
358                     }
359                 }));
360             }
361         });
362         Observable<Integer> refCounted = observable.publish().refCount();
363         @SuppressWarnings("unchecked")
364         Observer<Integer> observer = mock(Observer.class);
365         Subscription first = refCounted.subscribe(observer);
366         assertEquals(1, subscriptionCount.get());
367         Subscription second = refCounted.subscribe(observer);
368         assertEquals(1, subscriptionCount.get());
369         first.unsubscribe();
370         assertEquals(0, unsubscriptionCount.get());
371         second.unsubscribe();
372         assertEquals(1, unsubscriptionCount.get());
373     }

375     @Test
376     public void testRefCount() {
377         TestScheduler s = new TestScheduler();
378         Observable<Long> interval = Observable.interval(100, TimeUnit.MILLISECONDS, s).publish().refCount();
379 
380         // subscribe list1
381         final List<Long> list1 = new ArrayList<Long>();
382         Subscription s1 = interval.subscribe(new Action1<Long>() {
383 
384             @Override
385             public void call(Long t1) {
386                 list1.add(t1);
387             }
388 
389         });
390         s.advanceTimeBy(200, TimeUnit.MILLISECONDS);
391 
392         assertEquals(2, list1.size());
393         assertEquals(0L, list1.get(0).longValue());
394         assertEquals(1L, list1.get(1).longValue());
395 
396         // subscribe list2
397         final List<Long> list2 = new ArrayList<Long>();
398         Subscription s2 = interval.subscribe(new Action1<Long>() {
399 
400             @Override
401             public void call(Long t1) {
402                 list2.add(t1);
403             }
404 
405         });
406         s.advanceTimeBy(300, TimeUnit.MILLISECONDS);
407 
408         // list 1 should have 5 items
409         assertEquals(5, list1.size());
410         assertEquals(2L, list1.get(2).longValue());
411         assertEquals(3L, list1.get(3).longValue());
412         assertEquals(4L, list1.get(4).longValue());
413 
414         // list 2 should only have 3 items
415         assertEquals(3, list2.size());
416         assertEquals(2L, list2.get(0).longValue());
417         assertEquals(3L, list2.get(1).longValue());
418         assertEquals(4L, list2.get(2).longValue());
419 
420         // unsubscribe list1
421         s1.unsubscribe();
422 
423         // advance further
424         s.advanceTimeBy(300, TimeUnit.MILLISECONDS);
425 
426         // list 1 should still have 5 items
427         assertEquals(5, list1.size());
428 
429         // list 2 should have 6 items
430         assertEquals(6, list2.size());
431         assertEquals(5L, list2.get(3).longValue());
432         assertEquals(6L, list2.get(4).longValue());
433         assertEquals(7L, list2.get(5).longValue());
434 
435         // unsubscribe list2
436         s2.unsubscribe();
437 
438         // advance further
439         s.advanceTimeBy(1000, TimeUnit.MILLISECONDS);
440 
441         // subscribing a new one should start over because the source should have been unsubscribed
442         // subscribe list3
443         final List<Long> list3 = new ArrayList<Long>();
444         interval.subscribe(new Action1<Long>() {
445 
446             @Override
447             public void call(Long t1) {
448                 list3.add(t1);
449             }
450 
451         });
452         s.advanceTimeBy(200, TimeUnit.MILLISECONDS);
453 
454         assertEquals(2, list3.size());
455         assertEquals(0L, list3.get(0).longValue());
456         assertEquals(1L, list3.get(1).longValue());
457 
458     }
