36  /**
37   * LIMIT clause handler compatible with SQL Server 2005 and later.
38   *
39   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
40   */
41  public class SQLServer2005LimitHandler extends AbstractLimitHandler {
42  	private static final String SELECT = "select";
43  	private static final String SELECT_WITH_SPACE = SELECT + ' ';
44  	private static final String FROM = "from";
45  	private static final String DISTINCT = "distinct";
46  	private static final String ORDER_BY = "order by";
47  
48  	private static final Pattern ALIAS_PATTERN = Pattern.compile( "(?i)\\sas\\s(.)+$" );
49  
50  	// Flag indicating whether TOP(?) expression has been added to the original query.
51  	private boolean topAdded;
52  	// True if offset greater than 0.
53  	private boolean hasOffset = true;
54  
55  	/**
56  	 * Constructs a SQLServer2005LimitHandler
57  	 *
58  	 * @param sql The SQL
59  	 * @param selection The row selection options
60  	 */
61  	public SQLServer2005LimitHandler(String sql, RowSelection selection) {
62  		super( sql, selection );

85  	@Override
86  	public int convertToFirstRowValue(int zeroBasedFirstResult) {
87  		// Our dialect paginated results aren't zero based. The first row should get the number 1 and so on
88  		return zeroBasedFirstResult + 1;

91  	/**
92  	 * Add a LIMIT clause to the given SQL SELECT (HHH-2655: ROW_NUMBER for Paging)
93  	 *
94  	 * The LIMIT SQL will look like:
95  	 *
96  	 * <pre>
97  	 * WITH query AS (
98  	 *   SELECT inner_query.*
99  	 *        , ROW_NUMBER() OVER (ORDER BY CURRENT_TIMESTAMP) as __hibernate_row_nr__
100 	 *     FROM ( original_query_with_top_if_order_by_present_and_all_aliased_columns ) inner_query
101 	 * )
102 	 * SELECT alias_list FROM query WHERE __hibernate_row_nr__ >= offset AND __hibernate_row_nr__ < offset + last
103 	 * </pre>
104 	 *
105 	 * When offset equals {@literal 0}, only <code>TOP(?)</code> expression is added to the original query.
106 	 *
107 	 * @return A new SQL statement with the LIMIT clause applied.
108 	 */
109 	@Override
110 	public String getProcessedSql() {
111 		final StringBuilder sb = new StringBuilder( sql );
112 		if ( sb.charAt( sb.length() - 1 ) == ';' ) {
113 			sb.setLength( sb.length() - 1 );
114 		}
115 
116 		if ( LimitHelper.hasFirstRow( selection ) ) {
117 			final String selectClause = fillAliasInSelectClause( sb );
118 
119 			final int orderByIndex = shallowIndexOfWord( sb, ORDER_BY, 0 );
120 			if ( orderByIndex > 0 ) {
121 				// ORDER BY requires using TOP.
122 				addTopExpression( sb );
123 			}
124 
125 			encloseWithOuterQuery( sb );
126 
127 			// Wrap the query within a with statement:
128 			sb.insert( 0, "WITH query AS (" ).append( ") SELECT " ).append( selectClause ).append( " FROM query " );
129 			sb.append( "WHERE __hibernate_row_nr__ >= ? AND __hibernate_row_nr__ < ?" );
130 		}
131 		else {
132 			hasOffset = false;
133 			addTopExpression( sb );
134 		}
135 
136 		return sb.toString();
137 	}

139 	@Override
140 	public int bindLimitParametersAtStartOfQuery(PreparedStatement statement, int index) throws SQLException {
141 		if ( topAdded ) {
142 			// Binding TOP(?)
143 			statement.setInt( index, getMaxOrLimit() - 1 );
144 			return 1;
145 		}
146 		return 0;
147 	}
148 
149 	@Override
150 	public int bindLimitParametersAtEndOfQuery(PreparedStatement statement, int index) throws SQLException {
151 		return hasOffset ? super.bindLimitParametersAtEndOfQuery( statement, index ) : 0;
152 	}

154 	/**
155 	 * Adds missing aliases in provided SELECT clause and returns coma-separated list of them.
156 	 * If query takes advantage of expressions like {@literal *} or {@literal {table}.*} inside SELECT clause,
157 	 * method returns {@literal *}.
158 	 *
159 	 * @param sb SQL query.
160 	 *
161 	 * @return List of aliases separated with comas or {@literal *}.
162 	 */
163 	protected String fillAliasInSelectClause(StringBuilder sb) {
164 		final List<String> aliases = new LinkedList<String>();
165 		final int startPos = shallowIndexOf( sb, SELECT_WITH_SPACE, 0 );
166 		int endPos = shallowIndexOfWord( sb, FROM, startPos );
167 		int nextComa = startPos;
168 		int prevComa = startPos;
169 		int unique = 0;
170 		boolean selectsMultipleColumns = false;
171 
172 		while ( nextComa != -1 ) {
173 			prevComa = nextComa;
174 			nextComa = shallowIndexOf( sb, ",", nextComa );
175 			if ( nextComa > endPos ) {
176 				break;
177 			}
178 			if ( nextComa != -1 ) {
179 				final String expression = sb.substring( prevComa, nextComa );
180 				if ( selectsMultipleColumns( expression ) ) {
181 					selectsMultipleColumns = true;
182 				}
183 				else {
184 					String alias = getAlias( expression );
185 					if ( alias == null ) {
186 						// Inserting alias. It is unlikely that we would have to add alias, but just in case.
187 						alias = StringHelper.generateAlias( "page", unique );
188 						sb.insert( nextComa, " as " + alias );
189 						int aliasExprLength = ( " as " + alias ).length();
190 						++unique;
191 						nextComa += aliasExprLength;
192 						endPos += aliasExprLength;
193 					}
194 					aliases.add( alias );
195 				}
196 				++nextComa;
197 			}
198 		}
199 		// Processing last column.
200 		// Refreshing end position, because we might have inserted new alias.
201 		endPos = shallowIndexOfWord( sb, FROM, startPos );
202 		final String expression = sb.substring( prevComa, endPos );
203 		if ( selectsMultipleColumns( expression ) ) {
204 			selectsMultipleColumns = true;
205 		}
206 		else {
207 			String alias = getAlias( expression );
208 			if ( alias == null ) {
209 				// Inserting alias. It is unlikely that we would have to add alias, but just in case.
210 				alias = StringHelper.generateAlias( "page", unique );
211 				sb.insert( endPos - 1, " as " + alias );
212 			}
213 			aliases.add( alias );
214 		}
215 
216 		// In case of '*' or '{table}.*' expressions adding an alias breaks SQL syntax, returning '*'.
217 		return selectsMultipleColumns ? "*" : StringHelper.join( ", ", aliases.iterator() );
218 	}
219 
220 	/**
221 	 * @param expression Select expression.
222 	 *
223 	 * @return {@code true} when expression selects multiple columns, {@code false} otherwise.
224 	 */
225 	private boolean selectsMultipleColumns(String expression) {
226 		final String lastExpr = expression.trim().replaceFirst( "(?i)(.)*\\s", "" );
227 		return "*".equals( lastExpr ) || lastExpr.endsWith( ".*" );
228 	}
229 
230 	/**
231 	 * Returns alias of provided single column selection or {@code null} if not found.
232 	 * Alias should be preceded with {@code AS} keyword.
233 	 *
234 	 * @param expression Single column select expression.
235 	 *
236 	 * @return Column alias.
237 	 */
238 	private String getAlias(String expression) {
239 		final Matcher matcher = ALIAS_PATTERN.matcher( expression );
240 		if ( matcher.find() ) {
241 			// Taking advantage of Java regular expressions greedy behavior while extracting the last AS keyword.
242 			// Note that AS keyword can appear in CAST operator, e.g. 'cast(tab1.col1 as varchar(255)) as col1'.
243 			return matcher.group( 0 ).replaceFirst( "(?i)(.)*\\sas\\s", "" ).trim();
244 		}
245 		return null;
246 	}

248 	/**
249 	 * Encloses original SQL statement with outer query that provides {@literal __hibernate_row_nr__} column.
250 	 *
251 	 * @param sql SQL query.
252 	 */
253 	protected void encloseWithOuterQuery(StringBuilder sql) {
254 		sql.insert( 0, "SELECT inner_query.*, ROW_NUMBER() OVER (ORDER BY CURRENT_TIMESTAMP) as __hibernate_row_nr__ FROM ( " );
255 		sql.append( " ) inner_query " );
256 	}

258 	/**
259 	 * Adds {@code TOP} expression. Parameter value is bind in
260 	 * {@link #bindLimitParametersAtStartOfQuery(PreparedStatement, int)} method.
261 	 *
262 	 * @param sql SQL query.
263 	 */
264 	protected void addTopExpression(StringBuilder sql) {
265 		final int distinctStartPos = shallowIndexOfWord( sql, DISTINCT, 0 );
266 		if ( distinctStartPos > 0 ) {
267 			// Place TOP after DISTINCT.
268 			sql.insert( distinctStartPos + DISTINCT.length(), " TOP(?)" );
269 		}
270 		else {
271 			final int selectStartPos = shallowIndexOf( sql, SELECT_WITH_SPACE, 0 );
272 			// Place TOP after SELECT.
273 			sql.insert( selectStartPos + SELECT.length(), " TOP(?)" );
274 		}
275 		topAdded = true;
276 	}

278 	/**
279 	 * Returns index of the first case-insensitive match of search term surrounded by spaces
280 	 * that is not enclosed in parentheses.
281 	 *
282 	 * @param sb String to search.
283 	 * @param search Search term.
284 	 * @param fromIndex The index from which to start the search.
285 	 *
286 	 * @return Position of the first match, or {@literal -1} if not found.
287 	 */
288 	private static int shallowIndexOfWord(final StringBuilder sb, final String search, int fromIndex) {
289 		final int index = shallowIndexOf( sb, ' ' + search + ' ', fromIndex );
290 		// In case of match adding one because of space placed in front of search term.
291 		return index != -1 ? ( index + 1 ) : -1;
292 	}
293 
294 	/**
295 	 * Returns index of the first case-insensitive match of search term that is not enclosed in parentheses.
296 	 *
297 	 * @param sb String to search.
298 	 * @param search Search term.
299 	 * @param fromIndex The index from which to start the search.
300 	 *
301 	 * @return Position of the first match, or {@literal -1} if not found.
302 	 */
303 	private static int shallowIndexOf(StringBuilder sb, String search, int fromIndex) {
304 		// case-insensitive match
305 		final String lowercase = sb.toString().toLowerCase();
306 		final int len = lowercase.length();
307 		final int searchlen = search.length();
308 		int pos = -1;
309 		int depth = 0;
310 		int cur = fromIndex;
311 		do {
312 			pos = lowercase.indexOf( search, cur );
313 			if ( pos != -1 ) {
314 				for ( int iter = cur; iter < pos; iter++ ) {
315 					final char c = sb.charAt( iter );
316 					if ( c == '(' ) {
317 						depth = depth + 1;
318 					}
319 					else if ( c == ')' ) {
320 						depth = depth - 1;
321 					}
322 				}
323 				cur = pos + searchlen;
324 			}
325 		} while ( cur < len && depth != 0 && pos != -1 );
326 		return depth == 0 ? pos : -1;
327 	}
