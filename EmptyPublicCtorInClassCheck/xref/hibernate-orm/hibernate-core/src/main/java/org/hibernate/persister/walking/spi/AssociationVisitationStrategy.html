24  package org.hibernate.persister.walking.spi;
25  
26  import org.hibernate.loader.plan.spi.FetchSource;
27  
28  /**
29   * Strategy for walking associations as defined by the Hibernate metamodel.  Is essentially a callback listener for
30   * interesting events while walking a metamodel graph
31   * <p/>
32   * {@link #start()} and {@link #finish()} are called at the start and at the finish of the process.
33   * <p/>
34   * Walking might start with an entity or a collection depending on where the walker is asked to start.  When starting
35   * with an entity, {@link #startingEntity}/{@link #finishingEntity} ()} will be the outer set of calls.  When starting
36   * with a collection, {@link #startingCollection}/{@link #finishingCollection} will be the outer set of calls.
37   *
38   * @author Steve Ebersole
39   */
40  public interface AssociationVisitationStrategy {
41  	/**
42  	 * Notification we are preparing to start visitation.
43  	 */
44  	public void start();
45  
46  	/**
47  	 * Notification we are finished visitation.
48  	 */
49  	public void finish();
50  
51  	/**
52  	 * Notification we are starting to walk an entity.
53  	 *
54  	 * @param entityDefinition The entity we are preparing to walk
55  	 */
56  	public void startingEntity(EntityDefinition entityDefinition);
57  
58  	/**
59  	 * Notification we are finishing walking an entity.
60  	 *
61  	 * @param entityDefinition The entity we are finishing walking.
62  	 */
63  	public void finishingEntity(EntityDefinition entityDefinition);
64  
65  	/**
66  	 * Notification we are starting to walk the identifier of an entity.
67  	 *
68  	 * @param entityIdentifierDefinition The identifier we are preparing to walk
69  	 */
70  	public void startingEntityIdentifier(EntityIdentifierDefinition entityIdentifierDefinition);
71  
72  	/**
73  	 * Notification we are finishing walking an entity.
74  	 *
75  	 * @param entityIdentifierDefinition The identifier we are finishing walking.
76  	 */
77  	public void finishingEntityIdentifier(EntityIdentifierDefinition entityIdentifierDefinition);
78  
79  	/**
80  	 * Notification that we are starting to walk a collection
81  	 *
82  	 * @param collectionDefinition The collection we are preparing to walk
83  	 */
84  	public void startingCollection(CollectionDefinition collectionDefinition);
85  
86  	/**
87  	 * Notification that we are finishing walking a collection
88  	 *
89  	 * @param collectionDefinition The collection we are finishing
90  	 */
91  	public void finishingCollection(CollectionDefinition collectionDefinition);
92  
93  	/**
94  	 * Notification that we are starting to walk the index of a collection (List/Map).  In the case of a Map,
95  	 * if the indices (the keys) are entities this will be followed up by a call to {@link #startingEntity}
96  	 *
97  	 * @param collectionIndexDefinition The collection index we are preparing to walk.
98  	 */
99  	public void startingCollectionIndex(CollectionIndexDefinition collectionIndexDefinition);
100 
101 	/**
102 	 * Notification that we are finishing walking the index of a collection (List/Map).
103 	 *
104 	 * @param collectionIndexDefinition The collection index we are finishing
105 	 */
106 	public void finishingCollectionIndex(CollectionIndexDefinition collectionIndexDefinition);
107 
108 	/**
109 	 * Notification that we are starting to look at the element definition for the collection.  If the collection
110 	 * elements are entities this will be followed up by a call to {@link #startingEntity}
111 	 *
112 	 * @param elementDefinition The collection element we are preparing to walk..
113 	 */
114 	public void startingCollectionElements(CollectionElementDefinition elementDefinition);
115 
116 	/**
117 	 * Notification that we are finishing walking the elements of a collection (List/Map).
118 	 *
119 	 * @param elementDefinition The collection element we are finishing
120 	 */
121 	public void finishingCollectionElements(CollectionElementDefinition elementDefinition);
122 
123 	/**
124 	 * Notification that we are preparing to walk a composite.  This is called only for:<ul>
125 	 *     <li>
126 	 *         top-level composites for entity attributes. composite entity identifiers do not route through here, see
127 	 *         {@link #startingEntityIdentifier} if you need to hook into walking the top-level cid composite.
128 	 *     </li>
129 	 *     <li>
130 	 *         All forms of nested composite paths
131 	 *     </li>
132 	 * </ul>
133 	 *
134 	 * @param compositionDefinition The composite we are preparing to walk.
135 	 */
136 	public void startingComposite(CompositionDefinition compositionDefinition);
137 
138 	/**
139 	 * Notification that we are done walking a composite.  Called on the back-end of the situations listed
140 	 * on {@link #startingComposite}
141 	 *
142 	 * @param compositionDefinition The composite we are finishing
143 	 */
144 	public void finishingComposite(CompositionDefinition compositionDefinition);
145 
146 	// get rid of these ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
147 //	public void startingCompositeCollectionElement(CompositeCollectionElementDefinition compositionElementDefinition);
148 //	public void finishingCompositeCollectionElement(CompositeCollectionElementDefinition compositionElementDefinition);
149 	// ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
150 
151 	/**
152 	 * Notification that we are preparing to walk an attribute.  May be followed by calls to {@link #startingEntity}
153 	 * (one-to-one, many-to-one), {@link #startingComposite}, or {@link #startingCollection}.
154 	 *
155 	 * @param attributeDefinition The attribute we are preparing to walk.
156 	 *
157 	 * @return {@code true} if the walking should continue; {@code false} if walking should stop.
158 	 */
159 	public boolean startingAttribute(AttributeDefinition attributeDefinition);
160 
161 	/**
162 	 * Notification that we are finishing walking an attribute.
163 	 *
164 	 * @param attributeDefinition The attribute we are done walking
165 	 */
166 	public void finishingAttribute(AttributeDefinition attributeDefinition);
167 
168 	public void foundAny(AnyMappingDefinition anyDefinition);
169 
170 	public void associationKeyRegistered(AssociationKey associationKey);
171 	public FetchSource registeredFetchSource(AssociationKey associationKey);
172 	public void foundCircularAssociation(AssociationAttributeDefinition attributeDefinition);
173 	public boolean isDuplicateAssociationKey(AssociationKey associationKey);
174 
175 }
