24  package org.hibernate.hql.internal.ast.tree;
25  
26  import java.sql.Types;
27  import javax.persistence.AttributeConverter;
28  
29  import org.hibernate.hql.internal.antlr.HqlSqlTokenTypes;
30  import org.hibernate.hql.internal.ast.util.ColumnHelper;
31  import org.hibernate.type.SingleColumnType;
32  import org.hibernate.type.StandardBasicTypes;
33  import org.hibernate.type.Type;
34  import org.hibernate.type.descriptor.converter.AttributeConverterTypeAdapter;
35  
36  import antlr.SemanticException;
37  
38  /**
39   * Represents a literal.
40   *
41   * @author josh
42   * @author Steve Ebersole
43   */
44  public class LiteralNode extends AbstractSelectExpression implements HqlSqlTokenTypes, ExpectedTypeAwareNode {
45  	private Type expectedType;
46  
47  	public void setScalarColumnText(int i) throws SemanticException {
48  		ColumnHelper.generateSingleScalarColumn( this, i );
49  	}
50  
51  	public Type getDataType() {
52  		if ( expectedType != null ) {
53  			return expectedType;
54  		}
55  
56  		switch ( getType() ) {
57  			case NUM_INT:
58  				return StandardBasicTypes.INTEGER;
59  			case NUM_FLOAT:
60  				return StandardBasicTypes.FLOAT;
61  			case NUM_LONG:
62  				return StandardBasicTypes.LONG;
63  			case NUM_DOUBLE:
64  				return StandardBasicTypes.DOUBLE;
65  			case NUM_BIG_INTEGER:
66  				return StandardBasicTypes.BIG_INTEGER;
67  			case NUM_BIG_DECIMAL:
68  				return StandardBasicTypes.BIG_DECIMAL;
69  			case QUOTED_STRING:
70  				return StandardBasicTypes.STRING;
71  			case TRUE:
72  			case FALSE:
73  				return StandardBasicTypes.BOOLEAN;
74  			default:
75  				return null;
76  		}
77  	}
78  
79  	public Object getLiteralValue() {
80  		String text = getText();
81  		if ( getType() == QUOTED_STRING ) {
82  			text = text.substring( 1, text.length() -1 );
83  		}
84  
85  		final Type inherentType = getDataType();
86  		if ( inherentType == null ) {
87  			return text;
88  		}
89  
90  		return ( (SingleColumnType) inherentType ).fromStringValue( text );
91  	}
92  
93  	@Override
94  	public void setExpectedType(Type expectedType) {
95  		if ( this.expectedType != null ) {
96  			return;
97  		}
98  
99  		if ( AttributeConverterTypeAdapter.class.isInstance( expectedType ) ) {
100 			final AttributeConverterTypeAdapter adapterType = (AttributeConverterTypeAdapter) expectedType;
101 			if ( getDataType().getReturnedClass().equals( adapterType.getModelType() ) ) {
102 				// apply the converter
103 				final AttributeConverter converter = ( (AttributeConverterTypeAdapter) expectedType ).getAttributeConverter();
104 				final Object converted = converter.convertToDatabaseColumn( getLiteralValue() );
105 				if ( isCharacterData( adapterType.sqlType() ) ) {
106 					setText( "'" + converted.toString() + "'" );
107 				}
108 				else {
109 					setText( converted.toString() );
110 				}
111 			}
112 			this.expectedType = expectedType;
113 		}
114 	}
115 
116 	private boolean isCharacterData(int typeCode) {
117 		return Types.VARCHAR == typeCode
118 				|| Types.CHAR == typeCode
119 				|| Types.NVARCHAR == typeCode
120 				|| Types.NCHAR == typeCode;
121 	}
122 
123 	@Override
124 	public Type getExpectedType() {
125 		return expectedType;
126 	}
127 }
