52  /**
53   * A {@link org.hibernate.context.spi.CurrentSessionContext} impl which scopes the notion of current
54   * session by the current thread of execution.  Unlike the JTA counterpart, threads do not give us a nice
55   * hook to perform any type of cleanup making it questionable for this impl to actually generate Session
56   * instances.  In the interest of usability, it was decided to have this default impl actually generate
57   * a session upon first request and then clean it up after the {@link org.hibernate.Transaction}
58   * associated with that session is committed/rolled-back.  In order for ensuring that happens, the
59   * sessions generated here are unusable until after {@link Session#beginTransaction()} has been
60   * called. If <tt>close()</tt> is called on a session managed by this class, it will be automatically
61   * unbound.
62   *
63   * Additionally, the static {@link #bind} and {@link #unbind} methods are provided to allow application
64   * code to explicitly control opening and closing of these sessions.  This, with some from of interception,
65   * is the preferred approach.  It also allows easy framework integration and one possible approach for
66   * implementing long-sessions.
67   *
68   * The {@link #buildOrObtainSession}, {@link #isAutoCloseEnabled}, {@link #isAutoFlushEnabled},
69   * {@link #getConnectionReleaseMode}, and {@link #buildCleanupSynch} methods are all provided to allow easy
70   * subclassing (for long-running session scenarios, for example).
71   *
72   * @author Steve Ebersole
73   */
74  public class ThreadLocalSessionContext extends AbstractCurrentSessionContext {
75  	private static final CoreMessageLogger LOG = Logger.getMessageLogger(
76  			CoreMessageLogger.class,
77  			ThreadLocalSessionContext.class.getName()
78  	);
79  
80  	private static final Class[] SESSION_PROXY_INTERFACES = new Class[] {
81  			Session.class,
82  			SessionImplementor.class,
83  			EventSource.class,
84  			TransactionContext.class,
85  			LobCreationContext.class
86  	};
87  
88  	/**
89  	 * A ThreadLocal maintaining current sessions for the given execution thread.
90  	 * The actual ThreadLocal variable is a java.util.Map to account for
91  	 * the possibility for multiple SessionFactory instances being used during execution
92  	 * of the given thread.
93  	 */
94  	private static final ThreadLocal<Map> CONTEXT_TL = new ThreadLocal<Map>();
95  
96  	/**
97  	 * Constructs a ThreadLocal
98  	 *
99  	 * @param factory The factory this context will service
100 	 */
101 	public ThreadLocalSessionContext(SessionFactoryImplementor factory) {
102 		super( factory );
103 	}
104 
105 	@Override
106 	public final Session currentSession() throws HibernateException {
107 		Session current = existingSession( factory() );
108 		if ( current == null ) {
109 			current = buildOrObtainSession();
110 			// register a cleanup sync
111 			current.getTransaction().registerSynchronization( buildCleanupSynch() );
112 			// wrap the session in the transaction-protection proxy
113 			if ( needsWrapping( current ) ) {
114 				current = wrap( current );
115 			}
116 			// then bind it
117 			doBind( current, factory() );
118 		}
119 		else {
120 			validateExistingSession( current );
121 		}
122 		return current;
123 	}
124 
125 	private boolean needsWrapping(Session session) {
126 		// try to make sure we don't wrap and already wrapped session
127 		if ( session != null ) {
128 			if ( Proxy.isProxyClass( session.getClass() ) ) {
129 				final InvocationHandler invocationHandler = Proxy.getInvocationHandler( session );
130 				if ( invocationHandler != null && TransactionProtectionWrapper.class.isInstance( invocationHandler ) ) {
131 					return false;
132 				}
133 			}
134 		}
135 		return true;
136 	}

138 	/**
139 	 * Getter for property 'factory'.
140 	 *
141 	 * @return Value for property 'factory'.
142 	 */
143 	protected SessionFactoryImplementor getFactory() {
144 		return factory();
145 	}

147 	/**
148 	 * Strictly provided for sub-classing purposes; specifically to allow long-session
149 	 * support.
150 	 * <p/>
151 	 * This implementation always just opens a new session.
152 	 *
153 	 * @return the built or (re)obtained session.
154 	 */
155 	@SuppressWarnings("deprecation")
156 	protected Session buildOrObtainSession() {
157 		return baseSessionBuilder()
158 				.autoClose( isAutoCloseEnabled() )
159 				.connectionReleaseMode( getConnectionReleaseMode() )
160 				.flushBeforeCompletion( isAutoFlushEnabled() )
161 				.openSession();
162 	}
163 
164 	protected CleanupSync buildCleanupSynch() {
165 		return new CleanupSync( factory() );
166 	}
167 
168 	/**
169 	 * Mainly for subclass usage.  This impl always returns true.
170 	 *
171 	 * @return Whether or not the the session should be closed by transaction completion.
172 	 */
173 	protected boolean isAutoCloseEnabled() {
174 		return true;
175 	}
176 
177 	/**
178 	 * Mainly for subclass usage.  This impl always returns true.
179 	 *
180 	 * @return Whether or not the the session should be flushed prior transaction completion.
181 	 */
182 	protected boolean isAutoFlushEnabled() {
183 		return true;
184 	}
185 
186 	/**
187 	 * Mainly for subclass usage.  This impl always returns after_transaction.
188 	 *
189 	 * @return The connection release mode for any built sessions.
190 	 */
191 	protected ConnectionReleaseMode getConnectionReleaseMode() {
192 		return factory().getSettings().getConnectionReleaseMode();
193 	}
194 
195 	protected Session wrap(Session session) {
196 		final TransactionProtectionWrapper wrapper = new TransactionProtectionWrapper( session );
197 		final Session wrapped = (Session) Proxy.newProxyInstance(
198 				Session.class.getClassLoader(),
199 				SESSION_PROXY_INTERFACES,
200 				wrapper
201 		);
202 		// yick!  need this for proper serialization/deserialization handling...
203 		wrapper.setWrapped( wrapped );
204 		return wrapped;
205 	}
206 
207 	/**
208 	 * Associates the given session with the current thread of execution.
209 	 *
210 	 * @param session The session to bind.
211 	 */
212 	public static void bind(org.hibernate.Session session) {
213 		final SessionFactory factory = session.getSessionFactory();
214 		cleanupAnyOrphanedSession( factory );
215 		doBind( session, factory );
216 	}
217 
218 	private static void cleanupAnyOrphanedSession(SessionFactory factory) {
219 		final Session orphan = doUnbind( factory, false );
220 		if ( orphan != null ) {
221 			LOG.alreadySessionBound();
222 			try {
223 				if ( orphan.getTransaction() != null && orphan.getTransaction().isActive() ) {
224 					try {
225 						orphan.getTransaction().rollback();
226 					}
227 					catch( Throwable t ) {
228 						LOG.debug( "Unable to rollback transaction for orphaned session", t );
229 					}
230 				}
231 				orphan.close();
232 			}
233 			catch( Throwable t ) {
234 				LOG.debug( "Unable to close orphaned session", t );
235 			}
236 		}
237 	}
238 
239 	/**
240 	 * Disassociates a previously bound session from the current thread of execution.
241 	 *
242 	 * @param factory The factory for which the session should be unbound.
243 	 * @return The session which was unbound.
244 	 */
245 	public static Session unbind(SessionFactory factory) {
246 		return doUnbind( factory, true );
247 	}
248 
249 	private static Session existingSession(SessionFactory factory) {
250 		final Map sessionMap = sessionMap();
251 		if ( sessionMap == null ) {
252 			return null;
253 		}
254 		return (Session) sessionMap.get( factory );
255 	}
256 
257 	protected static Map sessionMap() {
258 		return CONTEXT_TL.get();
259 	}
260 
261 	@SuppressWarnings({"unchecked"})
262 	private static void doBind(org.hibernate.Session session, SessionFactory factory) {
263 		Map sessionMap = sessionMap();
264 		if ( sessionMap == null ) {
265 			sessionMap = new HashMap();
266 			CONTEXT_TL.set( sessionMap );
267 		}
268 		sessionMap.put( factory, session );
269 	}
270 
271 	private static Session doUnbind(SessionFactory factory, boolean releaseMapIfEmpty) {
272 		Session session = null;
273 		final Map sessionMap = sessionMap();
274 		if ( sessionMap != null ) {
275 			session = (Session) sessionMap.remove( factory );
276 			if ( releaseMapIfEmpty && sessionMap.isEmpty() ) {
277 				CONTEXT_TL.set( null );
278 			}
279 		}
280 		return session;
281 	}

283 	/**
284 	 * Transaction sync used for cleanup of the internal session map.
285 	 */
286 	protected static class CleanupSync implements Synchronization, Serializable {
287 		protected final SessionFactory factory;
288 
289 		public CleanupSync(SessionFactory factory) {
290 			this.factory = factory;
291 		}
292 
293 		@Override
294 		public void beforeCompletion() {
295 		}
296 
297 		@Override
298 		public void afterCompletion(int i) {
299 			unbind( factory );
300 		}
301 	}
302 
303 	private class TransactionProtectionWrapper implements InvocationHandler, Serializable {
304 		private final Session realSession;
305 		private Session wrappedSession;
306 
307 		public TransactionProtectionWrapper(Session realSession) {
308 			this.realSession = realSession;
309 		}
310 
311 		@Override
312 		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
313 			final String methodName = method.getName(); 
314 			try {
315 				// If close() is called, guarantee unbind()
316 				if ( "close".equals( methodName ) ) {
317 					unbind( realSession.getSessionFactory() );
318 				}
319 				else if ( "toString".equals( methodName )
320 						|| "equals".equals( methodName )
321 						|| "hashCode".equals( methodName )
322 						|| "getStatistics".equals( methodName )
323 						|| "isOpen".equals( methodName )
324 						|| "getListeners".equals( methodName ) ) {
325 					// allow these to go through the the real session no matter what
326 					LOG.tracef( "Allowing invocation [%s] to proceed to real session", methodName );
327 				}
328 				else if ( !realSession.isOpen() ) {
329 					// essentially, if the real session is closed allow any
330 					// method call to pass through since the real session
331 					// will complain by throwing an appropriate exception;
332 					// NOTE that allowing close() above has the same basic effect,
333 					//   but we capture that there simply to doAfterTransactionCompletion the unbind...
334 					LOG.tracef( "Allowing invocation [%s] to proceed to real (closed) session", methodName );
335 				}
336 				else if ( !realSession.getTransaction().isActive() ) {
337 					// limit the methods available if no transaction is active
338 					if ( "beginTransaction".equals( methodName )
339 							|| "getTransaction".equals( methodName )
340 							|| "isTransactionInProgress".equals( methodName )
341 							|| "setFlushMode".equals( methodName )
342 							|| "getFactory".equals( methodName )
343 							|| "getSessionFactory".equals( methodName )
344 							|| "getTenantIdentifier".equals( methodName ) ) {
345 						LOG.tracef( "Allowing invocation [%s] to proceed to real (non-transacted) session", methodName );
346 					}
347 					else if ( "reconnect".equals( methodName ) || "disconnect".equals( methodName ) ) {
348 						// allow these (deprecated) methods to pass through
349 						LOG.tracef( "Allowing invocation [%s] to proceed to real (non-transacted) session - deprecated methods", methodName );
350 					}
351 					else {
352 						throw new HibernateException( methodName + " is not valid without active transaction" );
353 					}
354 				}
355 				LOG.tracef( "Allowing proxy invocation [%s] to proceed to real session", methodName );
356 				return method.invoke( realSession, args );
357 			}
358 			catch ( InvocationTargetException e ) {
359 				if (e.getTargetException() instanceof RuntimeException) {
360 					throw (RuntimeException)e.getTargetException();
361 				}
362 				throw e;
363 			}
364 		}
365 
366 		/**
367 		 * Setter for property 'wrapped'.
368 		 *
369 		 * @param wrapped Value to set for property 'wrapped'.
370 		 */
371 		public void setWrapped(Session wrapped) {
372 			this.wrappedSession = wrapped;
373 		}

377 
378 		private void writeObject(ObjectOutputStream oos) throws IOException {
379 			// if a ThreadLocalSessionContext-bound session happens to get
380 			// serialized, to be completely correct, we need to make sure
381 			// that unbinding of that session occurs.
382 			oos.defaultWriteObject();
383 			if ( existingSession( factory() ) == wrappedSession ) {
384 				unbind( factory() );
385 			}
386 		}
387 
388 		private void readObject(ObjectInputStream ois) throws IOException, ClassNotFoundException {
389 			// on the inverse, it makes sense that if a ThreadLocalSessionContext-
390 			// bound session then gets deserialized to go ahead and re-bind it to
391 			// the ThreadLocalSessionContext session map.
392 			ois.defaultReadObject();
393 			realSession.getTransaction().registerSynchronization( buildCleanupSynch() );
394 			doBind( wrappedSession, factory() );
395 		}
396 	}
