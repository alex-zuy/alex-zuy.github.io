47  @RequiresDialect( Oracle8iDialect.class )
48  public class QueryHintTest extends BaseCoreFunctionalTestCase {
49  	
50  	@Override
51  	protected Class<?>[] getAnnotatedClasses() {
52  		return new Class<?>[] { Employee.class, Department.class };
53  	}
54  	
55  	@Override
56  	protected void configure(Configuration configuration) {
57  		configuration.setProperty( AvailableSettings.DIALECT, QueryHintTestDialect.class.getName() );
58  		configuration.setProperty( AvailableSettings.USE_SQL_COMMENTS, "true" );
59  	}
60  	
61  	@Test
62  	public void testQueryHint() {
63  		Department department = new Department();
64  		department.name = "Sales";
65  		Employee employee1 = new Employee();
66  		employee1.department = department;
67  		Employee employee2 = new Employee();
68  		employee2.department = department;
69  		
70  		Session s = openSession();
71  		s.getTransaction().begin();
72  		s.persist( department );		s.persist( employee1 );
73  		s.persist( employee2 );
74  		s.getTransaction().commit();
75  		s.clear();
76  
77  		// test Query w/ a simple Oracle optimizer hint
78  		s.getTransaction().begin();
79  		Query query = s.createQuery( "FROM QueryHintTest$Employee e WHERE e.department.name = :departmentName" )
80  				.addQueryHint( "ALL_ROWS" )
81  				.setParameter( "departmentName", "Sales" );
82  		List results = query.list();
83  		s.getTransaction().commit();
84  		s.clear();
85  		
86  		assertEquals(results.size(), 2);
87  		assertTrue(QueryHintTestDialect.getProcessedSql().contains( "select /*+ ALL_ROWS */"));
88  		
89  		QueryHintTestDialect.resetProcessedSql();
90  		
91  		// test multiple hints
92  		s.getTransaction().begin();
93  		query = s.createQuery( "FROM QueryHintTest$Employee e WHERE e.department.name = :departmentName" )
94  				.addQueryHint( "ALL_ROWS" )
95  				.addQueryHint( "USE_CONCAT" )
96  				.setParameter( "departmentName", "Sales" );
97  		results = query.list();
98  		s.getTransaction().commit();
99  		s.clear();
100 		
101 		assertEquals(results.size(), 2);
102 		assertTrue(QueryHintTestDialect.getProcessedSql().contains( "select /*+ ALL_ROWS, USE_CONCAT */"));
103 		
104 		QueryHintTestDialect.resetProcessedSql();
105 		
106 		// ensure the insertion logic can handle a comment appended to the front
107 		s.getTransaction().begin();
108 		query = s.createQuery( "FROM QueryHintTest$Employee e WHERE e.department.name = :departmentName" )
109 				.setComment( "this is a test" )
110 				.addQueryHint( "ALL_ROWS" )
111 				.setParameter( "departmentName", "Sales" );
112 		results = query.list();
113 		s.getTransaction().commit();
114 		s.clear();
115 		
116 		assertEquals(results.size(), 2);
117 		assertTrue(QueryHintTestDialect.getProcessedSql().contains( "select /*+ ALL_ROWS */"));
118 		
119 		QueryHintTestDialect.resetProcessedSql();
120 		
121 		// test Criteria
122 		s.getTransaction().begin();
123 		Criteria criteria = s.createCriteria( Employee.class )
124 				.addQueryHint( "ALL_ROWS" )
125 				.createCriteria( "department" ).add( Restrictions.eq( "name", "Sales" ) );
126 		results = criteria.list();
127 		s.getTransaction().commit();
128 		s.close();
129 		
130 		assertEquals(results.size(), 2);
131 		assertTrue(QueryHintTestDialect.getProcessedSql().contains( "select /*+ ALL_ROWS */"));
132 	}

134 	/**
135 	 * Since the query hint is added to the SQL during Loader's executeQueryStatement -> preprocessSQL, rather than
136 	 * early on during the QueryTranslator or QueryLoader initialization, there's not an easy way to check the full SQL
137 	 * after completely processing it.  Instead, use this ridiculous hack to ensure Loader actually calls Dialect.
138 	 * 
139 	 * TODO: This is terrible.  Better ideas?
140 	 */
141 	public static class QueryHintTestDialect extends Oracle8iDialect {
142 		private static String processedSql;
143 		
144 		@Override
145 		public String getQueryHintString(String sql, List<String> hints) {
146 			processedSql = super.getQueryHintString( sql, hints );
147 			return processedSql;
148 		}
149 		
150 		public static String getProcessedSql() {
151 			return processedSql;
152 		}
153 		
154 		public static void resetProcessedSql() {
155 			processedSql = "";
156 		}
157 	}
