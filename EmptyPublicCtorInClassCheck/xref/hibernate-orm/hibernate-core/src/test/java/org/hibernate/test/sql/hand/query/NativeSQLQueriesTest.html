876 	private String buildLongString(int size, char baseChar) {
877 		StringBuilder buff = new StringBuilder();
878 		for( int i = 0; i < size; i++ ) {
879 			buff.append( baseChar );
880 		}
881 		return buff.toString();
882 	}

884 	private byte[] buildLongByteArray(int size, boolean on) {
885 		byte[] data = new byte[size];
886 		data[0] = mask( on );
887 		for ( int i = 0; i < size; i++ ) {
888 			data[i] = mask( on );
889 			on = !on;
890 		}
891 		return data;
892 	}
893 
894 	private byte mask(boolean on) {
895 		return on ? ( byte ) 1 : ( byte ) 0;
896 	}

899 	private static class UpperCasedAliasToEntityMapResultTransformer extends BasicTransformerAdapter implements Serializable {
900 		public Object transformTuple(Object[] tuple, String[] aliases) {
901 			Map result = new HashMap( tuple.length );
902 			for ( int i = 0; i < tuple.length; i++ ) {
903 				String alias = aliases[i];
904 				if ( alias != null ) {
905 					result.put( alias.toUpperCase(), tuple[i] );
906 				}
907 			}
908 			return result;
909 		}
