61  	private AdvancedEntity getAdvancedEntity(ManyToOneEntity manyToOne, OneToOneEntity oneToOne, ManyToManyEntity manyToManyEntity) {
62  		AdvancedEntity advancedEntity = new AdvancedEntity();
63  		advancedEntity.setId( 1L );
64  		advancedEntity.setNote( "Test note" );
65  		advancedEntity.getDynamicConfiguration().put( PROP_BOOLEAN, true );
66  		advancedEntity.getDynamicConfiguration().put( PROP_INT, 19 );
67  		advancedEntity.getDynamicConfiguration().put( PROP_FLOAT, 15.9f );
68  		advancedEntity.getDynamicConfiguration().put( PROP_MANY_TO_ONE, manyToOne );
69  		advancedEntity.getDynamicConfiguration().put( PROP_ONE_TO_ONE, oneToOne );
70  		advancedEntity.getDynamicConfiguration().put( INTERNAL_COMPONENT, new InternalComponent( "Internal value" ) );
71  		List<String> list = new ArrayList<String>();
72  		list.add( "One" );
73  		list.add( "Two" );
74  		list.add( "Three" );
75  		advancedEntity.getDynamicConfiguration().put( INTERNAL_LIST, list );
76  		Map<String, String> map = new HashMap<String, String>();
77  		map.put( "one", "1" );
78  		map.put( "two", "2" );
79  		advancedEntity.getDynamicConfiguration().put( INTERNAL_MAP, map );
80  		Map<String, ManyToManyEntity> mapWithManyToMany = new HashMap<String, ManyToManyEntity>();
81  		mapWithManyToMany.put( "entity1", manyToManyEntity );
82  		advancedEntity.getDynamicConfiguration().put( INTERNAL_MAP_WITH_MANY_TO_MANY, mapWithManyToMany );
83  		Set<String> set = new HashSet<String>();
84  		set.add( "Une" );
85  		set.add( "Due" );
86  		advancedEntity.getDynamicConfiguration().put( INTERNAL_SET, set );
87  		Set<InternalComponent> componentSet = new HashSet<InternalComponent>();
88  		componentSet.add( new InternalComponent( "Ein" ) );
89  		componentSet.add( new InternalComponent( "Zwei" ) );
90  		advancedEntity.getDynamicConfiguration().put( INTERNAL_SET_OF_COMPONENTS, componentSet );
91  		advancedEntity.getDynamicConfiguration().put( AGE_USER_TYPE, new Age( 18 ) );
92  		List<Age> ages = new ArrayList<Age>();
93  		ages.add( new Age( 1 ) );
94  		ages.add( new Age( 2 ) );
95  		ages.add( new Age( 3 ) );
96  
97  		advancedEntity.getDynamicConfiguration().put( INTERNAL_LIST_OF_USER_TYPES, ages );
98  		return advancedEntity;
99  	}

103 	//smoke test to make sure that hibernate & envers are working with the entity&mappings
104 	public void shouldInitData() {
105 		//given
106 		ManyToOneEntity manyToOne = getManyToOneEntity();
107 		OneToOneEntity oneToOne = getOneToOneEntity();
108 		ManyToManyEntity manyToManyEntity = getManyToManyEntity();
109 		AdvancedEntity advancedEntity = getAdvancedEntity( manyToOne, oneToOne, manyToManyEntity );
110 
111 		//rev 1
112 		Session session = openSession();
113 		session.getTransaction().begin();
114 		session.save( manyToOne );
115 		session.save( oneToOne );
116 		session.save( manyToManyEntity );
117 		session.save( advancedEntity );
118 		session.getTransaction().commit();
119 
120 		//rev 2
121 		session.getTransaction().begin();
122 		InternalComponent internalComponent = (InternalComponent) advancedEntity.getDynamicConfiguration()
123 				.get( INTERNAL_COMPONENT );
124 		internalComponent.setProperty( "new value" );
125 		session.save( advancedEntity );
126 		session.getTransaction().commit();
127 
128 		//rev 3
129 		session.getTransaction().begin();
130 		List<String> internalList = (List) advancedEntity.getDynamicConfiguration().get( INTERNAL_LIST );
131 		internalList.add( "four" );
132 		session.save( advancedEntity );
133 		session.getTransaction().commit();
134 
135 		//rev 4
136 		session.getTransaction().begin();
137 		Map<String, String> map = (Map) advancedEntity.getDynamicConfiguration().get( INTERNAL_MAP );
138 		map.put( "three", "3" );
139 		session.save( advancedEntity );
140 		session.getTransaction().commit();
141 
142 		//rev 5
143 		session.getTransaction().begin();
144 		Map<String, ManyToManyEntity> mapWithManyToMany = (Map) advancedEntity.getDynamicConfiguration()
145 				.get( INTERNAL_MAP_WITH_MANY_TO_MANY );
146 		ManyToManyEntity manyToManyEntity2 = new ManyToManyEntity( 2L, "new value" );
147 		mapWithManyToMany.put( "entity2", manyToManyEntity2 );
148 		session.save( manyToManyEntity2 );
149 		session.save( advancedEntity );
150 		session.getTransaction().commit();
151 
152 		//rev 6
153 		session.getTransaction().begin();
154 		mapWithManyToMany = (Map) advancedEntity.getDynamicConfiguration().get( INTERNAL_MAP_WITH_MANY_TO_MANY );
155 		mapWithManyToMany.clear();
156 		session.save( advancedEntity );
157 		session.getTransaction().commit();
158 
159 		//rev 7
160 		session.getTransaction().begin();
161 		Set<InternalComponent> internalComponentSet = (Set) advancedEntity.getDynamicConfiguration()
162 				.get( INTERNAL_SET_OF_COMPONENTS );
163 		internalComponentSet.add( new InternalComponent( "drei" ) );
164 		session.save( advancedEntity );
165 		session.getTransaction().commit();
166 
167 		//rev 8
168 		session.getTransaction().begin();
169 		advancedEntity.getDynamicConfiguration().put( AGE_USER_TYPE, new Age( 19 ) );
170 		session.save( advancedEntity );
171 		session.getTransaction().commit();
172 
173 		//rev 9
174 		session.getTransaction().begin();
175 		List<Age> ages = (List<Age>) advancedEntity.getDynamicConfiguration().get( INTERNAL_LIST_OF_USER_TYPES );
176 		ages.add( new Age( 4 ) );
177 		session.save( advancedEntity );
178 		session.getTransaction().commit();
179 
180 		//rev this, should not create revision
181 		session.getTransaction().begin();
182 		session.getTransaction().commit();
183 
184 		//sanity check. Loaded entity should be equal to one that we created.
185 		AdvancedEntity advancedEntityActual = (AdvancedEntity) session.load( AdvancedEntity.class, 1L );
186 
187 		Assert.assertEquals( advancedEntity, advancedEntityActual );
188 	}

315 						AuditEntity.relatedId( "dynamicConfiguration_" + PROP_MANY_TO_ONE )

326 				.forEntitiesAtRevision( AdvancedEntity.class, 1 )

328 						AuditEntity.property( "dynamicConfiguration_" + INTERNAL_COMPONENT+"_property")
329 								.eq( internalComponent.getProperty() )
