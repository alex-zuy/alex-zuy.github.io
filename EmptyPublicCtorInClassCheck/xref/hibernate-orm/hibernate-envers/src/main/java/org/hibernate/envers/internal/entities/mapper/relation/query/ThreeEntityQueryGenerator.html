42  /**
43   * Selects data from a relation middle-table and a two related versions entity.
44   *
45   * @author Adam Warski (adam at warski dot org)
46   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
47   */
48  public final class ThreeEntityQueryGenerator extends AbstractRelationQueryGenerator {
49  	private final String queryString;
50  	private final String queryRemovedString;
51  
52  	public ThreeEntityQueryGenerator(
53  			GlobalConfiguration globalCfg, AuditEntitiesConfiguration verEntCfg,
54  			AuditStrategy auditStrategy, String versionsMiddleEntityName,
55  			MiddleIdData referencingIdData, MiddleIdData referencedIdData,
56  			MiddleIdData indexIdData, boolean revisionTypeInId,
57  			MiddleComponentData... componentData) {
58  		super( verEntCfg, referencingIdData, revisionTypeInId );
59  
60  		/*
61  		 * The valid query that we need to create:
62  		 *   SELECT new list(ee, e, f) FROM versionsReferencedEntity e, versionsIndexEntity f, middleEntity ee
63  		 *   WHERE
64  		 * (entities referenced by the middle table; id_ref_ed = id of the referenced entity)
65  		 *     ee.id_ref_ed = e.id_ref_ed AND
66  		 * (entities referenced by the middle table; id_ref_ind = id of the index entity)
67  		 *     ee.id_ref_ind = f.id_ref_ind AND
68  		 * (only entities referenced by the association; id_ref_ing = id of the referencing entity)
69  		 *     ee.id_ref_ing = :id_ref_ing AND
70  		 * (selecting e entities at revision :revision)
71  		 *   --> for DefaultAuditStrategy:
72  		 *     e.revision = (SELECT max(e2.revision) FROM versionsReferencedEntity e2
73  		 *       WHERE e2.revision <= :revision AND e2.id = e.id)
74  		 *
75  		 *   --> for ValidityAuditStrategy:
76  		 *     e.revision <= :revision and (e.endRevision > :revision or e.endRevision is null)
77  		 *
78  		 *     AND
79  		 *
80  		 * (selecting f entities at revision :revision)
81  		 *   --> for DefaultAuditStrategy:
82  		 *     f.revision = (SELECT max(f2.revision) FROM versionsIndexEntity f2
83  		 *       WHERE f2.revision <= :revision AND f2.id_ref_ed = f.id_ref_ed)
84  		 *
85  		 *   --> for ValidityAuditStrategy:
86  		 *     f.revision <= :revision and (f.endRevision > :revision or f.endRevision is null)
87  		 *
88  		 *     AND
89  		 *
90  		 * (the association at revision :revision)
91  		 *   --> for DefaultAuditStrategy:
92  		 *     ee.revision = (SELECT max(ee2.revision) FROM middleEntity ee2
93  		 *       WHERE ee2.revision <= :revision AND ee2.originalId.* = ee.originalId.*)
94  		 *
95  		 *   --> for ValidityAuditStrategy:
96  		 *     ee.revision <= :revision and (ee.endRevision > :revision or ee.endRevision is null)
97  		 *     and ( strtestent1_.REVEND>? or strtestent1_.REVEND is null )
98  		 *     and ( strtestent1_.REVEND>? or strtestent1_.REVEND is null )
99  		 *     and ( ternarymap0_.REVEND>? or ternarymap0_.REVEND is null )
100 		 *
101 		 * (only non-deleted entities and associations)
102 		 *     ee.revision_type != DEL AND
103 		 *     e.revision_type != DEL AND
104 		 *     f.revision_type != DEL
105 		 */
106 		final QueryBuilder commonPart = commonQueryPart(
107 				referencedIdData,
108 				indexIdData,
109 				versionsMiddleEntityName,
110 				verEntCfg.getOriginalIdPropName()
111 		);
112 		final QueryBuilder validQuery = commonPart.deepCopy();
113 		final QueryBuilder removedQuery = commonPart.deepCopy();
114 		createValidDataRestrictions(
115 				globalCfg, auditStrategy, referencedIdData, versionsMiddleEntityName, validQuery,
116 				validQuery.getRootParameters(), true, componentData
117 		);
118 		createValidAndRemovedDataRestrictions(
119 				globalCfg, auditStrategy, referencedIdData, versionsMiddleEntityName, removedQuery, componentData
120 		);
121 
122 		queryString = queryToString( validQuery );
123 		queryRemovedString = queryToString( removedQuery );
124 	}
125 
126 	/**
127 	 * Compute common part for both queries.
128 	 */
129 	private QueryBuilder commonQueryPart(
130 			MiddleIdData referencedIdData, MiddleIdData indexIdData,
131 			String versionsMiddleEntityName, String originalIdPropertyName) {
132 		final String eeOriginalIdPropertyPath = MIDDLE_ENTITY_ALIAS + "." + originalIdPropertyName;
133 		// SELECT new list(ee) FROM middleEntity ee
134 		final QueryBuilder qb = new QueryBuilder( versionsMiddleEntityName, MIDDLE_ENTITY_ALIAS );
135 		qb.addFrom( referencedIdData.getAuditEntityName(), REFERENCED_ENTITY_ALIAS );
136 		qb.addFrom( indexIdData.getAuditEntityName(), INDEX_ENTITY_ALIAS );
137 		qb.addProjection(
138 				"new list", MIDDLE_ENTITY_ALIAS + ", " + REFERENCED_ENTITY_ALIAS + ", " + INDEX_ENTITY_ALIAS,
139 				false, false
140 		);
141 		// WHERE
142 		final Parameters rootParameters = qb.getRootParameters();
143 		// ee.id_ref_ed = e.id_ref_ed
144 		referencedIdData.getPrefixedMapper().addIdsEqualToQuery(
145 				rootParameters, eeOriginalIdPropertyPath, referencedIdData.getOriginalMapper(),
146 				REFERENCED_ENTITY_ALIAS + "." + originalIdPropertyName
147 		);
148 		// ee.id_ref_ind = f.id_ref_ind
149 		indexIdData.getPrefixedMapper().addIdsEqualToQuery(
150 				rootParameters, eeOriginalIdPropertyPath, indexIdData.getOriginalMapper(),
151 				INDEX_ENTITY_ALIAS + "." + originalIdPropertyName
152 		);
153 		// ee.originalId.id_ref_ing = :id_ref_ing
154 		referencingIdData.getPrefixedMapper().addNamedIdEqualsToQuery( rootParameters, originalIdPropertyName, true );
155 		return qb;
156 	}

158 	/**
159 	 * Creates query restrictions used to retrieve only actual data.
160 	 */
161 	private void createValidDataRestrictions(
162 			GlobalConfiguration globalCfg, AuditStrategy auditStrategy,
163 			MiddleIdData referencedIdData, String versionsMiddleEntityName, QueryBuilder qb,
164 			Parameters rootParameters, boolean inclusive, MiddleComponentData... componentData) {
165 		final String revisionPropertyPath = verEntCfg.getRevisionNumberPath();
166 		final String originalIdPropertyName = verEntCfg.getOriginalIdPropName();
167 		final String eeOriginalIdPropertyPath = MIDDLE_ENTITY_ALIAS + "." + originalIdPropertyName;
168 		final String revisionTypePropName = getRevisionTypePath();
169 		// (selecting e entities at revision :revision)
170 		// --> based on auditStrategy (see above)
171 		auditStrategy.addEntityAtRevisionRestriction(
172 				globalCfg,
173 				qb,
174 				rootParameters,
175 				REFERENCED_ENTITY_ALIAS + "." + revisionPropertyPath,
176 				REFERENCED_ENTITY_ALIAS + "." + verEntCfg.getRevisionEndFieldName(),
177 				false,
178 				referencedIdData,
179 				revisionPropertyPath,
180 				originalIdPropertyName,
181 				REFERENCED_ENTITY_ALIAS,
182 				REFERENCED_ENTITY_ALIAS_DEF_AUD_STR,
183 				true
184 		);
185 		// (selecting f entities at revision :revision)
186 		// --> based on auditStrategy (see above)
187 		auditStrategy.addEntityAtRevisionRestriction(
188 				globalCfg,
189 				qb,
190 				rootParameters,
191 				INDEX_ENTITY_ALIAS + "." + revisionPropertyPath,
192 				INDEX_ENTITY_ALIAS + "." + verEntCfg.getRevisionEndFieldName(),
193 				false,
194 				referencedIdData,
195 				revisionPropertyPath,
196 				originalIdPropertyName,
197 				INDEX_ENTITY_ALIAS,
198 				INDEX_ENTITY_ALIAS_DEF_AUD_STR,
199 				true
200 		);
201 		// (with ee association at revision :revision)
202 		// --> based on auditStrategy (see above)
203 		auditStrategy.addAssociationAtRevisionRestriction(
204 				qb, rootParameters, revisionPropertyPath, verEntCfg.getRevisionEndFieldName(), true,
205 				referencingIdData, versionsMiddleEntityName, eeOriginalIdPropertyPath, revisionPropertyPath,
206 				originalIdPropertyName, MIDDLE_ENTITY_ALIAS, inclusive, componentData
207 		);
208 		// ee.revision_type != DEL
209 		rootParameters.addWhereWithNamedParam( revisionTypePropName, "!=", DEL_REVISION_TYPE_PARAMETER );
210 		// e.revision_type != DEL
211 		rootParameters.addWhereWithNamedParam(
212 				REFERENCED_ENTITY_ALIAS + "." + revisionTypePropName,
213 				false,
214 				"!=",
215 				DEL_REVISION_TYPE_PARAMETER
216 		);
217 		// f.revision_type != DEL
218 		rootParameters.addWhereWithNamedParam(
219 				INDEX_ENTITY_ALIAS + "." + revisionTypePropName,
220 				false,
221 				"!=",
222 				DEL_REVISION_TYPE_PARAMETER
223 		);
224 	}
225 
226 	/**
227 	 * Create query restrictions used to retrieve actual data and deletions that took place at exactly given revision.
228 	 */
229 	private void createValidAndRemovedDataRestrictions(
230 			GlobalConfiguration globalCfg, AuditStrategy auditStrategy,
231 			MiddleIdData referencedIdData, String versionsMiddleEntityName,
232 			QueryBuilder remQb, MiddleComponentData... componentData) {
233 		final Parameters disjoint = remQb.getRootParameters().addSubParameters( "or" );
234 		// Restrictions to match all valid rows.
235 		final Parameters valid = disjoint.addSubParameters( "and" );
236 		// Restrictions to match all rows deleted at exactly given revision.
237 		final Parameters removed = disjoint.addSubParameters( "and" );
238 		final String revisionPropertyPath = verEntCfg.getRevisionNumberPath();
239 		final String revisionTypePropName = getRevisionTypePath();
240 		// Excluding current revision, because we need to match data valid at the previous one.
241 		createValidDataRestrictions(
242 				globalCfg, auditStrategy, referencedIdData, versionsMiddleEntityName, remQb, valid, false, componentData
243 		);
244 		// ee.revision = :revision
245 		removed.addWhereWithNamedParam( revisionPropertyPath, "=", REVISION_PARAMETER );
246 		// e.revision = :revision
247 		removed.addWhereWithNamedParam(
248 				REFERENCED_ENTITY_ALIAS + "." + revisionPropertyPath,
249 				false,
250 				"=",
251 				REVISION_PARAMETER
252 		);
253 		// f.revision = :revision
254 		removed.addWhereWithNamedParam(
255 				INDEX_ENTITY_ALIAS + "." + revisionPropertyPath,
256 				false,
257 				"=",
258 				REVISION_PARAMETER
259 		);
260 		// ee.revision_type = DEL
261 		removed.addWhereWithNamedParam( revisionTypePropName, "=", DEL_REVISION_TYPE_PARAMETER );
262 		// e.revision_type = DEL
263 		removed.addWhereWithNamedParam(
264 				REFERENCED_ENTITY_ALIAS + "." + revisionTypePropName,
265 				false,
266 				"=",
267 				DEL_REVISION_TYPE_PARAMETER
268 		);
269 		// f.revision_type = DEL
270 		removed.addWhereWithNamedParam(
271 				INDEX_ENTITY_ALIAS + "." + revisionTypePropName,
272 				false,
273 				"=",
274 				DEL_REVISION_TYPE_PARAMETER
275 		);
276 	}
