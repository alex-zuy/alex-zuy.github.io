79  	public RevisionInfoConfiguration(GlobalConfiguration globalCfg) {
80  		this.globalCfg = globalCfg;
81  		if ( globalCfg.isUseRevisionEntityWithNativeId() ) {
82  			revisionInfoEntityName = "org.hibernate.envers.DefaultRevisionEntity";
83  		}
84  		else {
85  			revisionInfoEntityName = "org.hibernate.envers.enhanced.SequenceIdRevisionEntity";
86  		}
87  		revisionInfoIdData = new PropertyData( "id", "id", "field", null );
88  		revisionInfoTimestampData = new PropertyData( "timestamp", "timestamp", "field", null );
89  		modifiedEntityNamesData = new PropertyData( "modifiedEntityNames", "modifiedEntityNames", "field", null );
90  		revisionInfoTimestampType = new LongType();
91  
92  		revisionPropType = "integer";
93  	}

95  	private Document generateDefaultRevisionInfoXmlMapping() {
96  		final Document document = XMLHelper.getDocumentFactory().createDocument();
97  
98  		final Element classMapping = MetadataTools.createEntity(
99  				document,
100 				new AuditTableData( null, null, globalCfg.getDefaultSchemaName(), globalCfg.getDefaultCatalogName() ),
101 				null,
102 				null
103 		);
104 
105 		classMapping.addAttribute( "name", revisionInfoEntityName );
106 		classMapping.addAttribute( "table", "REVINFO" );
107 
108 		final Element idProperty = MetadataTools.addNativelyGeneratedId(
109 				classMapping,
110 				revisionInfoIdData.getName(),
111 				revisionPropType,
112 				globalCfg.isUseRevisionEntityWithNativeId()
113 		);
114 		MetadataTools.addColumn( idProperty, "REV", null, null, null, null, null, null, false );
115 
116 		final Element timestampProperty = MetadataTools.addProperty(
117 				classMapping,
118 				revisionInfoTimestampData.getName(),
119 				revisionInfoTimestampType.getName(),
120 				true,
121 				false
122 		);
123 		MetadataTools.addColumn( timestampProperty, "REVTSTMP", null, null, null, null, null, null, false );
124 
125 		if ( globalCfg.isTrackEntitiesChangedInRevision() ) {
126 			generateEntityNamesTrackingTableMapping(
127 					classMapping,
128 					"modifiedEntityNames",
129 					globalCfg.getDefaultSchemaName(),
130 					globalCfg.getDefaultCatalogName(),
131 					"REVCHANGES",
132 					"REV",
133 					"ENTITYNAME",
134 					"string"
135 			);
136 		}
137 
138 		return document;
139 	}

141 	/**
142 	 * Generates mapping that represents a set of primitive types.<br />
143 	 * <code>
144 	 * &lt;set name="propertyName" table="joinTableName" schema="joinTableSchema" catalog="joinTableCatalog"
145 	 * &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;cascade="persist, delete" lazy="false" fetch="join"&gt;<br />
146 	 * &nbsp;&nbsp;&nbsp;&lt;key column="joinTablePrimaryKeyColumnName" /&gt;<br />
147 	 * &nbsp;&nbsp;&nbsp;&lt;element type="joinTableValueColumnType"&gt;<br />
148 	 * &nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&lt;column name="joinTableValueColumnName" /&gt;<br />
149 	 * &nbsp;&nbsp;&nbsp;&lt;/element&gt;<br />
150 	 * &lt;/set&gt;
151 	 * </code>
152 	 */
153 	private void generateEntityNamesTrackingTableMapping(
154 			Element classMapping,
155 			String propertyName,
156 			String joinTableSchema,
157 			String joinTableCatalog,
158 			String joinTableName,
159 			String joinTablePrimaryKeyColumnName,
160 			String joinTableValueColumnName,
161 			String joinTableValueColumnType) {
162 		final Element set = classMapping.addElement( "set" );
163 		set.addAttribute( "name", propertyName );
164 		set.addAttribute( "table", joinTableName );
165 		set.addAttribute( "schema", joinTableSchema );
166 		set.addAttribute( "catalog", joinTableCatalog );
167 		set.addAttribute( "cascade", "persist, delete" );
168 		set.addAttribute( "fetch", "join" );
169 		set.addAttribute( "lazy", "false" );
170 		final Element key = set.addElement( "key" );
171 		key.addAttribute( "column", joinTablePrimaryKeyColumnName );
172 		final Element element = set.addElement( "element" );
173 		element.addAttribute( "type", joinTableValueColumnType );
174 		final Element column = element.addElement( "column" );
175 		column.addAttribute( "name", joinTableValueColumnName );
176 	}

178 	private Element generateRevisionInfoRelationMapping() {
179 		final Document document = XMLHelper.getDocumentFactory().createDocument();
180 		final Element revRelMapping = document.addElement( "key-many-to-one" );
181 		revRelMapping.addAttribute( "type", revisionPropType );
182 		revRelMapping.addAttribute( "class", revisionInfoEntityName );
183 
184 		if ( revisionPropSqlType != null ) {
185 			// Putting a fake name to make Hibernate happy. It will be replaced later anyway.
186 			MetadataTools.addColumn( revRelMapping, "*", null, null, null, revisionPropSqlType, null, null, false );
187 		}
188 
189 		return revRelMapping;
190 	}

192 	private void searchForRevisionInfoCfgInProperties(
193 			XClass clazz,
194 			ReflectionManager reflectionManager,
195 			MutableBoolean revisionNumberFound,
196 			MutableBoolean revisionTimestampFound,
197 			MutableBoolean modifiedEntityNamesFound,
198 			String accessType) {
199 		for ( XProperty property : clazz.getDeclaredProperties( accessType ) ) {
200 			final RevisionNumber revisionNumber = property.getAnnotation( RevisionNumber.class );
201 			final RevisionTimestamp revisionTimestamp = property.getAnnotation( RevisionTimestamp.class );
202 			final ModifiedEntityNames modifiedEntityNames = property.getAnnotation( ModifiedEntityNames.class );
203 
204 			if ( revisionNumber != null ) {
205 				if ( revisionNumberFound.isSet() ) {
206 					throw new MappingException( "Only one property may be annotated with @RevisionNumber!" );
207 				}
208 
209 				final XClass revisionNumberClass = property.getType();
210 				if ( reflectionManager.equals( revisionNumberClass, Integer.class ) ||
211 						reflectionManager.equals( revisionNumberClass, Integer.TYPE ) ) {
212 					revisionInfoIdData = new PropertyData( property.getName(), property.getName(), accessType, null );
213 					revisionNumberFound.set();
214 				}
215 				else if ( reflectionManager.equals( revisionNumberClass, Long.class ) ||
216 						reflectionManager.equals( revisionNumberClass, Long.TYPE ) ) {
217 					revisionInfoIdData = new PropertyData( property.getName(), property.getName(), accessType, null );
218 					revisionNumberFound.set();
219 
220 					// The default is integer
221 					revisionPropType = "long";
222 				}
223 				else {
224 					throw new MappingException(
225 							"The field annotated with @RevisionNumber must be of type " +
226 									"int, Integer, long or Long"
227 					);
228 				}
229 
230 				// Getting the @Column definition of the revision number property, to later use that info to
231 				// generate the same mapping for the relation from an audit table's revision number to the
232 				// revision entity revision number.
233 				final Column revisionPropColumn = property.getAnnotation( Column.class );
234 				if ( revisionPropColumn != null ) {
235 					revisionPropSqlType = revisionPropColumn.columnDefinition();
236 				}
237 			}
238 
239 			if ( revisionTimestamp != null ) {
240 				if ( revisionTimestampFound.isSet() ) {
241 					throw new MappingException( "Only one property may be annotated with @RevisionTimestamp!" );
242 				}
243 
244 				final XClass revisionTimestampClass = property.getType();
245 				if ( reflectionManager.equals( revisionTimestampClass, Long.class ) ||
246 						reflectionManager.equals( revisionTimestampClass, Long.TYPE ) ||
247 						reflectionManager.equals( revisionTimestampClass, Date.class ) ||
248 						reflectionManager.equals( revisionTimestampClass, java.sql.Date.class ) ) {
249 					revisionInfoTimestampData = new PropertyData(
250 							property.getName(),
251 							property.getName(),
252 							accessType,
253 							null
254 					);
255 					revisionTimestampFound.set();
256 				}
257 				else {
258 					throw new MappingException(
259 							"The field annotated with @RevisionTimestamp must be of type " +
260 									"long, Long, java.util.Date or java.sql.Date"
261 					);
262 				}
263 			}
264 
265 			if ( modifiedEntityNames != null ) {
266 				if ( modifiedEntityNamesFound.isSet() ) {
267 					throw new MappingException( "Only one property may be annotated with @ModifiedEntityNames!" );
268 				}
269 				final XClass modifiedEntityNamesClass = property.getType();
270 				if ( reflectionManager.equals( modifiedEntityNamesClass, Set.class ) &&
271 						reflectionManager.equals( property.getElementClass(), String.class ) ) {
272 					modifiedEntityNamesData = new PropertyData(
273 							property.getName(),
274 							property.getName(),
275 							accessType,
276 							null
277 					);
278 					modifiedEntityNamesFound.set();
279 				}
280 				else {
281 					throw new MappingException(
282 							"The field annotated with @ModifiedEntityNames must be of Set<String> type."
283 					);
284 				}
285 			}
286 		}
287 	}

289 	private void searchForRevisionInfoCfg(
290 			XClass clazz, ReflectionManager reflectionManager,
291 			MutableBoolean revisionNumberFound, MutableBoolean revisionTimestampFound,
292 			MutableBoolean modifiedEntityNamesFound) {
293 		final XClass superclazz = clazz.getSuperclass();
294 		if ( !"java.lang.Object".equals( superclazz.getName() ) ) {
295 			searchForRevisionInfoCfg(
296 					superclazz,
297 					reflectionManager,
298 					revisionNumberFound,
299 					revisionTimestampFound,
300 					modifiedEntityNamesFound
301 			);
302 		}
303 
304 		searchForRevisionInfoCfgInProperties(
305 				clazz, reflectionManager, revisionNumberFound, revisionTimestampFound,
306 				modifiedEntityNamesFound, "field"
307 		);
308 		searchForRevisionInfoCfgInProperties(
309 				clazz, reflectionManager, revisionNumberFound, revisionTimestampFound,
310 				modifiedEntityNamesFound, "property"
311 		);
312 	}

314 	public RevisionInfoConfigurationResult configure(Configuration cfg, ReflectionManager reflectionManager) {
315 		boolean revisionEntityFound = false;
316 		RevisionInfoGenerator revisionInfoGenerator = null;
317 		Class<?> revisionInfoClass = null;
318 
319 		final Iterator<PersistentClass> classes = cfg.getClassMappings();
320 		while ( classes.hasNext() ) {
321 			PersistentClass pc = classes.next();
322 			XClass clazz;
323 			try {
324 				clazz = reflectionManager.classForName( pc.getClassName(), this.getClass() );
325 			}
326 			catch (ClassNotFoundException e) {
327 				throw new MappingException( e );
328 			}
329 
330 			final RevisionEntity revisionEntity = clazz.getAnnotation( RevisionEntity.class );
331 			if ( revisionEntity != null ) {
332 				if ( revisionEntityFound ) {
333 					throw new MappingException( "Only one entity may be annotated with @RevisionEntity!" );
334 				}
335 
336 				// Checking if custom revision entity isn't audited
337 				if ( clazz.getAnnotation( Audited.class ) != null ) {
338 					throw new MappingException( "An entity annotated with @RevisionEntity cannot be audited!" );
339 				}
340 
341 				revisionEntityFound = true;
342 
343 				final MutableBoolean revisionNumberFound = new MutableBoolean();
344 				final MutableBoolean revisionTimestampFound = new MutableBoolean();
345 				final MutableBoolean modifiedEntityNamesFound = new MutableBoolean();
346 
347 				searchForRevisionInfoCfg(
348 						clazz,
349 						reflectionManager,
350 						revisionNumberFound,
351 						revisionTimestampFound,
352 						modifiedEntityNamesFound
353 				);
354 
355 				if ( !revisionNumberFound.isSet() ) {
356 					throw new MappingException(
357 							"An entity annotated with @RevisionEntity must have a field annotated " +
358 									"with @RevisionNumber!"
359 					);
360 				}
361 
362 				if ( !revisionTimestampFound.isSet() ) {
363 					throw new MappingException(
364 							"An entity annotated with @RevisionEntity must have a field annotated " +
365 									"with @RevisionTimestamp!"
366 					);
367 				}
368 
369 				revisionInfoEntityName = pc.getEntityName();
370 				revisionInfoClass = pc.getMappedClass();
371 				final Class<? extends RevisionListener> revisionListenerClass = getRevisionListenerClass( revisionEntity.value() );
372 				revisionInfoTimestampType = pc.getProperty( revisionInfoTimestampData.getName() ).getType();
373 				if ( globalCfg.isTrackEntitiesChangedInRevision()
374 						|| (globalCfg.isUseRevisionEntityWithNativeId() && DefaultTrackingModifiedEntitiesRevisionEntity.class
375 						.isAssignableFrom( revisionInfoClass ))
376 						|| (!globalCfg.isUseRevisionEntityWithNativeId() && SequenceIdTrackingModifiedEntitiesRevisionEntity.class
377 						.isAssignableFrom( revisionInfoClass ))
378 						|| modifiedEntityNamesFound.isSet() ) {
379 					// If tracking modified entities parameter is enabled, custom revision info entity is a subtype
380 					// of DefaultTrackingModifiedEntitiesRevisionEntity class, or @ModifiedEntityNames annotation is used.
381 					revisionInfoGenerator = new DefaultTrackingModifiedEntitiesRevisionInfoGenerator(
382 							revisionInfoEntityName,
383 							revisionInfoClass, revisionListenerClass, revisionInfoTimestampData, isTimestampAsDate(),
384 							modifiedEntityNamesData
385 					);
386 					globalCfg.setTrackEntitiesChangedInRevision( true );
387 				}
388 				else {
389 					revisionInfoGenerator = new DefaultRevisionInfoGenerator(
390 							revisionInfoEntityName, revisionInfoClass,
391 							revisionListenerClass, revisionInfoTimestampData, isTimestampAsDate()
392 					);
393 				}
394 			}
395 		}
396 
397 		// In case of a custom revision info generator, the mapping will be null.
398 		Document revisionInfoXmlMapping = null;
399 
400 		final Class<? extends RevisionListener> revisionListenerClass = getRevisionListenerClass( RevisionListener.class );
401 
402 		if ( revisionInfoGenerator == null ) {
403 			if ( globalCfg.isTrackEntitiesChangedInRevision() ) {
404 				revisionInfoClass = globalCfg.isUseRevisionEntityWithNativeId() ?
405 						DefaultTrackingModifiedEntitiesRevisionEntity.class
406 						:
407 						SequenceIdTrackingModifiedEntitiesRevisionEntity.class;
408 				revisionInfoEntityName = revisionInfoClass.getName();
409 				revisionInfoGenerator = new DefaultTrackingModifiedEntitiesRevisionInfoGenerator(
410 						revisionInfoEntityName, revisionInfoClass,
411 						revisionListenerClass, revisionInfoTimestampData, isTimestampAsDate(), modifiedEntityNamesData
412 				);
413 			}
414 			else {
415 				revisionInfoClass = globalCfg.isUseRevisionEntityWithNativeId() ? DefaultRevisionEntity.class
416 						: SequenceIdRevisionEntity.class;
417 				revisionInfoGenerator = new DefaultRevisionInfoGenerator(
418 						revisionInfoEntityName, revisionInfoClass,
419 						revisionListenerClass, revisionInfoTimestampData, isTimestampAsDate()
420 				);
421 			}
422 			revisionInfoXmlMapping = generateDefaultRevisionInfoXmlMapping();
423 		}
424 
425 		return new RevisionInfoConfigurationResult(
426 				revisionInfoGenerator, revisionInfoXmlMapping,
427 				new RevisionInfoQueryCreator(
428 						revisionInfoEntityName, revisionInfoIdData.getName(),
429 						revisionInfoTimestampData.getName(), isTimestampAsDate()
430 				),
431 				generateRevisionInfoRelationMapping(),
432 				new RevisionInfoNumberReader( revisionInfoClass, revisionInfoIdData ),
433 				globalCfg.isTrackEntitiesChangedInRevision() ? new ModifiedEntityNamesReader(
434 						revisionInfoClass,
435 						modifiedEntityNamesData
436 				)
437 						: null,
438 				revisionInfoEntityName, revisionInfoClass, revisionInfoTimestampData
439 		);
440 	}

442 	private boolean isTimestampAsDate() {
443 		final String typename = revisionInfoTimestampType.getName();
444 		return "date".equals( typename ) || "time".equals( typename ) || "timestamp".equals( typename );
445 	}

447 	/**
448 	 * @param defaultListener Revision listener that shall be applied if {@code org.hibernate.envers.revision_listener}
449 	 * parameter has not been set.
450 	 *
451 	 * @return Revision listener.
452 	 */
453 	private Class<? extends RevisionListener> getRevisionListenerClass(Class<? extends RevisionListener> defaultListener) {
454 		if ( globalCfg.getRevisionListenerClass() != null ) {
455 			return globalCfg.getRevisionListenerClass();
456 		}
457 		return defaultListener;
