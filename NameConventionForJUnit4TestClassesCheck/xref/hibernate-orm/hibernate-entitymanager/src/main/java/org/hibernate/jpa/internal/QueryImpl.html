113 	@SuppressWarnings({ "unchecked", "RedundantCast" })
114 	private void extractParameterInfo(Map<String,Class> namedParameterTypeRedefinition) {
115 		if ( ! org.hibernate.internal.AbstractQueryImpl.class.isInstance( query ) ) {
116 			throw new IllegalStateException( "Unknown query type for parameter extraction" );
117 		}
118 
119 		boolean hadJpaPositionalParameters = false;
120 
121 		final ParameterMetadata parameterMetadata = org.hibernate.internal.AbstractQueryImpl.class.cast( query ).getParameterMetadata();
122 
123 		// extract named params
124 		for ( String name : (Set<String>) parameterMetadata.getNamedParameterNames() ) {
125 			final NamedParameterDescriptor descriptor = parameterMetadata.getNamedParameterDescriptor( name );
126 			Class javaType = namedParameterTypeRedefinition.get( name );
127 			if ( javaType != null && mightNeedRedefinition( javaType, descriptor.getExpectedType() ) ) {
128 				descriptor.resetExpectedType(
129 						sfi().getTypeResolver().heuristicType( javaType.getName() )
130 				);
131 			}
132 			else if ( descriptor.getExpectedType() != null ) {
133 				javaType = descriptor.getExpectedType().getReturnedClass();
134 			}
135 
136 			if ( descriptor.isJpaStyle() ) {
137 				hadJpaPositionalParameters = true;
138 				final Integer position = Integer.valueOf( name );
139 				registerParameter( new JpaPositionalParameterRegistrationImpl( this, query, position, javaType ) );
140 			}
141 			else {
142 				registerParameter( new ParameterRegistrationImpl( this, query, name, javaType ) );
143 			}
144 		}
145 
146 		if ( hadJpaPositionalParameters ) {
147 			if ( parameterMetadata.getOrdinalParameterCount() > 0 ) {
148 				throw new IllegalArgumentException(
149 						"Cannot mix JPA positional parameters and native Hibernate positional/ordinal parameters"
150 				);
151 			}
152 		}
153 
154 		// extract Hibernate native positional parameters
155 		for ( int i = 0, max = parameterMetadata.getOrdinalParameterCount(); i < max; i++ ) {
156 			final OrdinalParameterDescriptor descriptor = parameterMetadata.getOrdinalParameterDescriptor( i + 1 );
157 			Class javaType = descriptor.getExpectedType() == null ? null : descriptor.getExpectedType().getReturnedClass();
158 			registerParameter( new ParameterRegistrationImpl( this, query, i+1, javaType ) );
159 		}
160 	}

166 	private boolean mightNeedRedefinition(Class javaType, Type expectedType) {
167 		// only redefine dates/times/timestamps that are not wrapped in a CompositeCustomType
168 		if ( expectedType == null )
169 			return java.util.Date.class.isAssignableFrom( javaType );
170 		else
171 			return java.util.Date.class.isAssignableFrom( javaType )
172 					&& !CompositeCustomType.class.isAssignableFrom( expectedType.getClass() );
173 	}

246 		@Override
247 		public void bindValue(T value) {
248 			validateBinding( getParameterType(), value, null );
249 
250 			if ( name != null ) {
251 				if ( value instanceof Collection ) {
252 					nativeQuery.setParameterList( name, (Collection) value );
253 				}
254 				else {
255 					nativeQuery.setParameter( name, value );
256 				}
257 			}
258 			else {
259 				nativeQuery.setParameter( position - 1, value );
260 			}
261 
262 			bind = new ParameterBindImpl<T>( value, null );
263 		}
264 
265 		@Override
266 		public void bindValue(T value, TemporalType specifiedTemporalType) {
267 			validateBinding( getParameterType(), value, specifiedTemporalType );
268 
269 			if ( Date.class.isInstance( value ) ) {
270 				if ( name != null ) {
271 					if ( specifiedTemporalType == DATE ) {
272 						nativeQuery.setDate( name, (Date) value );
273 					}
274 					else if ( specifiedTemporalType == TIME ) {
275 						nativeQuery.setTime( name, (Date) value );
276 					}
277 					else if ( specifiedTemporalType == TIMESTAMP ) {
278 						nativeQuery.setTimestamp( name, (Date) value );
279 					}
280 				}
281 				else {
282 					if ( specifiedTemporalType == DATE ) {
283 						nativeQuery.setDate( position - 1, (Date) value );
284 					}
285 					else if ( specifiedTemporalType == TIME ) {
286 						nativeQuery.setTime( position - 1, (Date) value );
287 					}
288 					else if ( specifiedTemporalType == TIMESTAMP ) {
289 						nativeQuery.setTimestamp( position - 1, (Date) value );
290 					}
291 				}
292 			}
293 			else if ( Calendar.class.isInstance( value ) ) {
294 				if ( name != null ) {
295 					if ( specifiedTemporalType == DATE ) {
296 						nativeQuery.setCalendarDate( name, (Calendar) value );
297 					}
298 					else if ( specifiedTemporalType == TIME ) {
299 						throw new IllegalArgumentException( "not yet implemented" );
300 					}
301 					else if ( specifiedTemporalType == TIMESTAMP ) {
302 						nativeQuery.setCalendar( name, (Calendar) value );
303 					}
304 				}
305 				else {
306 					if ( specifiedTemporalType == DATE ) {
307 						nativeQuery.setCalendarDate( position - 1, (Calendar) value );
308 					}
309 					else if ( specifiedTemporalType == TIME ) {
310 						throw new IllegalArgumentException( "not yet implemented" );
311 					}
312 					else if ( specifiedTemporalType == TIMESTAMP ) {
313 						nativeQuery.setCalendar( position - 1, (Calendar) value );
314 					}
315 				}
316 			}
317 			else {
318 				throw new IllegalArgumentException(
319 						"Unexpected type [" + value + "] passed with TemporalType; expecting Date or Calendar"
320 				);
321 			}
322 
323 			bind = new ParameterBindImpl<T>( value, specifiedTemporalType );
324 		}

443 	@SuppressWarnings({ "unchecked", "RedundantCast" })
444 	public List<X> getResultList() {
445 		getEntityManager().checkOpen( true );
446 		checkTransaction();
447 		beforeQuery();
448 		try {
449 			return list();
450 		}
451 		catch (QueryExecutionRequestException he) {
452 			throw new IllegalStateException(he);
453 		}
454 		catch( TypeMismatchException e ) {
455 			throw new IllegalArgumentException(e);
456 		}
457 		catch (HibernateException he) {
458 			throw getEntityManager().convert( he );
459 		}

462 	/**
463 	 * For JPA native SQL queries, we may need to perform a flush before executing the query.
464 	 */
465 	private void beforeQuery() {
466 		final org.hibernate.Query query = getHibernateQuery();
467 		if ( ! SQLQuery.class.isInstance( query ) ) {
468 			// this need only exists for native SQL queries, not JPQL or Criteria queries (both of which do
469 			// partial auto flushing already).
470 			return;
471 		}
472 
473 		final SQLQuery sqlQuery = (SQLQuery) query;
474 		if ( sqlQuery.getSynchronizedQuerySpaces() != null && ! sqlQuery.getSynchronizedQuerySpaces().isEmpty() ) {
475 			// The application defined query spaces on the Hibernate native SQLQuery which means the query will already
476 			// perform a partial flush according to the defined query spaces, no need to do a full flush.
477 			return;
478 		}
479 
480 		// otherwise we need to flush.  the query itself is not required to execute in a transaction; if there is
481 		// no transaction, the flush would throw a TransactionRequiredException which would potentially break existing
482 		// apps, so we only do the flush if a transaction is in progress.
483 		if ( getEntityManager().isTransactionInProgress() ) {
484 			getEntityManager().flush();
485 		}
486 	}

488 	@Override
489 	@SuppressWarnings({ "unchecked", "RedundantCast" })
490 	public X getSingleResult() {
491 		getEntityManager().checkOpen( true );
492 		checkTransaction();
493 		beforeQuery();
494 		try {
495 			final List<X> result = list();
496 
497 			if ( result.size() == 0 ) {
498 				NoResultException nre = new NoResultException( "No entity found for query" );
499 				getEntityManager().handlePersistenceException( nre );
500 				throw nre;
501 			}
502 			else if ( result.size() > 1 ) {
503 				final Set<X> uniqueResult = new HashSet<X>(result);
504 				if ( uniqueResult.size() > 1 ) {
505 					NonUniqueResultException nure = new NonUniqueResultException( "result returns more than one elements" );
506 					getEntityManager().handlePersistenceException( nure );
507 					throw nure;
508 				}
509 				else {
510 					return uniqueResult.iterator().next();
511 				}
512 			}
513 			else {
514 				return result.get( 0 );
515 			}
516 		}
517 		catch (QueryExecutionRequestException he) {
518 			throw new IllegalStateException(he);
519 		}
520 		catch( TypeMismatchException e ) {
521 			throw new IllegalArgumentException(e);
522 		}
523 		catch (HibernateException he) {
524 			throw getEntityManager().convert( he );
525 		}
526 	}

528 	@Override
529 	@SuppressWarnings({ "unchecked" })
530 	public <T> T unwrap(Class<T> tClass) {
531 		if ( org.hibernate.Query.class.isAssignableFrom( tClass ) ) {
532 			return (T) query;
533 		}
534 		if ( QueryImpl.class.isAssignableFrom( tClass ) ) {
535 			return (T) this;
536 		}
537 		if ( HibernateQuery.class.isAssignableFrom( tClass ) ) {
538 			return (T) this;
539 		}
540 
541 		throw new PersistenceException(
542 				String.format(
543 						"Unsure how to unwrap %s impl [%s] as requested type [%s]",
544 						Query.class.getSimpleName(),
545 						this.getClass().getName(),
546 						tClass.getName()
547 				)
548 		);
549 	}

565 	private List<X> list() {
566 		if (getEntityGraphQueryHint() != null) {
567 			SessionImplementor sessionImpl = (SessionImplementor) getEntityManager().getSession();
568 			HQLQueryPlan entityGraphQueryPlan = new HQLQueryPlan( getHibernateQuery().getQueryString(), false,
569 					sessionImpl.getEnabledFilters(), sessionImpl.getFactory(), getEntityGraphQueryHint() );
570 			// Safe to assume QueryImpl at this point.
571 			unwrap( org.hibernate.internal.QueryImpl.class ).setQueryPlan( entityGraphQueryPlan );
572 		}
573 		return query.list();
574 	}
