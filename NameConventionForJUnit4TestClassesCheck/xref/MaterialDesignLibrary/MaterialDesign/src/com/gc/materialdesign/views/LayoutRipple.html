16  	private Float xRippleOrigin;
17  	private Float yRippleOrigin;
18  	float rippleBorderRadius = 0;// radius
19  	
20  	public LayoutRipple(Context context, AttributeSet attrs) {
21  		super(context, attrs);
22  		setAttributes(attrs);
23  	}
24  	
25  	@Override
26  	protected void onInitDefaultValues() {
27  		minWidth = 20;
28  		minHeight = 20;
29  		backgroundColor = 0x00ffffff;
30  		rippleSpeed = 20f;// default speed
31  	}
32  	
33  	@Override
34  	protected void setAttributes(AttributeSet attrs) {
35  		super.setAttributes(attrs);
36  		//设定涟漪最外层的边界弧度，是圆角矩形，如果不设置，则是普通矩形
37  		rippleBorderRadius = attrs.getAttributeFloatValue(MATERIALDESIGNXML, "rippleBorderRadius", 0);
38  	}
39  	
40  	// ### RIPPLE EFFECT ###
41  	
42  	public Bitmap makeCircle() {
43  		Bitmap output = Bitmap.createBitmap(getWidth(), getHeight(), Config.ARGB_8888);
44  		x = (xRippleOrigin == null) ? x : xRippleOrigin;
45  		y = (yRippleOrigin == null) ? y : yRippleOrigin;
46  		return makeCircleFromBitmap(output);
47  	}
48  	
49  	@Override
50  	protected void onDraw(Canvas canvas) {
51  		super.onDraw(canvas);
52  		if (x != -1) {
53  			Rect src = new Rect(0, 0, getWidth(), getHeight());
54  			Rect dst = new Rect(0, 0, getWidth(), getHeight());
55  			canvas.drawBitmap(cropRoundRect(makeCircle()), src, dst, null);
56  		}
57  		invalidate();
58  	}
59  	
60  	/**
61  	 * @param bitmap
62  	 * @return 设置涟漪的边界，涟漪在这个区域里面可见。这里可以设置四角的弧度数
63  	 */
64  	public Bitmap cropRoundRect(Bitmap bitmap) {
65  	    Bitmap output = Bitmap.createBitmap(bitmap.getWidth(),
66  	    bitmap.getHeight(), Config.ARGB_8888);
67  	    Canvas canvas = new Canvas(output);
68  
69  	    final int color = 0xff424242;
70  	    final Paint paint = new Paint();
71  	    final Rect rect = new Rect(0, 0, bitmap.getWidth(), bitmap.getHeight());
72  
73  	    paint.setAntiAlias(true);
74  	    canvas.drawARGB(0, 0, 0, 0);
75  	    paint.setColor(color);
76  	    RectF rectF = new RectF(0, 0, bitmap.getWidth(), bitmap.getHeight());
77  	    canvas.drawRoundRect(rectF, rippleBorderRadius, rippleBorderRadius, paint);
78  
79  	    paint.setXfermode(new PorterDuffXfermode(Mode.SRC_IN));
80  	    canvas.drawBitmap(bitmap, rect, rect, paint);
81  	    return output;
82  	}
