45  public class JmsListenerAnnotationBeanPostProcessorTests {
46  
47  	@Test
48  	public void simpleMessageListener() {
49  		ConfigurableApplicationContext context = new AnnotationConfigApplicationContext(
50  				Config.class, SimpleMessageListenerTestBean.class);
51  
52  		JmsListenerContainerTestFactory factory = context.getBean(JmsListenerContainerTestFactory.class);
53  		assertEquals("One container should have been registered", 1, factory.getListenerContainers().size());
54  		MessageListenerTestContainer container = factory.getListenerContainers().get(0);
55  
56  		JmsListenerEndpoint endpoint = container.getEndpoint();
57  		assertEquals("Wrong endpoint type", MethodJmsListenerEndpoint.class, endpoint.getClass());
58  		MethodJmsListenerEndpoint methodEndpoint = (MethodJmsListenerEndpoint) endpoint;
59  		assertNotNull(methodEndpoint.getBean());
60  		assertNotNull(methodEndpoint.getMethod());
61  
62  		SimpleMessageListenerContainer listenerContainer = new SimpleMessageListenerContainer();
63  		methodEndpoint.setupListenerContainer(listenerContainer);
64  		assertNotNull(listenerContainer.getMessageListener());
65  
66  		assertTrue("Should have been started " + container, container.isStarted());
67  		context.close(); // Close and stop the listeners
68  		assertTrue("Should have been stopped " + container, container.isStopped());
69  	}
70  
71  	@Test
72  	public void metaAnnotationIsDiscovered() {
73  		ConfigurableApplicationContext context = new AnnotationConfigApplicationContext(
74  				Config.class, MetaAnnotationTestBean.class);
75  
76  		JmsListenerContainerTestFactory factory = context.getBean(JmsListenerContainerTestFactory.class);
77  		assertEquals("one container should have been registered", 1, factory.getListenerContainers().size());
78  		JmsListenerEndpoint endpoint = factory.getListenerContainers().get(0).getEndpoint();
79  		assertEquals("metaTestQueue", ((AbstractJmsListenerEndpoint) endpoint).getDestination());
80  	}
