101         public static AutoImportDangledState fromString(String value) {
102             if ("no".equalsIgnoreCase(value)) {
103                 return NO;
104             } else if ("yes".equalsIgnoreCase(value)) {
105                 return YES;
106             } else if ("closed".equalsIgnoreCase(value)) {
107                 return CLOSED;

133     @Inject
134     public GatewayMetaState(Settings settings, ThreadPool threadPool, NodeEnvironment nodeEnv,
135                             TransportNodesListGatewayMetaState nodesListGatewayMetaState, LocalAllocateDangledIndices allocateDangledIndices,
136                             NodeIndexDeletedAction nodeIndexDeletedAction) throws Exception {
137         super(settings);
138         this.nodeEnv = nodeEnv;
139         this.threadPool = threadPool;
140         this.format = XContentType.fromRestContentType(settings.get("format", "smile"));
141         this.allocateDangledIndices = allocateDangledIndices;
142         this.nodeIndexDeletedAction = nodeIndexDeletedAction;
143         nodesListGatewayMetaState.init(this);
144 
145         if (this.format == XContentType.SMILE) {
146             Map<String, String> params = Maps.newHashMap();
147             params.put("binary", "true");
148             formatParams = new ToXContent.MapParams(params);
149             Map<String, String> gatewayModeParams = Maps.newHashMap();
150             gatewayModeParams.put("binary", "true");
151             gatewayModeParams.put(MetaData.CONTEXT_MODE_PARAM, MetaData.CONTEXT_MODE_GATEWAY);
152             gatewayModeFormatParams = new ToXContent.MapParams(gatewayModeParams);
153         } else {
154             formatParams = ToXContent.EMPTY_PARAMS;
155             Map<String, String> gatewayModeParams = Maps.newHashMap();
156             gatewayModeParams.put(MetaData.CONTEXT_MODE_PARAM, MetaData.CONTEXT_MODE_GATEWAY);
157             gatewayModeFormatParams = new ToXContent.MapParams(gatewayModeParams);
158         }
159 
160         this.autoImportDangled = AutoImportDangledState.fromString(settings.get(GATEWAY_AUTO_IMPORT_DANGLED, settings.get(GATEWAY_LOCAL_AUTO_IMPORT_DANGLED, AutoImportDangledState.YES.toString())));
161         this.danglingTimeout = settings.getAsTime(GATEWAY_DANGLING_TIMEOUT, settings.getAsTime(GATEWAY_LOCAL_DANGLING_TIMEOUT, TimeValue.timeValueHours(2)));
162 
163         logger.debug("using gateway.local.auto_import_dangled [{}], with gateway.dangling_timeout [{}]", this.autoImportDangled, this.danglingTimeout);
164         if (DiscoveryNode.masterNode(settings) || DiscoveryNode.dataNode(settings)) {
165             nodeEnv.ensureAtomicMoveSupported();
166         }
167         if (DiscoveryNode.masterNode(settings)) {
168             try {
169                 ensureNoPre019State();
170                 pre20Upgrade();
171                 long start = System.currentTimeMillis();
172                 loadState();
173                 logger.debug("took {} to load state", TimeValue.timeValueMillis(System.currentTimeMillis() - start));
174             } catch (Exception e) {
175                 logger.error("failed to read local state, exiting...", e);
176                 throw e;
177             }
178         }

189     @Override
190     public void clusterChanged(ClusterChangedEvent event) {
191         if (event.state().blocks().disableStatePersistence()) {
192             // reset the current metadata, we need to start fresh...
193             this.currentMetaData = null;
194             return;
195         }
196 
197         MetaData newMetaData = event.state().metaData();
198         // we don't check if metaData changed, since we might be called several times and we need to check dangling...
199 
200         boolean success = true;
201         // only applied to master node, writing the global and index level states
202         if (event.state().nodes().localNode().masterNode()) {
203             // check if the global state changed?
204             if (currentMetaData == null || !MetaData.isGlobalStateEquals(currentMetaData, newMetaData)) {
205                 try {
206                     writeGlobalState("changed", newMetaData);
207                 } catch (Throwable e) {
208                     success = false;
209                 }
210             }
211 
212             // check and write changes in indices
213             for (IndexMetaData indexMetaData : newMetaData) {
214                 String writeReason = null;
215                 IndexMetaData currentIndexMetaData;
216                 if (currentMetaData == null) {
217                     // a new event..., check from the state stored
218                     try {
219                         currentIndexMetaData = loadIndexState(indexMetaData.index());
220                     } catch (IOException ex) {
221                         throw new ElasticsearchException("failed to load index state", ex);
222                     }
223                 } else {
224                     currentIndexMetaData = currentMetaData.index(indexMetaData.index());
225                 }
226                 if (currentIndexMetaData == null) {
227                     writeReason = "freshly created";
228                 } else if (currentIndexMetaData.version() != indexMetaData.version()) {
229                     writeReason = "version changed from [" + currentIndexMetaData.version() + "] to [" + indexMetaData.version() + "]";
230                 }
231 
232                 // we update the writeReason only if we really need to write it
233                 if (writeReason == null) {
234                     continue;
235                 }
236 
237                 try {
238                     writeIndex(writeReason, indexMetaData, currentIndexMetaData);
239                 } catch (Throwable e) {
240                     success = false;
241                 }
242             }
243         }
244 
245         // delete indices that were there before, but are deleted now
246         // we need to do it so they won't be detected as dangling
247         if (currentMetaData != null) {
248             // only delete indices when we already received a state (currentMetaData != null)
249             // and we had a go at processing dangling indices at least once
250             // this will also delete the _state of the index itself
251             for (IndexMetaData current : currentMetaData) {
252                 if (danglingIndices.containsKey(current.index())) {
253                     continue;
254                 }
255                 if (!newMetaData.hasIndex(current.index())) {
256                     logger.debug("[{}] deleting index that is no longer part of the metadata (indices: [{}])", current.index(), newMetaData.indices().keys());
257                     if (nodeEnv.hasNodeFile()) {
258                         try {
259                             final Index idx = new Index(current.index());
260                             MetaDataStateFormat.deleteMetaState(nodeEnv.indexPaths(idx));
261                             nodeEnv.deleteIndexDirectorySafe(idx);
262                         } catch (LockObtainFailedException ex) {
263                             logger.debug("[{}] failed to delete index - at least one shards is still locked", ex, current.index());
264                         } catch (Exception ex) {
265                             logger.warn("[{}] failed to delete index", ex, current.index());
266                         }
267                     }
268                     try {
269                         nodeIndexDeletedAction.nodeIndexStoreDeleted(event.state(), current.index(), event.state().nodes().localNodeId());
270                     } catch (Throwable e) {
271                         logger.debug("[{}] failed to notify master on local index store deletion", e, current.index());
272                     }
273                 }
274             }
275         }
276 
277         // handle dangling indices, we handle those for all nodes that have a node file (data or master)
278         if (nodeEnv.hasNodeFile()) {
279             if (danglingTimeout.millis() >= 0) {
280                 synchronized (danglingMutex) {
281                     for (String danglingIndex : danglingIndices.keySet()) {
282                         if (newMetaData.hasIndex(danglingIndex)) {
283                             logger.debug("[{}] no longer dangling (created), removing", danglingIndex);
284                             DanglingIndex removed = danglingIndices.remove(danglingIndex);
285                             FutureUtils.cancel(removed.future);
286                         }
287                     }
288                     // delete indices that are no longer part of the metadata
289                     try {
290                         for (String indexName : nodeEnv.findAllIndices()) {
291                             // if we have the index on the metadata, don't delete it
292                             if (newMetaData.hasIndex(indexName)) {
293                                 continue;
294                             }
295                             if (danglingIndices.containsKey(indexName)) {
296                                 // already dangling, continue
297                                 continue;
298                             }
299                             final IndexMetaData indexMetaData = loadIndexState(indexName);
300                             final Index index = new Index(indexName);
301                             if (indexMetaData != null) {
302                                 try {
303                                     // the index deletion might not have worked due to shards still being locked
304                                     // we have three cases here:
305                                     //  - we acquired all shards locks here --> we can import the dangeling index
306                                     //  - we failed to acquire the lock --> somebody else uses it - DON'T IMPORT
307                                     //  - we acquired successfully but the lock list is empty --> no shards present - DON'T IMPORT
308                                     // in the last case we should in-fact try to delete the directory since it might be a leftover...
309                                     final List<ShardLock> shardLocks = nodeEnv.lockAllForIndex(index);
310                                     if (shardLocks.isEmpty()) {
311                                         // no shards - try to remove the directory
312                                         nodeEnv.deleteIndexDirectorySafe(index);
313                                         continue;
314                                     }
315                                     IOUtils.closeWhileHandlingException(shardLocks);
316                                 } catch (IOException ex) {
317                                     logger.warn("[{}] skipping locked dangling index, exists on local file system, but not in cluster metadata, auto import to cluster state is set to [{}]", ex, indexName, autoImportDangled);
318                                     continue;
319                                 }
320                                 if(autoImportDangled.shouldImport()){
321                                     logger.info("[{}] dangling index, exists on local file system, but not in cluster metadata, auto import to cluster state [{}]", indexName, autoImportDangled);
322                                     danglingIndices.put(indexName, new DanglingIndex(indexName, null));
323                                 } else if (danglingTimeout.millis() == 0) {
324                                     logger.info("[{}] dangling index, exists on local file system, but not in cluster metadata, timeout set to 0, deleting now", indexName);
325                                     try {
326                                         nodeEnv.deleteIndexDirectorySafe(index);
327                                     } catch (LockObtainFailedException ex) {
328                                         logger.debug("[{}] failed to delete index - at least one shards is still locked", ex, indexName);
329                                     } catch (Exception ex) {
330                                         logger.warn("[{}] failed to delete dangling index", ex, indexName);
331                                     }
332                                 } else {
333                                     logger.info("[{}] dangling index, exists on local file system, but not in cluster metadata, scheduling to delete in [{}], auto import to cluster state [{}]", indexName, danglingTimeout, autoImportDangled);
334                                     danglingIndices.put(indexName, new DanglingIndex(indexName, threadPool.schedule(danglingTimeout, ThreadPool.Names.SAME, new RemoveDanglingIndex(index))));
335                                 }
336                             }
337                         }
338                     } catch (Throwable e) {
339                         logger.warn("failed to find dangling indices", e);
340                     }
341                 }
342             }
343             if (autoImportDangled.shouldImport() && !danglingIndices.isEmpty()) {
344                 final List<IndexMetaData> dangled = Lists.newArrayList();
345                 for (String indexName : danglingIndices.keySet()) {
346                     IndexMetaData indexMetaData;
347                     try {
348                         indexMetaData = loadIndexState(indexName);
349                     } catch (IOException ex) {
350                         throw new ElasticsearchException("failed to load index state", ex);
351                     }
352                     if (indexMetaData == null) {
353                         logger.debug("failed to find state for dangling index [{}]", indexName);
354                         continue;
355                     }
356                     // we might have someone copying over an index, renaming the directory, handle that
357                     if (!indexMetaData.index().equals(indexName)) {
358                         logger.info("dangled index directory name is [{}], state name is [{}], renaming to directory name", indexName, indexMetaData.index());
359                         indexMetaData = IndexMetaData.builder(indexMetaData).index(indexName).build();
360                     }
361                     if (autoImportDangled == AutoImportDangledState.CLOSED) {
362                         indexMetaData = IndexMetaData.builder(indexMetaData).state(IndexMetaData.State.CLOSE).build();
363                     }
364                     if (indexMetaData != null) {
365                         dangled.add(indexMetaData);
366                     }
367                 }
368                 IndexMetaData[] dangledIndices = dangled.toArray(new IndexMetaData[dangled.size()]);
369                 try {
370                     allocateDangledIndices.allocateDangled(dangledIndices, new LocalAllocateDangledIndices.Listener() {
371                         @Override
372                         public void onResponse(LocalAllocateDangledIndices.AllocateDangledResponse response) {
373                             logger.trace("allocated dangled");
374                         }
375 
376                         @Override
377                         public void onFailure(Throwable e) {
378                             logger.info("failed to send allocated dangled", e);
379                         }
380                     });
381                 } catch (Throwable e) {
382                     logger.warn("failed to send allocate dangled", e);
383                 }
384             }
385         }
386 
387         if (success) {
388             currentMetaData = newMetaData;
389         }
390     }

427     private void writeIndex(String reason, IndexMetaData indexMetaData, @Nullable IndexMetaData previousIndexMetaData) throws Exception {
428         logger.trace("[{}] writing state, reason [{}]", indexMetaData.index(), reason);
429         final boolean deleteOldFiles = previousIndexMetaData != null && previousIndexMetaData.version() != indexMetaData.version();
430         final MetaDataStateFormat<IndexMetaData> writer = indexStateFormat(format, formatParams, deleteOldFiles);
431         try {
432             writer.write(indexMetaData, INDEX_STATE_FILE_PREFIX, indexMetaData.version(), nodeEnv.indexPaths(new Index(indexMetaData.index())));
433         } catch (Throwable ex) {
434             logger.warn("[{}]: failed to write index state", ex, indexMetaData.index());
435             throw new IOException("failed to write state for [" + indexMetaData.index() + "]", ex);
436         }
437     }

439     private void writeGlobalState(String reason, MetaData metaData) throws Exception {
440         logger.trace("{} writing state, reason [{}]", GLOBAL_STATE_LOG_TYPE, reason);
441         final MetaDataStateFormat<MetaData> writer = globalStateFormat(format, gatewayModeFormatParams, true);
442         try {
443             writer.write(metaData, GLOBAL_STATE_FILE_PREFIX, metaData.version(), nodeEnv.nodeDataPaths());
444         } catch (Throwable ex) {
445             logger.warn("{}: failed to write global state", ex, GLOBAL_STATE_LOG_TYPE);
446             throw new IOException("failed to write global state", ex);
447         }
448     }

450     private MetaData loadState() throws Exception {
451         MetaData globalMetaData = loadGlobalState();
452         MetaData.Builder metaDataBuilder;
453         if (globalMetaData != null) {
454             metaDataBuilder = MetaData.builder(globalMetaData);
455         } else {
456             metaDataBuilder = MetaData.builder();
457         }
458 
459         final Set<String> indices = nodeEnv.findAllIndices();
460         for (String index : indices) {
461             IndexMetaData indexMetaData = loadIndexState(index);
462             if (indexMetaData == null) {
463                 logger.debug("[{}] failed to find metadata for existing index location", index);
464             } else {
465                 metaDataBuilder.put(indexMetaData, false);
466             }
467         }
468         return metaDataBuilder.build();
469     }

481     /**
482      * Throws an IAE if a pre 0.19 state is detected
483      */
484     private void ensureNoPre019State() throws Exception {
485         for (Path dataLocation : nodeEnv.nodeDataPaths()) {
486             final Path stateLocation = dataLocation.resolve(MetaDataStateFormat.STATE_DIR_NAME);
487             if (!Files.exists(stateLocation)) {
488                 continue;
489             }
490             try (DirectoryStream<Path> stream = Files.newDirectoryStream(stateLocation)) {
491                 for (Path stateFile : stream) {
492                     if (logger.isTraceEnabled()) {
493                         logger.trace("[upgrade]: processing [" + stateFile.getFileName() + "]");
494                     }
495                     final String name = stateFile.getFileName().toString();
496                     if (name.startsWith("metadata-")) {
497                         throw new ElasticsearchIllegalStateException("Detected pre 0.19 metadata file please upgrade to a version before "
498                                 + Version.CURRENT.minimumCompatibilityVersion()
499                                 + " first to upgrade state structures - metadata found: [" + stateFile.getParent().toAbsolutePath());
500                     }
501                 }
502             }
503         }
504     }

506     /**
507      * Elasticsearch 2.0 deprecated custom routing hash functions. So what we do here is that for old indices, we
508      * move this old & deprecated node setting to an index setting so that we can keep things backward compatible.
509      */
510     private void pre20Upgrade() throws Exception {
511         final Class<? extends HashFunction> pre20HashFunction = settings.getAsClass(DEPRECATED_SETTING_ROUTING_HASH_FUNCTION, null, "org.elasticsearch.cluster.routing.operation.hash.", "HashFunction");
512         final Boolean pre20UseType = settings.getAsBoolean(DEPRECATED_SETTING_ROUTING_USE_TYPE, null);
513         MetaData metaData = loadMetaState();
514         for (IndexMetaData indexMetaData : metaData) {
515             if (indexMetaData.settings().get(IndexMetaData.SETTING_LEGACY_ROUTING_HASH_FUNCTION) == null
516                     && indexMetaData.getCreationVersion().before(Version.V_2_0_0)) {
517                 // these settings need an upgrade
518                 Settings indexSettings = ImmutableSettings.builder().put(indexMetaData.settings())
519                         .put(IndexMetaData.SETTING_LEGACY_ROUTING_HASH_FUNCTION, pre20HashFunction == null ? DjbHashFunction.class : pre20HashFunction)
520                         .put(IndexMetaData.SETTING_LEGACY_ROUTING_USE_TYPE, pre20UseType == null ? false : pre20UseType)
521                         .build();
522                 IndexMetaData newMetaData = IndexMetaData.builder(indexMetaData)
523                         .version(indexMetaData.version())
524                         .settings(indexSettings)
525                         .build();
526                 writeIndex("upgrade", newMetaData, null);
527             } else if (indexMetaData.getCreationVersion().onOrAfter(Version.V_2_0_0)) {
528                 if (indexMetaData.getSettings().get(IndexMetaData.SETTING_LEGACY_ROUTING_HASH_FUNCTION) != null
529                         || indexMetaData.getSettings().get(IndexMetaData.SETTING_LEGACY_ROUTING_USE_TYPE) != null) {
530                     throw new ElasticsearchIllegalStateException("Indices created on or after 2.0 should NOT contain [" + IndexMetaData.SETTING_LEGACY_ROUTING_HASH_FUNCTION
531                             + "] + or [" + IndexMetaData.SETTING_LEGACY_ROUTING_USE_TYPE + "] in their index settings");
532                 }
533             }
534         }
535         if (pre20HashFunction != null || pre20UseType != null) {
536             logger.warn("Settings [{}] and [{}] are deprecated. Index settings from your old indices have been updated to record the fact that they "
537                     + "used some custom routing logic, you can now remove these settings from your `elasticsearch.yml` file", DEPRECATED_SETTING_ROUTING_HASH_FUNCTION, DEPRECATED_SETTING_ROUTING_USE_TYPE);
538         }
539     }

541     class RemoveDanglingIndex implements Runnable {
542 
543         private final Index index;
544 
545         RemoveDanglingIndex(Index index) {
546             this.index = index;
547         }
548 
549         @Override
550         public void run() {
551             synchronized (danglingMutex) {
552                 DanglingIndex remove = danglingIndices.remove(index.name());
553                 // no longer there...
554                 if (remove == null) {
555                     return;
556                 }
557                 logger.warn("[{}] deleting dangling index", index);
558 
559                 try {
560                     MetaDataStateFormat.deleteMetaState(nodeEnv.indexPaths(index));
561                     nodeEnv.deleteIndexDirectorySafe(index);
562                 } catch (Exception ex) {
563                     logger.debug("failed to delete dangling index", ex);
564                 }
565             }
566         }
567     }
