293         String polygonGeoJson = XContentFactory.jsonBuilder().startObject().field("type", "Polygon")
294                 .startArray("coordinates")
295                 .startArray()
296                 .startArray().value(100.0).value(1.0).endArray()
297                 .startArray().value(101.0).value(1.0).endArray()
298                 .startArray().value(101.0).value(0.0).endArray()
299                 .startArray().value(100.0).value(0.0).endArray()
300                 .startArray().value(100.0).value(1.0).endArray()
301                 .endArray()
302                 .startArray()
303                 .startArray().value(100.2).value(0.8).endArray()
304                 .startArray().value(100.2).value(0.2).endArray()
305                 .startArray().value(100.8).value(0.2).endArray()
306                 .startArray().value(100.8).value(0.8).endArray()
307                 .startArray().value(100.2).value(0.8).endArray()
308                 .endArray()
309                 .endArray()
310                 .endObject().string();
311 
312         List<Coordinate> shellCoordinates = new ArrayList<>();
313         shellCoordinates.add(new Coordinate(100, 0));
314         shellCoordinates.add(new Coordinate(101, 0));
315         shellCoordinates.add(new Coordinate(101, 1));
316         shellCoordinates.add(new Coordinate(100, 1));
317         shellCoordinates.add(new Coordinate(100, 0));
318 
319         List<Coordinate> holeCoordinates = new ArrayList<>();
320         holeCoordinates.add(new Coordinate(100.2, 0.2));
321         holeCoordinates.add(new Coordinate(100.8, 0.2));
322         holeCoordinates.add(new Coordinate(100.8, 0.8));
323         holeCoordinates.add(new Coordinate(100.2, 0.8));
324         holeCoordinates.add(new Coordinate(100.2, 0.2));
325 
326         LinearRing shell = GEOMETRY_FACTORY.createLinearRing(
327                 shellCoordinates.toArray(new Coordinate[shellCoordinates.size()]));
328         LinearRing[] holes = new LinearRing[1];
329         holes[0] = GEOMETRY_FACTORY.createLinearRing(
330                 holeCoordinates.toArray(new Coordinate[holeCoordinates.size()]));
331         Polygon expected = GEOMETRY_FACTORY.createPolygon(shell, holes);

350     @Test
351     public void testParse_multiPolygon() throws IOException {
352         String multiPolygonGeoJson = XContentFactory.jsonBuilder().startObject().field("type", "MultiPolygon")
353                 .startArray("coordinates")
354                 .startArray()//first poly (without holes)
355                 .startArray()
356                 .startArray().value(102.0).value(2.0).endArray()

358                 .startArray().value(103.0).value(3.0).endArray()

360                 .startArray().value(102.0).value(2.0).endArray()
361                 .endArray()
362                 .endArray()
363                 .startArray()//second poly (with hole)
364                 .startArray()
365                 .startArray().value(100.0).value(0.0).endArray()
366                 .startArray().value(101.0).value(0.0).endArray()
367                 .startArray().value(101.0).value(1.0).endArray()
368                 .startArray().value(100.0).value(1.0).endArray()
369                 .startArray().value(100.0).value(0.0).endArray()
370                 .endArray()
371                 .startArray()//hole
372                 .startArray().value(100.2).value(0.8).endArray()
373                 .startArray().value(100.2).value(0.2).endArray()
374                 .startArray().value(100.8).value(0.2).endArray()
375                 .startArray().value(100.8).value(0.8).endArray()
376                 .startArray().value(100.2).value(0.8).endArray()
377                 .endArray()
378                 .endArray()
379                 .endArray()
380                 .endObject().string();
381 
382         List<Coordinate> shellCoordinates = new ArrayList<>();
383         shellCoordinates.add(new Coordinate(100, 0));
384         shellCoordinates.add(new Coordinate(101, 0));
385         shellCoordinates.add(new Coordinate(101, 1));
386         shellCoordinates.add(new Coordinate(100, 1));
387         shellCoordinates.add(new Coordinate(100, 0));
388 
389         List<Coordinate> holeCoordinates = new ArrayList<>();
390         holeCoordinates.add(new Coordinate(100.2, 0.2));
391         holeCoordinates.add(new Coordinate(100.8, 0.2));
392         holeCoordinates.add(new Coordinate(100.8, 0.8));
393         holeCoordinates.add(new Coordinate(100.2, 0.8));
394         holeCoordinates.add(new Coordinate(100.2, 0.2));
395 
396         LinearRing shell = GEOMETRY_FACTORY.createLinearRing(shellCoordinates.toArray(new Coordinate[shellCoordinates.size()]));
397         LinearRing[] holes = new LinearRing[1];
398         holes[0] = GEOMETRY_FACTORY.createLinearRing(holeCoordinates.toArray(new Coordinate[holeCoordinates.size()]));
399         Polygon withHoles = GEOMETRY_FACTORY.createPolygon(shell, holes);
400 
401         shellCoordinates = new ArrayList<>();
402         shellCoordinates.add(new Coordinate(102, 3));
403         shellCoordinates.add(new Coordinate(103, 3));
404         shellCoordinates.add(new Coordinate(103, 2));
405         shellCoordinates.add(new Coordinate(102, 2));
406         shellCoordinates.add(new Coordinate(102, 3));
407 
408 
409         shell = GEOMETRY_FACTORY.createLinearRing(shellCoordinates.toArray(new Coordinate[shellCoordinates.size()]));
410         Polygon withoutHoles = GEOMETRY_FACTORY.createPolygon(shell, null);
411 
412         Shape expected = shapeCollection(withoutHoles, withHoles);
413 
414         assertGeometryEquals(expected, multiPolygonGeoJson);
415     }

471     private void assertGeometryEquals(Shape expected, String geoJson) throws IOException {
472         XContentParser parser = JsonXContent.jsonXContent.createParser(geoJson);
473         parser.nextToken();
474         ElasticsearchGeoAssertions.assertEquals(expected, ShapeBuilder.parse(parser).build());
475     }

477     private ShapeCollection<Shape> shapeCollection(Shape... shapes) {
478         return new ShapeCollection<>(Arrays.asList(shapes), SPATIAL_CONTEXT);
479     }
480 
481     private ShapeCollection<Shape> shapeCollection(Geometry... geoms) {
482         List<Shape> shapes = new ArrayList<>(geoms.length);
483         for (Geometry geom : geoms) {
484             shapes.add(jtsGeom(geom));
485         }
486         return new ShapeCollection<>(shapes, SPATIAL_CONTEXT);
487     }
488 
489     private JtsGeometry jtsGeom(Geometry geom) {
490         return new JtsGeometry(geom, SPATIAL_CONTEXT, false, false);
491     }
