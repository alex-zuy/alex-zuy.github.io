24  package org.hibernate.jpamodelgen.test.xmlonly;
25  
26  import org.hibernate.jpamodelgen.JPAMetaModelEntityProcessor;
27  import org.hibernate.jpamodelgen.test.util.CompilationTest;
28  import org.hibernate.jpamodelgen.test.util.WithClasses;
29  import org.hibernate.jpamodelgen.test.util.WithProcessorOption;
30  import org.junit.Test;
31  
32  import static org.hibernate.jpamodelgen.test.util.TestUtil.assertMetamodelClassGeneratedFor;
33  import static org.hibernate.jpamodelgen.test.util.TestUtil.assertPresenceOfFieldInMetamodelFor;
34  
35  /**
36   * @author Hardy Ferentschik
37   */
38  @WithClasses({ Car.class, Course.class, Option.class, Period.class, Teacher.class, Tire.class, XmlOnly.class })
39  @WithProcessorOption(key = JPAMetaModelEntityProcessor.PERSISTENCE_XML_OPTION,
40  		value = "org/hibernate/jpamodelgen/test/xmlonly/persistence.xml")
41  public class XmlOnlyTest extends CompilationTest {
42  	@Test
43  	public void testMetaModelGeneratedForXmlConfiguredEntity() {
44  		assertMetamodelClassGeneratedFor( XmlOnly.class );
45  	}
46  
47  	@Test
48  	public void testMetaModelGeneratedForManyToManyFieldAccessWithoutTargetEntity() {
49  		assertPresenceOfFieldInMetamodelFor( Course.class, "qualifiedTeachers", "Type should be inferred from field" );
50  		assertPresenceOfFieldInMetamodelFor( Teacher.class, "qualifiedFor", "Type should be inferred from field" );
51  	}
52  
53  	@Test
54  	public void testMetaModelGeneratedForOneToManyPropertyAccessWithoutTargetEntity() {
55  		assertPresenceOfFieldInMetamodelFor( Car.class, "tires", "Type should be inferred from field" );
56  		assertPresenceOfFieldInMetamodelFor( Tire.class, "car", "Type should be inferred from field" );
57  	}
58  
59  	@Test
60  	public void testMetaModelGeneratedForEmbeddable() {
61  		assertPresenceOfFieldInMetamodelFor( Option.class, "period", "Embedded expected" );
62  		assertPresenceOfFieldInMetamodelFor( Period.class, "start", "Embedded expected" );
63  		assertPresenceOfFieldInMetamodelFor( Period.class, "end", "Embedded expected" );
64  	}
65  }
