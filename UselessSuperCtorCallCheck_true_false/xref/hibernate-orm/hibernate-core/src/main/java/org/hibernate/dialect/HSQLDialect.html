363 	private static final ViolatedConstraintNameExtracter EXTRACTER_18 = new TemplatedViolatedConstraintNameExtracter() {
364 		@Override
365 		public String extractConstraintName(SQLException sqle) {
366 			String constraintName = null;
367 
368 			final int errorCode = JdbcExceptionHelper.extractErrorCode( sqle );
369 
370 			if ( errorCode == -8 ) {
371 				constraintName = extractUsingTemplate(
372 						"Integrity constraint violation ", " table:", sqle.getMessage()
373 				);
374 			}
375 			else if ( errorCode == -9 ) {
376 				constraintName = extractUsingTemplate(
377 						"Violation of unique index: ", " in statement [", sqle.getMessage()
378 				);
379 			}
380 			else if ( errorCode == -104 ) {
381 				constraintName = extractUsingTemplate(
382 						"Unique constraint violation: ", " in statement [", sqle.getMessage()
383 				);
384 			}
385 			else if ( errorCode == -177 ) {
386 				constraintName = extractUsingTemplate(
387 						"Integrity constraint violation - no parent ", " table:",
388 						sqle.getMessage()
389 				);
390 			}
391 			return constraintName;
392 		}
393 
394 	};

477 	// temporary table support ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
478 
479 	// Hibernate uses this information for temporary tables that it uses for its own operations
480 	// therefore the appropriate strategy is taken with different versions of HSQLDB
481 
482 	// All versions of HSQLDB support GLOBAL TEMPORARY tables where the table
483 	// definition is shared by all users but data is private to the session
484 	// HSQLDB 2.0 also supports session-based LOCAL TEMPORARY tables where
485 	// the definition and data is private to the session and table declaration
486 	// can happen in the middle of a transaction
487 
488 	@Override
489 	public boolean supportsTemporaryTables() {
490 		return true;
491 	}

495 		if ( hsqldbVersion < 20 ) {

499 			// With HSQLDB 2.0, the table name is qualified with MODULE to assist the drop
500 			// statement (in-case there is a global name beginning with HT_)
501 			return "MODULE.HT_" + baseTableName;
502 		}
503 	}

577 	/**
578 	 * For HSQLDB 2.0, this is a copy of the base class implementation.
579 	 * For HSQLDB 1.8, only READ_UNCOMMITTED is supported.
580 	 * <p/>
581 	 * {@inheritDoc}
582 	 */
583 	@Override
584 	public LockingStrategy getLockingStrategy(Lockable lockable, LockMode lockMode) {
585 		if ( lockMode == LockMode.PESSIMISTIC_FORCE_INCREMENT ) {
586 			return new PessimisticForceIncrementLockingStrategy( lockable, lockMode );
587 		}
588 		else if ( lockMode == LockMode.PESSIMISTIC_WRITE ) {
589 			return new PessimisticWriteSelectLockingStrategy( lockable, lockMode );
590 		}
591 		else if ( lockMode == LockMode.PESSIMISTIC_READ ) {
592 			return new PessimisticReadSelectLockingStrategy( lockable, lockMode );
593 		}
594 		else if ( lockMode == LockMode.OPTIMISTIC ) {
595 			return new OptimisticLockingStrategy( lockable, lockMode );
596 		}
597 		else if ( lockMode == LockMode.OPTIMISTIC_FORCE_INCREMENT ) {
598 			return new OptimisticForceIncrementLockingStrategy( lockable, lockMode );
599 		}
600 
601 		if ( hsqldbVersion < 20 ) {
602 			return new ReadUncommittedLockingStrategy( lockable, lockMode );
603 		}
604 		else {
605 			return new SelectLockingStrategy( lockable, lockMode );
606 		}
607 	}
