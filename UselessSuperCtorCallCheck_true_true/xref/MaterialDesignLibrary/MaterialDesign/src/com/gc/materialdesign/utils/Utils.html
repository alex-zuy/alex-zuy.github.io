11  	/**
12  	 * Convert Dp to Pixel
13  	 * 将dp转换为pixel
14  	 */
15  	public static int dpToPx(float dp, Resources resources){
16  		float px = TypedValue.applyDimension(TypedValue.COMPLEX_UNIT_DIP, dp, resources.getDisplayMetrics());
17  		return (int) px;
18  	}
19  	
20  	/**
21  	 * @param value
22  	 * @return 将dip或者dp转为float
23  	 */
24  	public static float dipOrDpToFloat(String value) {
25  		if (value.indexOf("dp") != -1) {
26  			value = value.replace("dp", "");
27  		}
28  		else {
29  			value = value.replace("dip", "");
30  		}
31  		return Float.parseFloat(value);
32  	}
33  	
34  	
35  	/**
36  	 * 这里看似是得到控件相对的坐标，但是如果这个滑动条在可以上下滚动的布局中就会出现问题。
37  	 * 因为这里的坐标都是死的，在上下滚动的view中父控件的top仍旧不变，但实际上是应该获得动态数值的。
38  	 * @param myView
39  	 * @return
40  	 */
41  	public static int getRelativeTop(View myView) {
42  		Rect bounds = new Rect();
43  		myView.getGlobalVisibleRect(bounds);
44  		return bounds.top;
45  	}
46  	
47  	public static int getRelativeLeft(View myView) {
48  //	    if (myView.getParent() == myView.getRootView())
49  		if(myView.getId() == android.R.id.content)
50  			return myView.getLeft();
51  		else
52  			return myView.getLeft() + getRelativeLeft((View) myView.getParent());
53  	}
