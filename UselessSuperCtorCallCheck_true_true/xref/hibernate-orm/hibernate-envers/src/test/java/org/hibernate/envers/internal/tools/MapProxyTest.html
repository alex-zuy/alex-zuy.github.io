14      @Test
15      public void shouldGenerateClassWithAppropriateGetter() throws Exception {
16          //given
17          Map<String, Object> map = new HashMap<String, Object>();
18          int ageExpected = 14;
19          map.put("age", ageExpected);
20          Map<String, Class<?>> properties = new HashMap<String, Class<?>>();
21          properties.put("age", Integer.class);
22          //when
23          Class testClass = MapProxyTool.classForName("TestClass1", properties, new ClassLoaderServiceImpl());
24          Object testClassInstance = testClass.getConstructor(Map.class).newInstance(map);
25  
26          //then
27          Getter getter = ReflectionTools.getGetter(testClass, "age", "property");
28          int age = (Integer) getter.get(testClassInstance);
29          Assert.assertEquals(ageExpected, age);
30      }
