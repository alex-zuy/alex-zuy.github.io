152 	public IdentifierGenerator createIdentifierGenerator(
153 			IdentifierGeneratorFactory identifierGeneratorFactory,
154 			Dialect dialect, 
155 			String defaultCatalog, 
156 			String defaultSchema, 
157 			RootClass rootClass) throws MappingException {
158 		
159 		Properties params = new Properties();
160 		
161 		//if the hibernate-mapping did not specify a schema/catalog, use the defaults
162 		//specified by properties - but note that if the schema/catalog were specified
163 		//in hibernate-mapping, or as params, they will already be initialized and
164 		//will override the values set here (they are in identifierGeneratorProperties)
165 		if ( defaultSchema!=null ) {
166 			params.setProperty(PersistentIdentifierGenerator.SCHEMA, defaultSchema);
167 		}
168 		if ( defaultCatalog!=null ) {
169 			params.setProperty(PersistentIdentifierGenerator.CATALOG, defaultCatalog);
170 		}
171 		
172 		//pass the entity-name, if not a collection-id
173 		if (rootClass!=null) {
174 			params.setProperty( IdentifierGenerator.ENTITY_NAME, rootClass.getEntityName() );
175 			params.setProperty( IdentifierGenerator.JPA_ENTITY_NAME, rootClass.getJpaEntityName() );
176 		}
177 		
178 		//init the table here instead of earlier, so that we can get a quoted table name
179 		//TODO: would it be better to simply pass the qualified table name, instead of
180 		//      splitting it up into schema/catalog/table names
181 		String tableName = getTable().getQuotedName(dialect);
182 		params.setProperty( PersistentIdentifierGenerator.TABLE, tableName );
183 		
184 		//pass the column name (a generated id almost always has a single column)
185 		String columnName = ( (Column) getColumnIterator().next() ).getQuotedName(dialect);
186 		params.setProperty( PersistentIdentifierGenerator.PK, columnName );
187 		
188 		if (rootClass!=null) {
189 			StringBuilder tables = new StringBuilder();
190 			Iterator iter = rootClass.getIdentityTables().iterator();
191 			while ( iter.hasNext() ) {
192 				Table table= (Table) iter.next();
193 				tables.append( table.getQuotedName(dialect) );
194 				if ( iter.hasNext() ) tables.append(", ");
195 			}
196 			params.setProperty( PersistentIdentifierGenerator.TABLES, tables.toString() );
197 		}
198 		else {
199 			params.setProperty( PersistentIdentifierGenerator.TABLES, tableName );
200 		}
201 
202 		if (identifierGeneratorProperties!=null) {
203 			params.putAll(identifierGeneratorProperties);
204 		}
205 
206 		// TODO : we should pass along all settings once "config lifecycle" is hashed out...
207 		params.put(
208 				Environment.PREFER_POOLED_VALUES_LO,
209 				mappings.getConfigurationProperties().getProperty( Environment.PREFER_POOLED_VALUES_LO, "false" )
210 		);
211 
212 		identifierGeneratorFactory.setDialect( dialect );
213 		return identifierGeneratorFactory.createIdentifierGenerator( identifierGeneratorStrategy, getType(), params );
214 		
215 	}

292 	public boolean isNullable() {
293 		if ( hasFormula() ) return true;
294 		boolean nullable = true;
295 		Iterator iter = getColumnIterator();
296 		while ( iter.hasNext() ) {
297 			if ( !( (Column) iter.next() ).isNullable() ) {
298 				nullable = false;
299 				return nullable; //shortcut
300 			}
301 		}
302 		return nullable;
303 	}

313 	public Type getType() throws MappingException {
314 		if ( type != null ) {
315 			return type;
316 		}
317 
318 		if ( typeName == null ) {
319 			throw new MappingException( "No type name" );
320 		}
321 		if ( typeParameters != null
322 				&& Boolean.valueOf( typeParameters.getProperty( DynamicParameterizedType.IS_DYNAMIC ) )
323 				&& typeParameters.get( DynamicParameterizedType.PARAMETER_TYPE ) == null ) {
324 			createParameterImpl();
325 		}
326 
327 		Type result = mappings.getTypeResolver().heuristicType( typeName, typeParameters );
328 		if ( result == null ) {
329 			String msg = "Could not determine type for: " + typeName;
330 			if ( table != null ) {
331 				msg += ", at table: " + table.getName();
332 			}
333 			if ( columns != null && columns.size() > 0 ) {
334 				msg += ", for columns: " + columns;
335 			}
336 			throw new MappingException( msg );
337 		}
338 
339 		return result;
340 	}

342 	public void setTypeUsingReflection(String className, String propertyName) throws MappingException {
343 		// NOTE : this is called as the last piece in setting SimpleValue type information, and implementations
344 		// rely on that fact, using it as a signal that all information it is going to get is defined at this point...
345 
346 		if ( typeName != null ) {
347 			// assume either (a) explicit type was specified or (b) determine was already performed
348 			return;
349 		}
350 
351 		if ( type != null ) {
352 			return;
353 		}
354 
355 		if ( attributeConverterDefinition == null ) {
356 			// this is here to work like legacy.  This should change when we integrate with metamodel to
357 			// look for SqlTypeDescriptor and JavaTypeDescriptor individually and create the BasicType (well, really
358 			// keep a registry of [SqlTypeDescriptor,JavaTypeDescriptor] -> BasicType...)
359 			if ( className == null ) {
360 				throw new MappingException( "you must specify types for a dynamic entity: " + propertyName );
361 			}
362 			typeName = ReflectHelper.reflectedPropertyClass( className, propertyName ).getName();
363 			return;
364 		}
365 
366 		// we had an AttributeConverter...
367 		type = buildAttributeConverterTypeAdapter();
368 	}

370 	/**
371 	 * Build a Hibernate Type that incorporates the JPA AttributeConverter.  AttributeConverter works totally in
372 	 * memory, meaning it converts between one Java representation (the entity attribute representation) and another
373 	 * (the value bound into JDBC statements or extracted from results).  However, the Hibernate Type system operates
374 	 * at the lower level of actually dealing directly with those JDBC objects.  So even though we have an
375 	 * AttributeConverter, we still need to "fill out" the rest of the BasicType data and bridge calls
376 	 * to bind/extract through the converter.
377 	 * <p/>
378 	 * Essentially the idea here is that an intermediate Java type needs to be used.  Let's use an example as a means
379 	 * to illustrate...  Consider an {@code AttributeConverter<Integer,String>}.  This tells Hibernate that the domain
380 	 * model defines this attribute as an Integer value (the 'entityAttributeJavaType'), but that we need to treat the
381 	 * value as a String (the 'databaseColumnJavaType') when dealing with JDBC (aka, the database type is a
382 	 * VARCHAR/CHAR):<ul>
383 	 *     <li>
384 	 *         When binding values to PreparedStatements we need to convert the Integer value from the entity
385 	 *         into a String and pass that String to setString.  The conversion is handled by calling
386 	 *         {@link AttributeConverter#convertToDatabaseColumn(Object)}
387 	 *     </li>
388 	 *     <li>
389 	 *         When extracting values from ResultSets (or CallableStatement parameters) we need to handle the
390 	 *         value via getString, and convert that returned String to an Integer.  That conversion is handled
391 	 *         by calling {@link AttributeConverter#convertToEntityAttribute(Object)}
392 	 *     </li>
393 	 * </ul>
394 	 *
395 	 * @return The built AttributeConverter -> Type adapter
396 	 *
397 	 * @todo : ultimately I want to see attributeConverterJavaType and attributeConverterJdbcTypeCode specify-able separately
398 	 * then we can "play them against each other" in terms of determining proper typing
399 	 *
400 	 * @todo : see if we already have previously built a custom on-the-fly BasicType for this AttributeConverter; see note below about caching
401 	 */
402 	@SuppressWarnings("unchecked")
403 	private Type buildAttributeConverterTypeAdapter() {
404 		// todo : validate the number of columns present here?
405 
406 		final Class entityAttributeJavaType = attributeConverterDefinition.getEntityAttributeType();
407 		final Class databaseColumnJavaType = attributeConverterDefinition.getDatabaseColumnType();
408 
409 
410 		// resolve the JavaTypeDescriptor ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
411 		// For the JavaTypeDescriptor portion we simply resolve the "entity attribute representation" part of
412 		// the AttributeConverter to resolve the corresponding descriptor.
413 		final JavaTypeDescriptor entityAttributeJavaTypeDescriptor = JavaTypeDescriptorRegistry.INSTANCE.getDescriptor( entityAttributeJavaType );
414 
415 
416 		// build the SqlTypeDescriptor adapter ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
417 		// Going back to the illustration, this should be a SqlTypeDescriptor that handles the Integer <-> String
418 		//		conversions.  This is the more complicated piece.  First we need to determine the JDBC type code
419 		//		corresponding to the AttributeConverter's declared "databaseColumnJavaType" (how we read that value out
420 		// 		of ResultSets).  See JdbcTypeJavaClassMappings for details.  Again, given example, this should return
421 		// 		VARCHAR/CHAR
422 		final int jdbcTypeCode = JdbcTypeJavaClassMappings.INSTANCE.determineJdbcTypeCodeForJavaClass( databaseColumnJavaType );
423 		// find the standard SqlTypeDescriptor for that JDBC type code.
424 		final SqlTypeDescriptor sqlTypeDescriptor = SqlTypeDescriptorRegistry.INSTANCE.getDescriptor( jdbcTypeCode );
425 		// find the JavaTypeDescriptor representing the "intermediate database type representation".  Back to the
426 		// 		illustration, this should be the type descriptor for Strings
427 		final JavaTypeDescriptor intermediateJavaTypeDescriptor = JavaTypeDescriptorRegistry.INSTANCE.getDescriptor( databaseColumnJavaType );
428 		// and finally construct the adapter, which injects the AttributeConverter calls into the binding/extraction
429 		// 		process...
430 		final SqlTypeDescriptor sqlTypeDescriptorAdapter = new AttributeConverterSqlTypeDescriptorAdapter(
431 				attributeConverterDefinition.getAttributeConverter(),
432 				sqlTypeDescriptor,
433 				intermediateJavaTypeDescriptor
434 		);
435 
436 		// todo : cache the AttributeConverterTypeAdapter in case that AttributeConverter is applied multiple times.
437 
438 		final String name = String.format(
439 				"BasicType adapter for AttributeConverter<%s,%s>",
440 				entityAttributeJavaType.getSimpleName(),
441 				databaseColumnJavaType.getSimpleName()
442 		);
443 		return new AttributeConverterTypeAdapter(
444 				name,
445 				attributeConverterDefinition.getAttributeConverter(),
446 				sqlTypeDescriptorAdapter,
447 				entityAttributeJavaType,
448 				databaseColumnJavaType,
449 				entityAttributeJavaTypeDescriptor
450 		);
451 	}

474 	public boolean[] getColumnInsertability() {
475 		boolean[] result = new boolean[ getColumnSpan() ];
476 		int i = 0;
477 		Iterator iter = getColumnIterator();
478 		while ( iter.hasNext() ) {
479 			Selectable s = (Selectable) iter.next();
480 			result[i++] = !s.isFormula();
481 		}
482 		return result;
483 	}

493 	private void createParameterImpl() {
494 		try {
495 			String[] columnsNames = new String[columns.size()];
496 			for ( int i = 0; i < columns.size(); i++ ) {
497 				Selectable column = columns.get(i);
498 				if (column instanceof Column){
499 					columnsNames[i] = ((Column) column).getName();
500 				}
501 			}
502 
503 			final XProperty xProperty = (XProperty) typeParameters.get( DynamicParameterizedType.XPROPERTY );
504 			// todo : not sure this works for handling @MapKeyEnumerated
505 			final Annotation[] annotations = xProperty == null
506 					? null
507 					: xProperty.getAnnotations();
508 
509 			typeParameters.put(
510 					DynamicParameterizedType.PARAMETER_TYPE,
511 					new ParameterTypeImpl(
512 							ReflectHelper.classForName(
513 									typeParameters.getProperty( DynamicParameterizedType.RETURNED_CLASS )
514 							),
515 							annotations,
516 							table.getCatalog(),
517 							table.getSchema(),
518 							table.getName(),
519 							Boolean.valueOf( typeParameters.getProperty( DynamicParameterizedType.IS_PRIMARY_KEY ) ),
520 							columnsNames
521 					)
522 			);
523 		}
524 		catch ( ClassNotFoundException cnfe ) {
525 			throw new MappingException( "Could not create DynamicParameterizedType for type: " + typeName, cnfe );
526 		}
527 	}
