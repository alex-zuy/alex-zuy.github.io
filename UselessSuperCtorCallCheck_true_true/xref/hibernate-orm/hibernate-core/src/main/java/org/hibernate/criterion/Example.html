40  /**
41   * Support for query by example.
42   *
43   * <pre>
44   * List results = session.createCriteria(Parent.class)
45   *     .add( Example.create(parent).ignoreCase() )
46   *     .createCriteria("child")
47   *         .add( Example.create( parent.getChild() ) )
48   *     .list();
49   * </pre>
50   *
51   * "Examples" may be mixed and matched with "Expressions" in the same Criteria.
52   *
53   * @see org.hibernate.Criteria
54   * @author Gavin King
55   */
56  
57  public class Example implements Criterion {
58  	private final Object exampleEntity;
59  	private PropertySelector selector;
60  
61  	private boolean isLikeEnabled;
62  	private Character escapeCharacter;
63  	private boolean isIgnoreCaseEnabled;
64  	private MatchMode matchMode;
65  
66  	private final Set<String> excludedProperties = new HashSet<String>();
67  
68  	/**
69  	 * Create a new Example criterion instance, which includes all non-null properties by default
70  	 *
71  	 * @param exampleEntity The example bean to use.
72  	 *
73  	 * @return a new instance of Example
74  	 */
75  	public static Example create(Object exampleEntity) {
76  		if ( exampleEntity == null ) {
77  			throw new NullPointerException( "null example entity" );
78  		}
79  		return new Example( exampleEntity, NotNullPropertySelector.INSTANCE );
80  	}

89  		this.exampleEntity = exampleEntity;

197 	@Override
198 	public String toSqlString(Criteria criteria, CriteriaQuery criteriaQuery) {
199 		final StringBuilder buf = new StringBuilder().append( '(' );
200 		final EntityPersister meta = criteriaQuery.getFactory().getEntityPersister(
201 				criteriaQuery.getEntityName( criteria )
202 		);
203 		final String[] propertyNames = meta.getPropertyNames();
204 		final Type[] propertyTypes = meta.getPropertyTypes();
205 
206 		final Object[] propertyValues = meta.getPropertyValues( exampleEntity );
207 		for ( int i=0; i<propertyNames.length; i++ ) {
208 			final Object propertyValue = propertyValues[i];
209 			final String propertyName = propertyNames[i];
210 
211 			final boolean isVersionProperty = i == meta.getVersionProperty();
212 			if ( ! isVersionProperty && isPropertyIncluded( propertyValue, propertyName, propertyTypes[i] ) ) {
213 				if ( propertyTypes[i].isComponentType() ) {
214 					appendComponentCondition(
215 						propertyName,
216 						propertyValue,
217 						(CompositeType) propertyTypes[i],
218 						criteria,
219 						criteriaQuery,
220 						buf
221 					);
222 				}
223 				else {
224 					appendPropertyCondition(
225 						propertyName,
226 						propertyValue,
227 						criteria,
228 						criteriaQuery,
229 						buf
230 					);
231 				}
232 			}
233 		}
234 
235 		if ( buf.length()==1 ) {
236 			buf.append( "1=1" );
237 		}
238 
239 		return buf.append( ')' ).toString();
240 	}

242 	@SuppressWarnings("SimplifiableIfStatement")
243 	private boolean isPropertyIncluded(Object value, String name, Type type) {
244 		if ( excludedProperties.contains( name ) ) {
245 			// was explicitly excluded
246 			return false;
247 		}
248 
249 		if ( type.isAssociationType() ) {
250 			// associations are implicitly excluded
251 			return false;
252 		}
253 
254 		return selector.include( value, name, type );
255 	}

257 	@Override
258 	public TypedValue[] getTypedValues(Criteria criteria, CriteriaQuery criteriaQuery) {
259 		final EntityPersister meta = criteriaQuery.getFactory().getEntityPersister(
260 				criteriaQuery.getEntityName( criteria )
261 		);
262 		final String[] propertyNames = meta.getPropertyNames();
263 		final Type[] propertyTypes = meta.getPropertyTypes();
264 
265 		final Object[] values = meta.getPropertyValues( exampleEntity );
266 		final List<TypedValue> list = new ArrayList<TypedValue>();
267 		for ( int i=0; i<propertyNames.length; i++ ) {
268 			final Object value = values[i];
269 			final Type type = propertyTypes[i];
270 			final String name = propertyNames[i];
271 
272 			final boolean isVersionProperty = i == meta.getVersionProperty();
273 
274 			if ( ! isVersionProperty && isPropertyIncluded( value, name, type ) ) {
275 				if ( propertyTypes[i].isComponentType() ) {
276 					addComponentTypedValues( name, value, (CompositeType) type, list, criteria, criteriaQuery );
277 				}
278 				else {
279 					addPropertyTypedValue( value, type, list );
280 				}
281 			}
282 		}
283 
284 		return list.toArray( new TypedValue[ list.size() ] );
285 	}
286 
287 	protected void addPropertyTypedValue(Object value, Type type, List<TypedValue> list) {
288 		if ( value != null ) {
289 			if ( value instanceof String ) {
290 				String string = (String) value;
291 				if ( isIgnoreCaseEnabled ) {
292 					string = string.toLowerCase();
293 				}
294 				if ( isLikeEnabled ) {
295 					string = matchMode.toMatchString( string );
296 				}
297 				value = string;
298 			}
299 			list.add( new TypedValue( type, value ) );
300 		}
301 	}
302 
303 	protected void addComponentTypedValues(
304 			String path, 
305 			Object component, 
306 			CompositeType type,
307 			List<TypedValue> list,
308 			Criteria criteria, 
309 			CriteriaQuery criteriaQuery) {
310 		if ( component != null ) {
311 			final String[] propertyNames = type.getPropertyNames();
312 			final Type[] subtypes = type.getSubtypes();
313 			final Object[] values = type.getPropertyValues( component, getEntityMode( criteria, criteriaQuery ) );
314 			for ( int i=0; i<propertyNames.length; i++ ) {
315 				final Object value = values[i];
316 				final Type subtype = subtypes[i];
317 				final String subpath = StringHelper.qualify( path, propertyNames[i] );
318 				if ( isPropertyIncluded( value, subpath, subtype ) ) {
319 					if ( subtype.isComponentType() ) {
320 						addComponentTypedValues( subpath, value, (CompositeType) subtype, list, criteria, criteriaQuery );
321 					}
322 					else {
323 						addPropertyTypedValue( value, subtype, list );
324 					}
325 				}
326 			}
327 		}
328 	}

330 	private EntityMode getEntityMode(Criteria criteria, CriteriaQuery criteriaQuery) {
331 		final EntityPersister meta = criteriaQuery.getFactory().getEntityPersister(
332 				criteriaQuery.getEntityName( criteria )
333 		);
334 		final EntityMode result = meta.getEntityMode();
335 		if ( ! meta.getEntityMetamodel().getTuplizer().isInstance( exampleEntity ) ) {
336 			throw new ClassCastException( exampleEntity.getClass().getName() );
337 		}
338 		return result;
339 	}

341 	protected void appendPropertyCondition(
342 			String propertyName,
343 			Object propertyValue,
344 			Criteria criteria,
345 			CriteriaQuery cq,
346 			StringBuilder buf) {
347 		final Criterion condition;
348 		if ( propertyValue != null ) {
349 			final boolean isString = propertyValue instanceof String;
350 			if ( isLikeEnabled && isString ) {
351 				condition = new LikeExpression(
352 						propertyName,
353 						(String) propertyValue,
354 						matchMode,
355 						escapeCharacter,
356 						isIgnoreCaseEnabled
357 				);
358 			}
359 			else {
360 				condition = new SimpleExpression( propertyName, propertyValue, "=", isIgnoreCaseEnabled && isString );
361 			}
362 		}
363 		else {
364 			condition = new NullExpression(propertyName);
365 		}
366 
367 		final String conditionFragment = condition.toSqlString( criteria, cq );
368 		if ( conditionFragment.trim().length() > 0 ) {
369 			if ( buf.length() > 1 ) {
370 				buf.append( " and " );
371 			}
372 			buf.append( conditionFragment );
373 		}
374 	}
375 
376 	protected void appendComponentCondition(
377 			String path,
378 			Object component,
379 			CompositeType type,
380 			Criteria criteria,
381 			CriteriaQuery criteriaQuery,
382 			StringBuilder buf) {
383 		if ( component != null ) {
384 			final String[] propertyNames = type.getPropertyNames();
385 			final Object[] values = type.getPropertyValues( component, getEntityMode( criteria, criteriaQuery ) );
386 			final Type[] subtypes = type.getSubtypes();
387 			for ( int i=0; i<propertyNames.length; i++ ) {
388 				final String subPath = StringHelper.qualify( path, propertyNames[i] );
389 				final Object value = values[i];
390 				if ( isPropertyIncluded( value, subPath, subtypes[i] ) ) {
391 					final Type subtype = subtypes[i];
392 					if ( subtype.isComponentType() ) {
393 						appendComponentCondition(
394 								subPath,
395 								value,
396 								(CompositeType) subtype,
397 								criteria,
398 								criteriaQuery,
399 								buf
400 						);
401 					}
402 					else {
403 						appendPropertyCondition(
404 								subPath,
405 								value,
406 								criteria,
407 								criteriaQuery,
408 								buf
409 						);
410 					}
411 				}
412 			}
413 		}
414 	}

424 	/**
425 	 * A strategy for choosing property values for inclusion in the query criteria.  Note that
426 	 * property selection (for inclusion) operates separately from excluding a property.  Excluded
427 	 * properties are not even passed in to the PropertySelector for consideration.
428 	 */
429 	public static interface PropertySelector extends Serializable {
430 		/**
431 		 * Determine whether the given property should be used in the criteria.
432 		 *
433 		 * @param propertyValue The property value (from the example bean)
434 		 * @param propertyName The name of the property
435 		 * @param type The type of the property
436 		 *
437 		 * @return {@code true} indicates the property should be included; {@code false} indiates it should not.
438 		 */
439 		public boolean include(Object propertyValue, String propertyName, Type type);
440 	}
441 
442 	/**
443 	 * Property selector that includes all properties
444 	 */
445 	public static final class AllPropertySelector implements PropertySelector {
446 		/**
447 		 * Singleton access
448 		 */
449 		public static final AllPropertySelector INSTANCE = new AllPropertySelector();
450 
451 		@Override
452 		public boolean include(Object object, String propertyName, Type type) {
453 			return true;
454 		}
455 
456 		private Object readResolve() {
457 			return INSTANCE;
458 		}
459 	}
460 
461 	/**
462 	 * Property selector that includes only properties that are not {@code null}
463 	 */
464 	public static final class NotNullPropertySelector implements PropertySelector {
465 		/**
466 		 * Singleton access
467 		 */
468 		public static final NotNullPropertySelector INSTANCE = new NotNullPropertySelector();
469 
470 		@Override
471 		public boolean include(Object object, String propertyName, Type type) {
472 			return object!=null;
473 		}
474 
475 		private Object readResolve() {
476 			return INSTANCE;
477 		}
478 	}
479 
480 	/**
481 	 * Property selector that includes only properties that are not {@code null} and non-zero (if numeric)
482 	 */
483 	public static final class NotNullOrZeroPropertySelector implements PropertySelector {
484 		/**
485 		 * Singleton access
486 		 */
487 		public static final NotNullOrZeroPropertySelector INSTANCE = new NotNullOrZeroPropertySelector();
488 
489 		@Override
490 		public boolean include(Object object, String propertyName, Type type) {
491 			return object != null
492 					&& ( !(object instanceof Number) || ( (Number) object ).longValue()!=0
493 			);
494 		}
495 
496 		private Object readResolve() {
497 			return INSTANCE;
498 		}
