61  /**
62   * Abstract strategy of building loadplan based on entity graph.
63   *
64   * The problem we're resolving here is, we have TWO trees to walk (only entity loading here):
65   * <ul>
66   * <ol>entity metadata and its associations</ol>
67   * <ol>entity graph and attribute nodes</ol>
68   * </ul>
69   *
70   * And most time, the entity graph tree is partial of entity metadata tree.
71   *
72   * So, the idea here is, we walk the entity metadata tree, just as how we build the static loadplan from mappings,
73   * and we try to match the node to entity graph ( and subgraph ), if there is a match, then the attribute is fetched,
74   * it is not, then depends on which property is used to apply this entity graph.
75   *
76   * @author Strong Liu <stliu@hibernate.org>
77   * @author Brett Meyer
78   */
79  public abstract class AbstractEntityGraphVisitationStrategy
80  		extends AbstractLoadPlanBuildingAssociationVisitationStrategy {
81  	private static final Logger LOG = CoreLogging.logger( AbstractEntityGraphVisitationStrategy.class );
82  	/**
83  	 * The JPA 2.1 SPEC's Entity Graph only defines _WHEN_ to load an attribute, it doesn't define _HOW_ to load it
84  	 * So I'm here just making an assumption that when it is EAGER, then we use JOIN, and when it is LAZY, then we use SELECT.
85  	 *
86  	 * NOTE: this may be changed in the near further, though ATM I have no idea how this will be changed to :)
87  	 * -- stliu
88  	 */
89  	protected static final FetchStrategy DEFAULT_EAGER = new FetchStrategy( FetchTiming.IMMEDIATE, FetchStyle.JOIN );
90  	protected static final FetchStrategy DEFAULT_LAZY = new FetchStrategy( FetchTiming.DELAYED, FetchStyle.SELECT );
91  	protected final LoadQueryInfluencers loadQueryInfluencers;
92  	// Queue containing entity/sub graphs to be visited.
93  	private final ArrayDeque<GraphNodeImplementor> graphStack = new ArrayDeque<GraphNodeImplementor>();
94  	// Queue containing attributes being visited, used eventually to determine the fetch strategy.
95  	private final ArrayDeque<AttributeNodeImplementor> attributeStack = new ArrayDeque<AttributeNodeImplementor>();
96  	// Queue of maps containing the current graph node's attributes.  Used for fast lookup, instead of iterating
97  	// over graphStack.peekLast().attributeImplementorNodes().
98  	private final ArrayDeque<Map<String, AttributeNodeImplementor>> attributeMapStack
99  			= new ArrayDeque<Map<String, AttributeNodeImplementor>>();
100 	private EntityReturn rootEntityReturn;
101 	private final LockMode lockMode;
102 
103 	protected AbstractEntityGraphVisitationStrategy(
104 			final SessionFactoryImplementor sessionFactory, final LoadQueryInfluencers loadQueryInfluencers,
105 			final LockMode lockMode) {
106 		super( sessionFactory );
107 		this.loadQueryInfluencers = loadQueryInfluencers;
108 		this.lockMode = lockMode;
109 	}

111 	@Override
112 	public void start() {
113 		super.start();
114 		graphStack.addLast( getRootEntityGraph() );
115 	}
116 
117 	@Override
118 	public void finish() {
119 		super.finish();
120 		graphStack.removeLast();
121 		//applying a little internal stack checking
122 		if ( !graphStack.isEmpty() || !attributeStack.isEmpty() || !attributeMapStack.isEmpty() ) {
123 			throw new WalkingException( "Internal stack error" );
124 		}
125 	}
126 
127 	@Override
128 	public void startingEntity(final EntityDefinition entityDefinition) {
129 		attributeMapStack.addLast( buildAttributeNodeMap() );
130 		super.startingEntity( entityDefinition );
131 	}
132 
133 	/**
134 	 * Build "name" -- "attribute node" map from the current entity graph we're visiting.
135 	 */
136 	protected Map<String, AttributeNodeImplementor> buildAttributeNodeMap() {
137 		GraphNodeImplementor graphNode = graphStack.peekLast();
138 		List<AttributeNodeImplementor<?>> attributeNodeImplementors = graphNode.attributeImplementorNodes();
139 		Map<String, AttributeNodeImplementor> attributeNodeImplementorMap = attributeNodeImplementors.isEmpty() ? Collections
140 				.<String, AttributeNodeImplementor>emptyMap() : new HashMap<String, AttributeNodeImplementor>(
141 				attributeNodeImplementors.size()
142 		);
143 		for ( AttributeNodeImplementor attribute : attributeNodeImplementors ) {
144 			attributeNodeImplementorMap.put( attribute.getAttributeName(), attribute );
145 		}
146 		return attributeNodeImplementorMap;
147 	}
148 
149 	@Override
150 	public void finishingEntity(final EntityDefinition entityDefinition) {
151 		attributeMapStack.removeLast();
152 		super.finishingEntity( entityDefinition );
153 	}
154 
155 	/**
156 	 * I'm using NULL-OBJECT pattern here, for attributes that not existing in the EntityGraph,
157 	 * a predefined NULL-ATTRIBUTE-NODE is pushed to the stack.
158 	 *
159 	 * and for an not existing sub graph, a predefined NULL-SUBGRAPH is pushed to the stack.
160 	 *
161 	 * So, whenever we're start visiting an attribute, there will be a attribute node pushed to the attribute stack,
162 	 * and a subgraph node pushed to the graph stack.
163 	 *
164 	 * when we're finish visiting an attribute, these two will be poped from each stack.
165 	 */
166 	@Override
167 	public boolean startingAttribute(AttributeDefinition attributeDefinition) {
168 		Map<String, AttributeNodeImplementor> attributeMap = attributeMapStack.peekLast();
169 		final String attrName = attributeDefinition.getName();
170 		AttributeNodeImplementor attributeNode = NON_EXIST_ATTRIBUTE_NODE;
171 		GraphNodeImplementor subGraphNode = NON_EXIST_SUBGRAPH_NODE;
172 		//the attribute is in the EntityGraph, so, let's continue
173 		if ( attributeMap.containsKey( attrName ) ) {
174 			attributeNode = attributeMap.get( attrName );
175 			//here we need to check if there is a subgraph (or sub key graph if it is an indexed attribute )
176 			Map<Class, Subgraph> subGraphs = attributeNode.getSubgraphs();
177 			Class javaType = attributeDefinition.getType().getReturnedClass();
178 			if ( !subGraphs.isEmpty() && subGraphs.containsKey( javaType ) ) {
179 				subGraphNode = (GraphNodeImplementor) subGraphs.get( javaType );
180 			}
181 
182 		}
183 		attributeStack.addLast( attributeNode );
184 		graphStack.addLast( subGraphNode );
185 		return super.startingAttribute( attributeDefinition );
186 	}
187 
188 
189 	@Override
190 	public void finishingAttribute(final AttributeDefinition attributeDefinition) {
191 		attributeStack.removeLast();
192 		graphStack.removeLast();
193 		super.finishingAttribute( attributeDefinition );
194 	}

197 	@Override
198 	public void startingCollectionElements(
199 			final CollectionElementDefinition elementDefinition) {
200 		AttributeNodeImplementor attributeNode = attributeStack.peekLast();
201 		GraphNodeImplementor subGraphNode = NON_EXIST_SUBGRAPH_NODE;
202 		Map<Class, Subgraph> subGraphs = attributeNode.getSubgraphs();
203 		Class javaType = elementDefinition.getType().getReturnedClass();
204 		if ( !subGraphs.isEmpty() && subGraphs.containsKey( javaType ) ) {
205 			subGraphNode = (GraphNodeImplementor) subGraphs.get( javaType );
206 		}
207 		graphStack.addLast( subGraphNode );
208 		super.startingCollectionElements( elementDefinition );
209 	}
210 
211 	@Override
212 	public void finishingCollectionElements(
213 			final CollectionElementDefinition elementDefinition) {
214 		super.finishingCollectionElements( elementDefinition );
215 		graphStack.removeLast();
216 	}

219 	@Override
220 	public void startingCollectionIndex(final CollectionIndexDefinition indexDefinition) {
221 		AttributeNodeImplementor attributeNode = attributeStack.peekLast();
222 		GraphNodeImplementor subGraphNode = NON_EXIST_SUBGRAPH_NODE;
223 		Map<Class, Subgraph> subGraphs = attributeNode.getKeySubgraphs();
224 		Class javaType = indexDefinition.getType().getReturnedClass();
225 		if ( !subGraphs.isEmpty() && subGraphs.containsKey( javaType ) ) {
226 			subGraphNode = (GraphNodeImplementor) subGraphs.get( javaType );
227 		}
228 		graphStack.addLast( subGraphNode );
229 		super.startingCollectionIndex( indexDefinition );
230 	}
231 
232 	@Override
233 	public void finishingCollectionIndex(final CollectionIndexDefinition indexDefinition) {
234 		super.finishingCollectionIndex( indexDefinition );
235 		graphStack.removeLast();
236 	}

245 	@Override
246 	protected void addRootReturn(final Return rootReturn) {
247 		if ( this.rootEntityReturn != null ) {
248 			throw new HibernateException( "Root return already identified" );
249 		}
250 		if ( !( rootReturn instanceof EntityReturn ) ) {
251 			throw new HibernateException( "Load entity graph only supports EntityReturn" );
252 		}
253 		this.rootEntityReturn = (EntityReturn) rootReturn;
254 	}
255 
256 	@Override
257 	protected FetchStrategy determineFetchStrategy(
258 			final AssociationAttributeDefinition attributeDefinition) {
259 		return attributeStack.peekLast() != NON_EXIST_ATTRIBUTE_NODE ? DEFAULT_EAGER : resolveImplicitFetchStrategyFromEntityGraph(
260 				attributeDefinition
261 		);
262 	}
263 
264 	protected abstract FetchStrategy resolveImplicitFetchStrategyFromEntityGraph(
265 			final AssociationAttributeDefinition attributeDefinition);
266 
267 	protected FetchStrategy adjustJoinFetchIfNeeded(
268 			AssociationAttributeDefinition attributeDefinition, FetchStrategy fetchStrategy) {
269 		if ( lockMode.greaterThan( LockMode.READ ) ) {
270 			return new FetchStrategy( fetchStrategy.getTiming(), FetchStyle.SELECT );
271 		}
272 
273 		final Integer maxFetchDepth = sessionFactory().getSettings().getMaximumFetchDepth();
274 		if ( maxFetchDepth != null && currentDepth() > maxFetchDepth ) {
275 			return new FetchStrategy( fetchStrategy.getTiming(), FetchStyle.SELECT );
276 		}
277 
278 		if ( attributeDefinition.getType().isCollectionType() && isTooManyCollections() ) {
279 			// todo : have this revert to batch or subselect fetching once "sql gen redesign" is in place
280 			return new FetchStrategy( fetchStrategy.getTiming(), FetchStyle.SELECT );
281 		}
282 
283 		return fetchStrategy;
284 	}

289 		LOG.debug( "Building LoadPlan..." );
290 		return new LoadPlanImpl( rootEntityReturn, getQuerySpaces() );

343 	@Override
344 	public void foundCircularAssociation(AssociationAttributeDefinition attributeDefinition) {
345 		final FetchStrategy fetchStrategy = determineFetchStrategy( attributeDefinition );
346 		if ( fetchStrategy.getStyle() != FetchStyle.JOIN ) {
347 			return; // nothing to do
348 		}
349 
350 		// Bi-directional association & the owning side was already visited.  If the current attribute node refers
351 		// to it, fetch.
352 		// ENTITY nature handled by super.
353 		final GraphNodeImplementor graphNode = graphStack.peekLast();
354 		if ( attributeDefinition.getAssociationNature() == AssociationAttributeDefinition.AssociationNature.COLLECTION
355 				&& ! graphNode.equals( NON_EXIST_SUBGRAPH_NODE)
356 				&& graphNode.containsAttribute( attributeDefinition.getName() )) {
357 			currentSource().buildCollectionAttributeFetch( attributeDefinition, fetchStrategy );
358 		}
359 		
360 		super.foundCircularAssociation( attributeDefinition );
361 	}
