88  	@Override
89  	@SuppressWarnings( {"unchecked"})
90  	public Serializable getSnapshot(CollectionPersister persister) throws HibernateException {
91  		final HashMap clonedMap = new HashMap( map.size() );
92  		for ( Object o : map.entrySet() ) {
93  			final Entry e = (Entry) o;
94  			final Object copy = persister.getElementType().deepCopy( e.getValue(), persister.getFactory() );
95  			clonedMap.put( e.getKey(), copy );
96  		}
97  		return clonedMap;
98  	}

106 	@Override
107 	public boolean equalsSnapshot(CollectionPersister persister) throws HibernateException {
108 		final Type elementType = persister.getElementType();
109 		final Map snapshotMap = (Map) getSnapshot();
110 		if ( snapshotMap.size() != this.map.size() ) {
111 			return false;
112 		}
113 
114 		for ( Object o : map.entrySet() ) {
115 			final Entry entry = (Entry) o;
116 			if ( elementType.isDirty( entry.getValue(), snapshotMap.get( entry.getKey() ), getSession() ) ) {
117 				return false;
118 			}
119 		}
120 		return true;
121 	}

170 	@Override
171 	@SuppressWarnings("unchecked")
172 	public Object put(Object key, Object value) {
173 		if ( isPutQueueEnabled() ) {
174 			final Object old = readElementByIndex( key );
175 			if ( old != UNKNOWN ) {
176 				queueOperation( new Put( key, value, old ) );
177 				return old;
178 			}
179 		}
180 		initialize( true );
181 		final Object old = map.put( key, value );
182 		// would be better to use the element-type to determine
183 		// whether the old and the new are equal here; the problem being
184 		// we do not necessarily have access to the element type in all
185 		// cases
186 		if ( value != old ) {
187 			dirty();
188 		}
189 		return old;
190 	}
191 
192 	@Override
193 	@SuppressWarnings("unchecked")
194 	public Object remove(Object key) {
195 		if ( isPutQueueEnabled() ) {
196 			final Object old = readElementByIndex( key );
197 			if ( old != UNKNOWN ) {
198 				queueOperation( new Remove( key, old ) );
199 				return old;
200 			}
201 		}
202 		// TODO : safe to interpret "map.remove(key) == null" as non-dirty?
203 		initialize( true );
204 		if ( map.containsKey( key ) ) {
205 			dirty();
206 		}
207 		return map.remove( key );
208 	}

210 	@Override
211 	@SuppressWarnings("unchecked")
212 	public void putAll(Map puts) {
213 		if ( puts.size() > 0 ) {
214 			initialize( true );
215 			for ( Object o : puts.entrySet() ) {
216 				final Entry entry = (Entry) o;
217 				put( entry.getKey(), entry.getValue() );
218 			}
219 		}
220 	}

222 	@Override
223 	@SuppressWarnings("unchecked")
224 	public void clear() {
225 		if ( isClearQueueEnabled() ) {
226 			queueOperation( new Clear() );
227 		}
228 		else {
229 			initialize( true );
230 			if ( ! map.isEmpty() ) {
231 				dirty();
232 				map.clear();
233 			}
234 		}
235 	}

271 	private transient List<Object[]> loadingEntries;
272 
273 	@Override
274 	@SuppressWarnings("unchecked")
275 	public Object readFrom(
276 			ResultSet rs,
277 			CollectionPersister persister,
278 			CollectionAliases descriptor,
279 			Object owner) throws HibernateException, SQLException {
280 		final Object element = persister.readElement( rs, owner, descriptor.getSuffixedElementAliases(), getSession() );
281 		if ( element != null ) {
282 			final Object index = persister.readIndex( rs, descriptor.getSuffixedIndexAliases(), getSession() );
283 			if ( loadingEntries == null ) {
284 				loadingEntries = new ArrayList<Object[]>();
285 			}
286 			loadingEntries.add( new Object[] { index, element } );
287 		}
288 		return element;
289 	}
290 
291 	@Override
292 	@SuppressWarnings("unchecked")
293 	public boolean endRead() {
294 		if ( loadingEntries != null ) {
295 			for ( Object[] entry : loadingEntries ) {
296 				map.put( entry[0], entry[1] );
297 			}
298 			loadingEntries = null;
299 		}
300 		return super.endRead();
301 	}

471 	@Override
472 	@SuppressWarnings("unchecked")
473 	public void initializeFromCache(CollectionPersister persister, Serializable disassembled, Object owner)
474 			throws HibernateException {
475 		final Serializable[] array = (Serializable[]) disassembled;
476 		final int size = array.length;
477 		beforeInitialize( persister, size );
478 		for ( int i = 0; i < size; i+=2 ) {
479 			map.put(
480 					persister.getIndexType().assemble( array[i], getSession(), owner ),
481 					persister.getElementType().assemble( array[i+1], getSession(), owner )
482 			);
483 		}
484 	}
485 
486 	@Override
487 	@SuppressWarnings("unchecked")
488 	public Serializable disassemble(CollectionPersister persister) throws HibernateException {
489 		final Serializable[] result = new Serializable[ map.size() * 2 ];
490 		final Iterator itr = map.entrySet().iterator();
491 		int i=0;
492 		while ( itr.hasNext() ) {
493 			final Map.Entry e = (Map.Entry) itr.next();
494 			result[i++] = persister.getIndexType().disassemble( e.getKey(), getSession(), null );
495 			result[i++] = persister.getElementType().disassemble( e.getValue(), getSession(), null );
496 		}
497 		return result;
498 
499 	}

501 	@Override
502 	@SuppressWarnings("unchecked")
503 	public Iterator getDeletes(CollectionPersister persister, boolean indexIsFormula) throws HibernateException {
504 		final List deletes = new ArrayList();
505 		for ( Object o : ((Map) getSnapshot()).entrySet() ) {
506 			final Entry e = (Entry) o;
507 			final Object key = e.getKey();
508 			if ( e.getValue() != null && map.get( key ) == null ) {
509 				deletes.add( indexIsFormula ? e.getValue() : key );
510 			}
511 		}
512 		return deletes.iterator();
513 	}

525 	public boolean needsUpdating(Object entry, int i, Type elemType) throws HibernateException {
526 		final Map sn = (Map) getSnapshot();
527 		final Map.Entry e = (Map.Entry) entry;
528 		final Object snValue = sn.get( e.getKey() );
529 		return e.getValue() != null
530 				&& snValue != null
531 				&& elemType.isDirty( snValue, e.getValue(), getSession() );
532 	}
