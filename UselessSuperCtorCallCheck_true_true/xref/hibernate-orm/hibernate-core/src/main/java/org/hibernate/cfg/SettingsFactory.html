78  	public Settings buildSettings(Properties props, ServiceRegistry serviceRegistry) {
79  		final boolean debugEnabled =  LOG.isDebugEnabled();
80  		final JdbcServices jdbcServices = serviceRegistry.getService( JdbcServices.class );
81  		final StrategySelector strategySelector = serviceRegistry.getService( StrategySelector.class );
82  
83  		Settings settings = new Settings();
84  
85  		//SessionFactory name:
86  
87  		String sessionFactoryName = props.getProperty( AvailableSettings.SESSION_FACTORY_NAME );
88  		settings.setSessionFactoryName( sessionFactoryName );
89  		settings.setSessionFactoryNameAlsoJndiName(
90  				ConfigurationHelper.getBoolean( AvailableSettings.SESSION_FACTORY_NAME_IS_JNDI, props, true )
91  		);
92  
93  		//JDBC and connection settings:
94  
95  		//Interrogate JDBC metadata
96  		ExtractedDatabaseMetaData meta = jdbcServices.getExtractedMetaDataSupport();
97  
98  		settings.setDataDefinitionImplicitCommit( meta.doesDataDefinitionCauseTransactionCommit() );
99  		settings.setDataDefinitionInTransactionSupported( meta.supportsDataDefinitionInTransaction() );
100 
101 		//use dialect default properties
102 		final Properties properties = new Properties();
103 		properties.putAll( jdbcServices.getDialect().getDefaultProperties() );
104 		properties.putAll( props );
105 
106 		// Transaction settings:
107 		settings.setJtaPlatform( serviceRegistry.getService( JtaPlatform.class ) );
108 
109 		MultiTableBulkIdStrategy multiTableBulkIdStrategy = strategySelector.resolveStrategy(
110 				MultiTableBulkIdStrategy.class,
111 				properties.getProperty( AvailableSettings.HQL_BULK_ID_STRATEGY )
112 		);
113 		if ( multiTableBulkIdStrategy == null ) {
114 			multiTableBulkIdStrategy = jdbcServices.getDialect().supportsTemporaryTables()
115 					? TemporaryTableBulkIdStrategy.INSTANCE
116 					: new PersistentTableBulkIdStrategy();
117 		}
118 		settings.setMultiTableBulkIdStrategy( multiTableBulkIdStrategy );
119 
120 		boolean flushBeforeCompletion = ConfigurationHelper.getBoolean(AvailableSettings.FLUSH_BEFORE_COMPLETION, properties);
121 		if ( debugEnabled ) {
122 			LOG.debugf( "Automatic flush during beforeCompletion(): %s", enabledDisabled(flushBeforeCompletion) );
123 		}
124 		settings.setFlushBeforeCompletionEnabled(flushBeforeCompletion);
125 
126 		boolean autoCloseSession = ConfigurationHelper.getBoolean(AvailableSettings.AUTO_CLOSE_SESSION, properties);
127 		if ( debugEnabled ) {
128 			LOG.debugf( "Automatic session close at end of transaction: %s", enabledDisabled(autoCloseSession) );
129 		}
130 		settings.setAutoCloseSessionEnabled(autoCloseSession);
131 
132 		//JDBC and connection settings:
133 
134 		int batchSize = ConfigurationHelper.getInt(AvailableSettings.STATEMENT_BATCH_SIZE, properties, 0);
135 		if ( !meta.supportsBatchUpdates() ) {
136 			batchSize = 0;
137 		}
138 		if ( batchSize > 0 && debugEnabled ) {
139 			LOG.debugf( "JDBC batch size: %s", batchSize );
140 		}
141 		settings.setJdbcBatchSize(batchSize);
142 
143 		boolean jdbcBatchVersionedData = ConfigurationHelper.getBoolean(AvailableSettings.BATCH_VERSIONED_DATA, properties, false);
144 		if ( batchSize > 0 && debugEnabled ) {
145 			LOG.debugf( "JDBC batch updates for versioned data: %s", enabledDisabled(jdbcBatchVersionedData) );
146 		}
147 		settings.setJdbcBatchVersionedData(jdbcBatchVersionedData);
148 
149 		boolean useScrollableResultSets = ConfigurationHelper.getBoolean(
150 				AvailableSettings.USE_SCROLLABLE_RESULTSET,
151 				properties,
152 				meta.supportsScrollableResults()
153 		);
154 		if ( debugEnabled ) {
155 			LOG.debugf( "Scrollable result sets: %s", enabledDisabled(useScrollableResultSets) );
156 		}
157 		settings.setScrollableResultSetsEnabled(useScrollableResultSets);
158 
159 		boolean wrapResultSets = ConfigurationHelper.getBoolean(AvailableSettings.WRAP_RESULT_SETS, properties, false);
160 		if ( debugEnabled ) {
161 			LOG.debugf( "Wrap result sets: %s", enabledDisabled(wrapResultSets) );
162 		}
163 		settings.setWrapResultSetsEnabled(wrapResultSets);
164 
165 		boolean useGetGeneratedKeys = ConfigurationHelper.getBoolean(AvailableSettings.USE_GET_GENERATED_KEYS, properties, meta.supportsGetGeneratedKeys());
166 		if ( debugEnabled ) {
167 			LOG.debugf( "JDBC3 getGeneratedKeys(): %s", enabledDisabled(useGetGeneratedKeys) );
168 		}
169 		settings.setGetGeneratedKeysEnabled(useGetGeneratedKeys);
170 
171 		Integer statementFetchSize = ConfigurationHelper.getInteger(AvailableSettings.STATEMENT_FETCH_SIZE, properties);
172 		if ( statementFetchSize != null && debugEnabled ) {
173 			LOG.debugf( "JDBC result set fetch size: %s", statementFetchSize );
174 		}
175 		settings.setJdbcFetchSize(statementFetchSize);
176 
177 		MultiTenancyStrategy multiTenancyStrategy = MultiTenancyStrategy.determineMultiTenancyStrategy( properties );
178 		if ( debugEnabled ) {
179 			LOG.debugf( "multi-tenancy strategy : %s", multiTenancyStrategy );
180 		}
181 		settings.setMultiTenancyStrategy( multiTenancyStrategy );
182 
183 		String releaseModeName = ConfigurationHelper.getString( AvailableSettings.RELEASE_CONNECTIONS, properties, "auto" );
184 		if ( debugEnabled ) {
185 			LOG.debugf( "Connection release mode: %s", releaseModeName );
186 		}
187 		ConnectionReleaseMode releaseMode;
188 		if ( "auto".equals(releaseModeName) ) {
189 			releaseMode = serviceRegistry.getService( TransactionFactory.class ).getDefaultReleaseMode();
190 		}
191 		else {
192 			releaseMode = ConnectionReleaseMode.parse( releaseModeName );
193 			if ( releaseMode == ConnectionReleaseMode.AFTER_STATEMENT ) {
194 				// we need to make sure the underlying JDBC connection access supports aggressive release...
195 				boolean supportsAgrressiveRelease = multiTenancyStrategy.requiresMultiTenantConnectionProvider()
196 						? serviceRegistry.getService( MultiTenantConnectionProvider.class ).supportsAggressiveRelease()
197 						: serviceRegistry.getService( ConnectionProvider.class ).supportsAggressiveRelease();
198 				if ( ! supportsAgrressiveRelease ) {
199 					LOG.unsupportedAfterStatement();
200 					releaseMode = ConnectionReleaseMode.AFTER_TRANSACTION;
201 				}
202 			}
203 		}
204 		settings.setConnectionReleaseMode( releaseMode );
205 
206 		final BatchFetchStyle batchFetchStyle = BatchFetchStyle.interpret( properties.get( AvailableSettings.BATCH_FETCH_STYLE ) );
207 		LOG.debugf( "Using BatchFetchStyle : " + batchFetchStyle.name() );
208 		settings.setBatchFetchStyle( batchFetchStyle );
209 
210 
211 		//SQL Generation settings:
212 
213 		String defaultSchema = properties.getProperty( AvailableSettings.DEFAULT_SCHEMA );
214 		String defaultCatalog = properties.getProperty( AvailableSettings.DEFAULT_CATALOG );
215 		if ( defaultSchema != null && debugEnabled ) {
216 			LOG.debugf( "Default schema: %s", defaultSchema );
217 		}
218 		if ( defaultCatalog != null && debugEnabled ) {
219 			LOG.debugf( "Default catalog: %s", defaultCatalog );
220 		}
221 		settings.setDefaultSchemaName( defaultSchema );
222 		settings.setDefaultCatalogName( defaultCatalog );
223 
224 		Integer maxFetchDepth = ConfigurationHelper.getInteger( AvailableSettings.MAX_FETCH_DEPTH, properties );
225 		if ( maxFetchDepth != null ) {
226 			LOG.debugf( "Maximum outer join fetch depth: %s", maxFetchDepth );
227 		}
228 		settings.setMaximumFetchDepth( maxFetchDepth );
229 
230 		int batchFetchSize = ConfigurationHelper.getInt(AvailableSettings.DEFAULT_BATCH_FETCH_SIZE, properties, 1);
231 		if ( debugEnabled ) {
232 			LOG.debugf( "Default batch fetch size: %s", batchFetchSize );
233 		}
234 		settings.setDefaultBatchFetchSize( batchFetchSize );
235 
236 		boolean comments = ConfigurationHelper.getBoolean( AvailableSettings.USE_SQL_COMMENTS, properties );
237 		if ( debugEnabled ) {
238 			LOG.debugf( "Generate SQL with comments: %s", enabledDisabled(comments) );
239 		}
240 		settings.setCommentsEnabled( comments );
241 
242 		boolean orderUpdates = ConfigurationHelper.getBoolean( AvailableSettings.ORDER_UPDATES, properties );
243 		if ( debugEnabled ) {
244 			LOG.debugf( "Order SQL updates by primary key: %s", enabledDisabled(orderUpdates) );
245 		}
246 		settings.setOrderUpdatesEnabled( orderUpdates );
247 
248 		boolean orderInserts = ConfigurationHelper.getBoolean(AvailableSettings.ORDER_INSERTS, properties);
249 		if ( debugEnabled ) {
250 			LOG.debugf( "Order SQL inserts for batching: %s", enabledDisabled(orderInserts) );
251 		}
252 		settings.setOrderInsertsEnabled( orderInserts );
253 
254 		String defaultNullPrecedence = ConfigurationHelper.getString(
255 				AvailableSettings.DEFAULT_NULL_ORDERING, properties, "none", "first", "last"
256 		);
257 		if ( debugEnabled ) {
258 			LOG.debugf( "Default null ordering: %s", defaultNullPrecedence );
259 		}
260 		settings.setDefaultNullPrecedence( NullPrecedence.parse( defaultNullPrecedence ) );
261 
262 		//Query parser settings:
263 
264 		settings.setQueryTranslatorFactory( createQueryTranslatorFactory( properties, serviceRegistry ) );
265 
266 		Map querySubstitutions = ConfigurationHelper.toMap( AvailableSettings.QUERY_SUBSTITUTIONS, " ,=;:\n\t\r\f", properties );
267 		if ( debugEnabled ) {
268 			LOG.debugf( "Query language substitutions: %s", querySubstitutions );
269 		}
270 		settings.setQuerySubstitutions( querySubstitutions );
271 
272 		boolean jpaqlCompliance = ConfigurationHelper.getBoolean( AvailableSettings.JPAQL_STRICT_COMPLIANCE, properties, false );
273 		if ( debugEnabled ) {
274 			LOG.debugf( "JPA-QL strict compliance: %s", enabledDisabled(jpaqlCompliance) );
275 		}
276 		settings.setStrictJPAQLCompliance( jpaqlCompliance );
277 
278 		// Second-level / query cache:
279 
280 		boolean useSecondLevelCache = ConfigurationHelper.getBoolean( AvailableSettings.USE_SECOND_LEVEL_CACHE, properties, true );
281 		if ( debugEnabled ) {
282 			LOG.debugf( "Second-level cache: %s", enabledDisabled(useSecondLevelCache) );
283 		}
284 		settings.setSecondLevelCacheEnabled( useSecondLevelCache );
285 
286 		boolean useQueryCache = ConfigurationHelper.getBoolean(AvailableSettings.USE_QUERY_CACHE, properties);
287 		if ( debugEnabled ) {
288 			LOG.debugf( "Query cache: %s", enabledDisabled(useQueryCache) );
289 		}
290 		settings.setQueryCacheEnabled( useQueryCache );
291 		if (useQueryCache) {
292 			settings.setQueryCacheFactory( createQueryCacheFactory( properties, serviceRegistry ) );
293 		}
294 
295 		settings.setRegionFactory( serviceRegistry.getService( RegionFactory.class ) );
296 
297 		boolean useMinimalPuts = ConfigurationHelper.getBoolean(
298 				AvailableSettings.USE_MINIMAL_PUTS, properties, settings.getRegionFactory().isMinimalPutsEnabledByDefault()
299 		);
300 		if ( debugEnabled ) {
301 			LOG.debugf( "Optimize cache for minimal puts: %s", enabledDisabled(useMinimalPuts) );
302 		}
303 		settings.setMinimalPutsEnabled( useMinimalPuts );
304 
305 		String prefix = properties.getProperty( AvailableSettings.CACHE_REGION_PREFIX );
306 		if ( StringHelper.isEmpty(prefix) ) {
307 			prefix=null;
308 		}
309 		if ( prefix != null && debugEnabled ) {
310 			LOG.debugf( "Cache region prefix: %s", prefix );
311 		}
312 		settings.setCacheRegionPrefix( prefix );
313 
314 		boolean useStructuredCacheEntries = ConfigurationHelper.getBoolean( AvailableSettings.USE_STRUCTURED_CACHE, properties, false );
315 		if ( debugEnabled ) {
316 			LOG.debugf( "Structured second-level cache entries: %s", enabledDisabled(useStructuredCacheEntries) );
317 		}
318 		settings.setStructuredCacheEntriesEnabled( useStructuredCacheEntries );
319 
320 		boolean useDirectReferenceCacheEntries = ConfigurationHelper.getBoolean(
321 				AvailableSettings.USE_DIRECT_REFERENCE_CACHE_ENTRIES,
322 				properties,
323 				false
324 		);
325 		if ( debugEnabled ) {
326 			LOG.debugf( "Second-level cache direct-reference entries: %s", enabledDisabled(useDirectReferenceCacheEntries) );
327 		}
328 		settings.setDirectReferenceCacheEntriesEnabled( useDirectReferenceCacheEntries );
329 
330 		boolean autoEvictCollectionCache = ConfigurationHelper.getBoolean( AvailableSettings.AUTO_EVICT_COLLECTION_CACHE, properties, false);
331 		if ( debugEnabled ) {
332 			LOG.debugf( "Automatic eviction of collection cache: %s", enabledDisabled(autoEvictCollectionCache) );
333 		}
334 		settings.setAutoEvictCollectionCache( autoEvictCollectionCache );
335 
336 		//Statistics and logging:
337 
338 		boolean useStatistics = ConfigurationHelper.getBoolean( AvailableSettings.GENERATE_STATISTICS, properties );
339 		if ( debugEnabled ) {
340 			LOG.debugf( "Statistics: %s", enabledDisabled(useStatistics) );
341 		}
342 		settings.setStatisticsEnabled( useStatistics );
343 
344 		boolean useIdentifierRollback = ConfigurationHelper.getBoolean( AvailableSettings.USE_IDENTIFIER_ROLLBACK, properties );
345 		if ( debugEnabled ) {
346 			LOG.debugf( "Deleted entity synthetic identifier rollback: %s", enabledDisabled(useIdentifierRollback) );
347 		}
348 		settings.setIdentifierRollbackEnabled( useIdentifierRollback );
349 
350 		//Schema export:
351 
352 		String autoSchemaExport = properties.getProperty( AvailableSettings.HBM2DDL_AUTO );
353 		if ( "validate".equals(autoSchemaExport) ) {
354 			settings.setAutoValidateSchema( true );
355 		}
356 		else if ( "update".equals(autoSchemaExport) ) {
357 			settings.setAutoUpdateSchema( true );
358 		}
359 		else if ( "create".equals(autoSchemaExport) ) {
360 			settings.setAutoCreateSchema( true );
361 		}
362 		else if ( "create-drop".equals( autoSchemaExport ) ) {
363 			settings.setAutoCreateSchema( true );
364 			settings.setAutoDropSchema( true );
365 		}
366 		else if ( !StringHelper.isEmpty( autoSchemaExport ) ) {
367 			LOG.warn( "Unrecognized value for \"hibernate.hbm2ddl.auto\": " + autoSchemaExport );
368 		}
369 		settings.setImportFiles( properties.getProperty( AvailableSettings.HBM2DDL_IMPORT_FILES ) );
370 
371 		EntityMode defaultEntityMode = EntityMode.parse( properties.getProperty( AvailableSettings.DEFAULT_ENTITY_MODE ) );
372 		if ( debugEnabled ) {
373 			LOG.debugf( "Default entity-mode: %s", defaultEntityMode );
374 		}
375 		settings.setDefaultEntityMode( defaultEntityMode );
376 
377 		boolean namedQueryChecking = ConfigurationHelper.getBoolean( AvailableSettings.QUERY_STARTUP_CHECKING, properties, true );
378 		if ( debugEnabled ) {
379 			LOG.debugf( "Named query checking : %s", enabledDisabled(namedQueryChecking) );
380 		}
381 		settings.setNamedQueryStartupCheckingEnabled( namedQueryChecking );
382 
383 		boolean checkNullability = ConfigurationHelper.getBoolean(AvailableSettings.CHECK_NULLABILITY, properties, true);
384 		if ( debugEnabled ) {
385 			LOG.debugf( "Check Nullability in Core (should be disabled when Bean Validation is on): %s", enabledDisabled(checkNullability) );
386 		}
387 		settings.setCheckNullability(checkNullability);
388 
389 		// TODO: Does EntityTuplizerFactory really need to be configurable? revisit for HHH-6383
390 		settings.setEntityTuplizerFactory( new EntityTuplizerFactory() );
391 
392 //		String provider = properties.getProperty( AvailableSettings.BYTECODE_PROVIDER );
393 //		log.info( "Bytecode provider name : " + provider );
394 //		BytecodeProvider bytecodeProvider = buildBytecodeProvider( provider );
395 //		settings.setBytecodeProvider( bytecodeProvider );
396 
397 		boolean initializeLazyStateOutsideTransactionsEnabled = ConfigurationHelper.getBoolean(
398 				AvailableSettings.ENABLE_LAZY_LOAD_NO_TRANS,
399 				properties,
400 				false
401 		);
402 		if ( debugEnabled ) {
403 			LOG.debugf( "Allow initialization of lazy state outside session : : %s", enabledDisabled( initializeLazyStateOutsideTransactionsEnabled ) );
404 		}
405 		settings.setInitializeLazyStateOutsideTransactions( initializeLazyStateOutsideTransactionsEnabled );
406 
407 		boolean jtaTrackByThread = ConfigurationHelper.getBoolean(
408 				AvailableSettings.JTA_TRACK_BY_THREAD,
409 				properties,
410 				true
411 		);
412 		if ( debugEnabled ) {
413 			LOG.debugf( "JTA Track by Thread: %s", enabledDisabled(jtaTrackByThread) );
414 		}
415 		settings.setJtaTrackByThread( jtaTrackByThread );
416 
417 		final String autoSessionEventsListenerName = properties.getProperty( AvailableSettings.AUTO_SESSION_EVENTS_LISTENER );
418 		final Class<? extends SessionEventListener> autoSessionEventsListener = autoSessionEventsListenerName == null
419 				? null
420 				: strategySelector.selectStrategyImplementor( SessionEventListener.class, autoSessionEventsListenerName );
421 
422 		final boolean logSessionMetrics = ConfigurationHelper.getBoolean(
423 				AvailableSettings.LOG_SESSION_METRICS,
424 				properties,
425 				useStatistics
426 
427 		);
428 		settings.setBaselineSessionEventsListenerBuilder(
429 				new BaselineSessionEventsListenerBuilder( logSessionMetrics, autoSessionEventsListener )
430 		);
431 
432 		return settings;
433

446 	private static String enabledDisabled(boolean value) {
447 		return value ? "enabled" : "disabled";
448 	}

450 	protected QueryCacheFactory createQueryCacheFactory(Properties properties, ServiceRegistry serviceRegistry) {
451 		String queryCacheFactoryClassName = ConfigurationHelper.getString(
452 				AvailableSettings.QUERY_CACHE_FACTORY, properties, StandardQueryCacheFactory.class.getName()
453 		);
454 		LOG.debugf( "Query cache factory: %s", queryCacheFactoryClassName );
455 		try {
456 			return (QueryCacheFactory) serviceRegistry.getService( ClassLoaderService.class )
457 					.classForName( queryCacheFactoryClassName )
458 					.newInstance();
459 		}
460 		catch (Exception e) {
461 			throw new HibernateException( "could not instantiate QueryCacheFactory: " + queryCacheFactoryClassName, e );
462 		}
463 	}

465 	public static RegionFactory createRegionFactory(Properties properties, boolean cachingEnabled) {
466 		// todo : REMOVE!  THIS IS TOTALLY A TEMPORARY HACK FOR org.hibernate.cfg.AnnotationBinder which will be going away
467 		String regionFactoryClassName = RegionFactoryInitiator.mapLegacyNames(
468 				ConfigurationHelper.getString(
469 						AvailableSettings.CACHE_REGION_FACTORY, properties, null
470 				)
471 		);
472 		if ( regionFactoryClassName == null ) {
473 			regionFactoryClassName = DEF_CACHE_REG_FACTORY;
474 		}
475 		LOG.debugf( "Cache region factory : %s", regionFactoryClassName );
476 		try {
477 			try {
478 				return (RegionFactory) org.hibernate.internal.util.ReflectHelper.classForName( regionFactoryClassName )
479 						.getConstructor( Properties.class )
480 						.newInstance( properties );
481 			}
482 			catch ( NoSuchMethodException e ) {
483 				// no constructor accepting Properties found, try no arg constructor
484 				LOG.debugf(
485 						"%s did not provide constructor accepting java.util.Properties; attempting no-arg constructor.",
486 						regionFactoryClassName
487 				);
488 				return (RegionFactory) org.hibernate.internal.util.ReflectHelper.classForName( regionFactoryClassName )
489 						.newInstance();
490 			}
491 		}
492 		catch ( Exception e ) {
493 			throw new HibernateException( "could not instantiate RegionFactory [" + regionFactoryClassName + "]", e );
494 		}
495 	}

499 				AvailableSettings.QUERY_TRANSLATOR, properties, "org.hibernate.hql.internal.ast.ASTQueryTranslatorFactory"
500 		);
501 		LOG.debugf( "Query translator: %s", className );
502 		try {
503 			return (QueryTranslatorFactory) serviceRegistry.getService( ClassLoaderService.class )
504 					.classForName( className )
505 					.newInstance();
506 		}
507 		catch ( Exception e ) {
508 			throw new HibernateException( "could not instantiate QueryTranslatorFactory: " + className, e );
509 		}
