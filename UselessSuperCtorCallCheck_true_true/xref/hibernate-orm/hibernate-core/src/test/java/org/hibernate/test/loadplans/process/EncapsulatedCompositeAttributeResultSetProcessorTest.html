141 	private void doCompare(SessionFactoryImplementor sf, OuterJoinLoadable persister) {
142 		final LoadQueryInfluencers influencers = LoadQueryInfluencers.NONE;
143 		final LockMode lockMode = LockMode.NONE;
144 		final int batchSize = 1;
145 
146 		final EntityJoinWalker walker = new EntityJoinWalker(
147 				persister,
148 				persister.getKeyColumnNames(),
149 				batchSize,
150 				lockMode,
151 				sf,
152 				influencers
153 		);
154 
155 		final LoadQueryDetails details = Helper.INSTANCE.buildLoadQueryDetails( persister, sf );
156 
157 		compare( walker, details );
158 	}

221 	private List<?> getResults(EntityPersister entityPersister ) {
222 		final LoadPlan plan = Helper.INSTANCE.buildLoadPlan( sessionFactory(), entityPersister );
223 
224 		final LoadQueryDetails queryDetails = Helper.INSTANCE.buildLoadQueryDetails( plan, sessionFactory() );
225 		final String sql = queryDetails.getSqlStatement();
226 		final ResultSetProcessor resultSetProcessor = queryDetails.getResultSetProcessor();
227 
228 		final List results = new ArrayList();
229 
230 		final Session workSession = openSession();
231 		workSession.beginTransaction();
232 		workSession.doWork(
233 				new Work() {
234 					@Override
235 					public void execute(Connection connection) throws SQLException {
236 						PreparedStatement ps = connection.prepareStatement( sql );
237 						ps.setInt( 1, 1 );
238 						ResultSet resultSet = ps.executeQuery();
239 						results.addAll(
240 								resultSetProcessor.extractResults(
241 										resultSet,
242 										(SessionImplementor) workSession,
243 										new QueryParameters(),
244 										new NamedParameterContext() {
245 											@Override
246 											public int[] getNamedParameterLocations(String name) {
247 												return new int[0];
248 											}
249 										},
250 										true,
251 										false,
252 										null,
253 										null
254 								)
255 						);
256 						resultSet.close();
257 						ps.close();
258 					}
259 				}
260 		);
261 		workSession.getTransaction().commit();
262 		workSession.close();
263 		return results;
264 	}
