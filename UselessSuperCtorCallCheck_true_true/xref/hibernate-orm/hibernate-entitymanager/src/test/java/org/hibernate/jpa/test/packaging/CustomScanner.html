13  public class CustomScanner implements Scanner {
14  	public static boolean isUsed = false;
15  	private Scanner delegate = new StandardScanner();
16  
17  	public static boolean isUsed() {
18  		return isUsed;
19  	}
20  
21  	public static void resetUsed() {
22  		isUsed = false;
23  	}
24  
25  	@Override
26  	public ScanResult scan(
27  			ScanEnvironment environment,
28  			ScanOptions options,
29  			ScanParameters parameters) {
30  		isUsed = true;
31  		return delegate.scan( environment, options, parameters );
32  	}
33  }
