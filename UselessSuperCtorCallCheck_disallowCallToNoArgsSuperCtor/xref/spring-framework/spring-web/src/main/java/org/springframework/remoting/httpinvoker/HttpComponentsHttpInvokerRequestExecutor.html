39  /**
40   * {@link org.springframework.remoting.httpinvoker.HttpInvokerRequestExecutor} implementation that uses
41   * <a href="http://hc.apache.org/httpcomponents-client-ga/httpclient/">Apache HttpComponents HttpClient</a>
42   * to execute POST requests.
43   *
44   * <p>Allows to use a pre-configured {@link org.apache.http.client.HttpClient}
45   * instance, potentially with authentication, HTTP connection pooling, etc.
46   * Also designed for easy subclassing, providing specific template methods.
47   *
48   * <p>As of Spring 3.2, this request executor requires Apache HttpComponents 4.2 or higher.
49   *
50   * @author Juergen Hoeller
51   * @since 3.1
52   * @see org.springframework.remoting.httpinvoker.SimpleHttpInvokerRequestExecutor
53   */
54  @SuppressWarnings("deprecation")
55  public class HttpComponentsHttpInvokerRequestExecutor extends AbstractHttpInvokerRequestExecutor {
56  
57  	private static final int DEFAULT_MAX_TOTAL_CONNECTIONS = 100;
58  
59  	private static final int DEFAULT_MAX_CONNECTIONS_PER_ROUTE = 5;
60  
61  	private static final int DEFAULT_READ_TIMEOUT_MILLISECONDS = (60 * 1000);
62  
63  	private HttpClient httpClient;
64  
65  
66  	/**
67  	 * Create a new instance of the HttpComponentsHttpInvokerRequestExecutor with a default
68  	 * {@link HttpClient} that uses a default {@code org.apache.http.impl.conn.PoolingClientConnectionManager}.
69  	 */
70  	public HttpComponentsHttpInvokerRequestExecutor() {
71  		org.apache.http.conn.scheme.SchemeRegistry schemeRegistry = new org.apache.http.conn.scheme.SchemeRegistry();
72  		schemeRegistry.register(new org.apache.http.conn.scheme.Scheme("http", 80, org.apache.http.conn.scheme.PlainSocketFactory.getSocketFactory()));
73  		schemeRegistry.register(new org.apache.http.conn.scheme.Scheme("https", 443, org.apache.http.conn.ssl.SSLSocketFactory.getSocketFactory()));
74  
75  		org.apache.http.impl.conn.PoolingClientConnectionManager connectionManager =
76  				new org.apache.http.impl.conn.PoolingClientConnectionManager(schemeRegistry);
77  		connectionManager.setMaxTotal(DEFAULT_MAX_TOTAL_CONNECTIONS);
78  		connectionManager.setDefaultMaxPerRoute(DEFAULT_MAX_CONNECTIONS_PER_ROUTE);
79  
80  		this.httpClient = new org.apache.http.impl.client.DefaultHttpClient(connectionManager);
81  		setReadTimeout(DEFAULT_READ_TIMEOUT_MILLISECONDS);
82  	}

108 	/**
109 	 * Set the connection timeout for the underlying HttpClient.
110 	 * A timeout value of 0 specifies an infinite timeout.
111 	 * @param timeout the timeout value in milliseconds
112 	 */
113 	public void setConnectTimeout(int timeout) {
114 		Assert.isTrue(timeout >= 0, "Timeout must be a non-negative value");
115 		getHttpClient().getParams().setIntParameter(org.apache.http.params.CoreConnectionPNames.CONNECTION_TIMEOUT, timeout);
116 	}
117 
118 	/**
119 	 * Set the socket read timeout for the underlying HttpClient.
120 	 * A timeout value of 0 specifies an infinite timeout.
121 	 * @param timeout the timeout value in milliseconds
122 	 * @see #DEFAULT_READ_TIMEOUT_MILLISECONDS
123 	 */
124 	public void setReadTimeout(int timeout) {
125 		Assert.isTrue(timeout >= 0, "Timeout must be a non-negative value");
126 		getHttpClient().getParams().setIntParameter(org.apache.http.params.CoreConnectionPNames.SO_TIMEOUT, timeout);
127 	}

130 	/**
131 	 * Execute the given request through the HttpClient.
132 	 * <p>This method implements the basic processing workflow:
133 	 * The actual work happens in this class's template methods.
134 	 * @see #createHttpPost
135 	 * @see #setRequestBody
136 	 * @see #executeHttpPost
137 	 * @see #validateResponse
138 	 * @see #getResponseBody
139 	 */
140 	@Override
141 	protected RemoteInvocationResult doExecuteRequest(
142 			HttpInvokerClientConfiguration config, ByteArrayOutputStream baos)
143 			throws IOException, ClassNotFoundException {
144 
145 		HttpPost postMethod = createHttpPost(config);
146 		setRequestBody(config, postMethod, baos);
147 		try {
148 			HttpResponse response = executeHttpPost(config, getHttpClient(), postMethod);
149 			validateResponse(config, response);
150 			InputStream responseBody = getResponseBody(config, response);
151 			return readRemoteInvocationResult(responseBody, config.getCodebaseUrl());
152 		}
153 		finally {
154 			postMethod.releaseConnection();
155 		}
156 	}
157 
158 	/**
159 	 * Create a HttpPost for the given configuration.
160 	 * <p>The default implementation creates a standard HttpPost with
161 	 * "application/x-java-serialized-object" as "Content-Type" header.
162 	 * @param config the HTTP invoker configuration that specifies the
163 	 * target service
164 	 * @return the HttpPost instance
165 	 * @throws java.io.IOException if thrown by I/O methods
166 	 */
167 	protected HttpPost createHttpPost(HttpInvokerClientConfiguration config) throws IOException {
168 		HttpPost httpPost = new HttpPost(config.getServiceUrl());
169 		LocaleContext localeContext = LocaleContextHolder.getLocaleContext();
170 		if (localeContext != null) {
171 			Locale locale = localeContext.getLocale();
172 			if (locale != null) {
173 				httpPost.addHeader(HTTP_HEADER_ACCEPT_LANGUAGE, StringUtils.toLanguageTag(locale));
174 			}
175 		}
176 		if (isAcceptGzipEncoding()) {
177 			httpPost.addHeader(HTTP_HEADER_ACCEPT_ENCODING, ENCODING_GZIP);
178 		}
179 		return httpPost;
180 	}
181 
182 	/**
183 	 * Set the given serialized remote invocation as request body.
184 	 * <p>The default implementation simply sets the serialized invocation as the
185 	 * HttpPost's request body. This can be overridden, for example, to write a
186 	 * specific encoding and to potentially set appropriate HTTP request headers.
187 	 * @param config the HTTP invoker configuration that specifies the target service
188 	 * @param httpPost the HttpPost to set the request body on
189 	 * @param baos the ByteArrayOutputStream that contains the serialized
190 	 * RemoteInvocation object
191 	 * @throws java.io.IOException if thrown by I/O methods
192 	 */
193 	protected void setRequestBody(
194 			HttpInvokerClientConfiguration config, HttpPost httpPost, ByteArrayOutputStream baos)
195 			throws IOException {
196 
197 		ByteArrayEntity entity = new ByteArrayEntity(baos.toByteArray());
198 		entity.setContentType(getContentType());
199 		httpPost.setEntity(entity);
200 	}

202 	/**
203 	 * Execute the given HttpPost instance.
204 	 * @param config the HTTP invoker configuration that specifies the target service
205 	 * @param httpClient the HttpClient to execute on
206 	 * @param httpPost the HttpPost to execute
207 	 * @return the resulting HttpResponse
208 	 * @throws java.io.IOException if thrown by I/O methods
209 	 */
210 	protected HttpResponse executeHttpPost(
211 			HttpInvokerClientConfiguration config, HttpClient httpClient, HttpPost httpPost)
212 			throws IOException {
213 
214 		return httpClient.execute(httpPost);
215 	}
216 
217 	/**
218 	 * Validate the given response as contained in the HttpPost object,
219 	 * throwing an exception if it does not correspond to a successful HTTP response.
220 	 * <p>Default implementation rejects any HTTP status code beyond 2xx, to avoid
221 	 * parsing the response body and trying to deserialize from a corrupted stream.
222 	 * @param config the HTTP invoker configuration that specifies the target service
223 	 * @param response the resulting HttpResponse to validate
224 	 * @throws java.io.IOException if validation failed
225 	 */
226 	protected void validateResponse(HttpInvokerClientConfiguration config, HttpResponse response)
227 			throws IOException {
228 
229 		StatusLine status = response.getStatusLine();
230 		if (status.getStatusCode() >= 300) {
231 			throw new NoHttpResponseException(
232 					"Did not receive successful HTTP response: status code = " + status.getStatusCode() +
233 					", status message = [" + status.getReasonPhrase() + "]");
234 		}
235 	}
236 
237 	/**
238 	 * Extract the response body from the given executed remote invocation request.
239 	 * <p>The default implementation simply fetches the HttpPost's response body stream.
240 	 * If the response is recognized as GZIP response, the InputStream will get wrapped
241 	 * in a GZIPInputStream.
242 	 * @param config the HTTP invoker configuration that specifies the target service
243 	 * @param httpResponse the resulting HttpResponse to read the response body from
244 	 * @return an InputStream for the response body
245 	 * @throws java.io.IOException if thrown by I/O methods
246 	 * @see #isGzipResponse
247 	 * @see java.util.zip.GZIPInputStream
248 	 */
249 	protected InputStream getResponseBody(HttpInvokerClientConfiguration config, HttpResponse httpResponse)
250 			throws IOException {
251 
252 		if (isGzipResponse(httpResponse)) {
253 			return new GZIPInputStream(httpResponse.getEntity().getContent());
254 		}
255 		else {
256 			return httpResponse.getEntity().getContent();
257 		}
258 	}
259 
260 	/**
261 	 * Determine whether the given response indicates a GZIP response.
262 	 * <p>The default implementation checks whether the HTTP "Content-Encoding"
263 	 * header contains "gzip" (in any casing).
264 	 * @param httpResponse the resulting HttpResponse to check
265 	 * @return whether the given response indicates a GZIP response
266 	 */
267 	protected boolean isGzipResponse(HttpResponse httpResponse) {
268 		Header encodingHeader = httpResponse.getFirstHeader(HTTP_HEADER_CONTENT_ENCODING);
269 		return (encodingHeader != null && encodingHeader.getValue() != null &&
270 				encodingHeader.getValue().toLowerCase().contains(ENCODING_GZIP));
271 	}
